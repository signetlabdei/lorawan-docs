77 void
78 LoraFrameHeader::Serialize (Buffer::Iterator start) const
79 {
80  NS_LOG_FUNCTION_NOARGS ();
81 
82  // Device Address field
83  start.WriteU32 (m_address.Get ());
84 
85  // fCtrl field
86  uint8_t fCtrl = 0;
87  fCtrl |= uint8_t (m_adr << 6 & 0b1000000);
88  fCtrl |= uint8_t (m_adrAckReq << 5 & 0b100000);
89  fCtrl |= uint8_t (m_ack << 4 & 0b10000);
90  fCtrl |= uint8_t (m_fPending << 3 & 0b1000);
91  fCtrl |= m_fOptsLen & 0b111;
92  start.WriteU8 (fCtrl);
93 
94  // FCnt field
95  start.WriteU16 (m_fCnt);
96 
97  // FOpts field
98  for (auto it = m_macCommands.begin (); it != m_macCommands.end (); it++)
99  {
100  NS_LOG_DEBUG ("Serializing a MAC command");
101  (*it)->Serialize (start);
102  }
103 
104  // FPort
105  start.WriteU8 (m_fPort);
106 }

108 uint32_t
109 LoraFrameHeader::Deserialize (Buffer::Iterator start)
110 {
111  NS_LOG_FUNCTION_NOARGS ();
112 
113  // Empty the list of MAC commands
114  m_macCommands.clear ();
115 
116  // Read from buffer and save into local variables
117  m_address.Set (start.ReadU32 ());
118  uint8_t fCtl = start.ReadU8 ();
119  m_adr = (fCtl >> 6) & 0b1;
120  m_adrAckReq = (fCtl >> 5) & 0b1;
121  m_ack = (fCtl >> 4) & 0b1;
122  m_fPending = (fCtl >> 3) & 0b1;
123  m_fOptsLen = fCtl & 0b111;
124  m_fCnt = start.ReadU16 ();
125 
126  NS_LOG_DEBUG ("Deserialized data: ");
127  NS_LOG_DEBUG ("Address: " << m_address.Print ());
128  NS_LOG_DEBUG ("ADR: " << unsigned(m_adr));
129  NS_LOG_DEBUG ("ADRAckReq: " << unsigned (m_adrAckReq));
130  NS_LOG_DEBUG ("Ack: " << unsigned (m_ack));
131  NS_LOG_DEBUG ("fPending: " << unsigned (m_fPending));
132  NS_LOG_DEBUG ("fOptsLen: " << unsigned (m_fOptsLen));
133  NS_LOG_DEBUG ("fCnt: " << unsigned (m_fCnt));
134 
135  // Deserialize MAC commands
136  NS_LOG_DEBUG ("Starting deserialization of MAC commands");
137  for (uint8_t byteNumber = 0; byteNumber < m_fOptsLen;)
138  {
139  uint8_t cid = start.PeekU8 ();
140  NS_LOG_DEBUG ("CID: " << unsigned(cid));
141 
142  // Divide Uplink and Downlink messages
143  // This needs to be done because they have the same CID, and the context
144  // about where this message will be Serialized/Deserialized (i.e., at the
145  // ED or at the NS) is umportant.
146  if (m_isUplink)
147  {
148  switch (cid)
149  {
150  // In the case of Uplink messages, the NS will deserialize the
151  // request for a link check
152  case (0x02):
153  {
154  NS_LOG_DEBUG ("Creating a LinkCheckReq command");
155  Ptr<LinkCheckReq> command = Create <LinkCheckReq> ();
156  byteNumber += command->Deserialize (start);
157  m_macCommands.push_back (command);
158  break;
159  }
160  case (0x03):
161  {
162  NS_LOG_DEBUG ("Creating a LinkAdrAns command");
163  Ptr<LinkAdrAns> command = Create <LinkAdrAns> ();
164  byteNumber += command->Deserialize (start);
165  m_macCommands.push_back (command);
166  break;
167  }
168  case (0x04):
169  {
170  NS_LOG_DEBUG ("Creating a DutyCycleAns command");
171  Ptr<DutyCycleAns> command = Create <DutyCycleAns> ();
172  byteNumber += command->Deserialize (start);
173  m_macCommands.push_back (command);
174  break;
175  }
176  case (0x05):
177  {
178  NS_LOG_DEBUG ("Creating a RxParamSetupAns command");
179  Ptr<RxParamSetupAns> command = Create <RxParamSetupAns> ();
180  byteNumber += command->Deserialize (start);
181  m_macCommands.push_back (command);
182  break;
183  }
184  case (0x06):
185  {
186  NS_LOG_DEBUG ("Creating a DevStatusAns command");
187  Ptr<DevStatusAns> command = Create <DevStatusAns> ();
188  byteNumber += command->Deserialize (start);
189  m_macCommands.push_back (command);
190  break;
191  }
192  case (0x07):
193  {
194  NS_LOG_DEBUG ("Creating a NewChannelAns command");
195  Ptr<NewChannelAns> command = Create <NewChannelAns> ();
196  byteNumber += command->Deserialize (start);
197  m_macCommands.push_back (command);
198  break;
199  }
200  case (0x08):
201  {
202  NS_LOG_DEBUG ("Creating a RxTimingSetupAns command");
203  Ptr<RxTimingSetupAns> command = Create <RxTimingSetupAns> ();
204  byteNumber += command->Deserialize (start);
205  m_macCommands.push_back (command);
206  break;
207  }
208  case (0x09):
209  {
210  NS_LOG_DEBUG ("Creating a TxParamSetupAns command");
211  Ptr<TxParamSetupAns> command = Create <TxParamSetupAns> ();
212  byteNumber += command->Deserialize (start);
213  m_macCommands.push_back (command);
214  break;
215  }
216  case (0x0A):
217  {
218  NS_LOG_DEBUG ("Creating a DlChannelAns command");
219  Ptr<DlChannelAns> command = Create <DlChannelAns> ();
220  byteNumber += command->Deserialize (start);
221  m_macCommands.push_back (command);
222  break;
223  }
224  default:
225  {
226  NS_LOG_ERROR ("CID not recognized during deserialization");
227  }
228  }
229  }
230  else
231  {
232  switch (cid)
233  {
234  // In the case of Downlink messages, the ED will deserialize the
235  // answer to a link check
236  case (0x02):
237  {
238  NS_LOG_DEBUG ("Creating a LinkCheckAns command");
239  Ptr<LinkCheckAns> command = Create <LinkCheckAns> ();
240  byteNumber += command->Deserialize (start);
241  m_macCommands.push_back (command);
242  break;
243  }
244  case (0x03):
245  {
246  NS_LOG_DEBUG ("Creating a LinkAdrReq command");
247  Ptr<LinkAdrReq> command = Create <LinkAdrReq> ();
248  byteNumber += command->Deserialize (start);
249  m_macCommands.push_back (command);
250  break;
251  }
252  case (0x04):
253  {
254  NS_LOG_DEBUG ("Creating a DutyCycleReq command");
255  Ptr<DutyCycleReq> command = Create <DutyCycleReq> ();
256  byteNumber += command->Deserialize (start);
257  m_macCommands.push_back (command);
258  break;
259  }
260  case (0x05):
261  {
262  NS_LOG_DEBUG ("Creating a RxParamSetupReq command");
263  Ptr<RxParamSetupReq> command = Create <RxParamSetupReq> ();
264  byteNumber += command->Deserialize (start);
265  m_macCommands.push_back (command);
266  break;
267  }
268  case (0x06):
269  {
270  NS_LOG_DEBUG ("Creating a DevStatusReq command");
271  Ptr<DevStatusReq> command = Create <DevStatusReq> ();
272  byteNumber += command->Deserialize (start);
273  m_macCommands.push_back (command);
274  break;
275  }
276  case (0x07):
277  {
278  NS_LOG_DEBUG ("Creating a NewChannelReq command");
279  Ptr<NewChannelReq> command = Create <NewChannelReq> ();
280  byteNumber += command->Deserialize (start);
281  m_macCommands.push_back (command);
282  break;
283  }
284  case (0x08):
285  {
286  NS_LOG_DEBUG ("Creating a RxTimingSetupReq command");
287  Ptr<RxTimingSetupReq> command = Create <RxTimingSetupReq> ();
288  byteNumber += command->Deserialize (start);
289  m_macCommands.push_back (command);
290  break;
291  }
292  case (0x09):
293  {
294  NS_LOG_DEBUG ("Creating a TxParamSetupReq command");
295  Ptr<TxParamSetupReq> command = Create <TxParamSetupReq> ();
296  byteNumber += command->Deserialize (start);
297  m_macCommands.push_back (command);
298  break;
299  }
300  default:
301  {
302  NS_LOG_ERROR ("CID not recognized during deserialization");
303  }
304  }
305  }
306  }
307 
308  m_fPort = uint8_t (start.ReadU8 ());
309 
310  return 8 + m_fOptsLen; // the number of bytes consumed.
311 }

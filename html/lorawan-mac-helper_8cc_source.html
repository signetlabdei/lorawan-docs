73 Ptr<LorawanMac>
74 LorawanMacHelper::Create (Ptr<Node> node, Ptr<NetDevice> device) const
75 {
76  Ptr<LorawanMac> mac = m_mac.Create<LorawanMac> ();
77  mac->SetDevice (device);
78 
79  // If we are operating on an end device, add an address to it
80  if (m_deviceType == ED_A && m_addrGen != 0)
81  {
82  mac->GetObject<ClassAEndDeviceLorawanMac> ()->SetDeviceAddress (m_addrGen->NextAddress ());
83  }
84 
85  // Add a basic list of channels based on the region where the device is
86  // operating
87  if (m_deviceType == ED_A)
88  {
89  Ptr<ClassAEndDeviceLorawanMac> edMac = mac->GetObject<ClassAEndDeviceLorawanMac> ();
90  switch (m_region)
91  {
92  case LorawanMacHelper::EU:
93  {
94  ConfigureForEuRegion (edMac);
95  break;
96  }
97  case LorawanMacHelper::ALOHA:
98  {
99  ConfigureForAlohaRegion (edMac);
100  break;
101  }
102  default:
103  {
104  NS_LOG_ERROR ("This region isn't supported yet!");
105  break;
106  }
107  }
108  }
109  else
110  {
111  Ptr<GatewayLorawanMac> gwMac = mac->GetObject<GatewayLorawanMac> ();
112  switch (m_region)
113  {
114  case LorawanMacHelper::EU:
115  {
116  ConfigureForEuRegion (gwMac);
117  break;
118  }
119  case LorawanMacHelper::ALOHA:
120  {
121  ConfigureForAlohaRegion (gwMac);
122  break;
123  }
124  default:
125  {
126  NS_LOG_ERROR ("This region isn't supported yet!");
127  break;
128  }
129  }
130  }
131  return mac;
132 }

360  std::vector<int> sfQuantity (7, 0);
361  for (NodeContainer::Iterator j = endDevices.Begin (); j != endDevices.End (); ++j)
362  {
363  Ptr<Node> object = *j;
364  Ptr<MobilityModel> position = object->GetObject<MobilityModel> ();
365  NS_ASSERT (position != 0);
366  Ptr<NetDevice> netDevice = object->GetDevice (0);
367  Ptr<LoraNetDevice> loraNetDevice = netDevice->GetObject<LoraNetDevice> ();

370  NS_ASSERT (mac != 0);
371 
372  // Try computing the distance from each gateway and find the best one
373  Ptr<Node> bestGateway = gateways.Get (0);
374  Ptr<MobilityModel> bestGatewayPosition = bestGateway->GetObject<MobilityModel> ();
375 
376  // Assume devices transmit at 14 dBm
377  double highestRxPower = channel->GetRxPower (14, position, bestGatewayPosition);
378 
379  for (NodeContainer::Iterator currentGw = gateways.Begin () + 1; currentGw != gateways.End ();
380  ++currentGw)
381  {
382  // Compute the power received from the current gateway
383  Ptr<Node> curr = *currentGw;
384  Ptr<MobilityModel> currPosition = curr->GetObject<MobilityModel> ();
385  double currentRxPower = channel->GetRxPower (14, position, currPosition); // dBm
386 
387  if (currentRxPower > highestRxPower)

496 std::vector<int>
497 LorawanMacHelper::SetSpreadingFactorsGivenDistribution (NodeContainer endDevices,
498  NodeContainer gateways,
499  std::vector<double> distribution)
500 {
501  NS_LOG_FUNCTION_NOARGS ();
502 
503  std::vector<int> sfQuantity (7, 0);
504  Ptr<UniformRandomVariable> uniformRV = CreateObject<UniformRandomVariable> ();
505  std::vector<double> cumdistr (6);
506  cumdistr[0] = distribution[0];
507  for (int i = 1; i < 7; ++i)
508  {
509  cumdistr[i] = distribution[i] + cumdistr[i - 1];
510  }
511 
512  NS_LOG_DEBUG ("Distribution: " << distribution[0] << " " << distribution[1] << " "
513  << distribution[2] << " " << distribution[3] << " "
514  << distribution[4] << " " << distribution[5]);
515  NS_LOG_DEBUG ("Cumulative distribution: " << cumdistr[0] << " " << cumdistr[1] << " "
516  << cumdistr[2] << " " << cumdistr[3] << " "
517  << cumdistr[4] << " " << cumdistr[5]);
518 
519  for (NodeContainer::Iterator j = endDevices.Begin (); j != endDevices.End (); ++j)
520  {
521  Ptr<Node> object = *j;
522  Ptr<MobilityModel> position = object->GetObject<MobilityModel> ();
523  NS_ASSERT (position != 0);
524  Ptr<NetDevice> netDevice = object->GetDevice (0);
525  Ptr<LoraNetDevice> loraNetDevice = netDevice->GetObject<LoraNetDevice> ();
526  NS_ASSERT (loraNetDevice != 0);
527  Ptr<ClassAEndDeviceLorawanMac> mac = loraNetDevice->GetMac ()->GetObject<ClassAEndDeviceLorawanMac> ();
528  NS_ASSERT (mac != 0);
529 
530  double prob = uniformRV->GetValue (0, 1);
531 
532  // NS_LOG_DEBUG ("Probability: " << prob);
533  if (prob < cumdistr[0])
534  {
535  mac->SetDataRate (5);
536  sfQuantity[0] = sfQuantity[0] + 1;
537  }
538  else if (prob > cumdistr[0] && prob < cumdistr[1])
539  {
540  mac->SetDataRate (4);
541  sfQuantity[1] = sfQuantity[1] + 1;
542  }
543  else if (prob > cumdistr[1] && prob < cumdistr[2])
544  {
545  mac->SetDataRate (3);
546  sfQuantity[2] = sfQuantity[2] + 1;
547  }
548  else if (prob > cumdistr[2] && prob < cumdistr[3])
549  {
550  mac->SetDataRate (2);
551  sfQuantity[3] = sfQuantity[3] + 1;
552  }
553  else if (prob > cumdistr[3] && prob < cumdistr[4])
554  {
555  mac->SetDataRate (1);
556  sfQuantity[4] = sfQuantity[4] + 1;
557  }
558  else
559  {
560  mac->SetDataRate (0);
561  sfQuantity[5] = sfQuantity[5] + 1;
562  }
563 
564  } // end loop on nodes
565 
566  return sfQuantity;
567

99 RandomVariableStream::SetStream (int64_t stream)
100 {
101  NS_LOG_FUNCTION (this << stream);
102  // negative values are not legal.
103  NS_ASSERT (stream >= -1);
104  delete m_rng;
105  if (stream == -1)
106  {
107  // The first 2^63 streams are reserved for automatic stream
108  // number assignment.
109  uint64_t nextStream = RngSeedManager::GetNextStreamIndex ();
110  NS_ASSERT (nextStream <= ((1ULL) << 63));
111  m_rng = new RngStream (RngSeedManager::GetSeed (),
112  nextStream,
113  RngSeedManager::GetRun ());
114  }
115  else
116  {
117  // The last 2^63 streams are reserved for deterministic stream
118  // number assignment.
119  uint64_t base = ((1ULL) << 63);
120  uint64_t target = base + stream;
121  m_rng = new RngStream (RngSeedManager::GetSeed (),
122  target,
123  RngSeedManager::GetRun ());
124  }
125  m_stream = stream;
126 }

333 double
334 SequentialRandomVariable::GetValue (void)
335 {
336  // Set the current sequence value if it hasn't been set.
337  NS_LOG_FUNCTION (this);
338  if (!m_isCurrentSet)
339  {
340  // Start the sequence at its minimium value.
341  m_current = m_min;
342  m_isCurrentSet = true;
343  }
344 
345  // Return a sequential series of values
346  double r = m_current;
347  if (++m_currentConsecutive == m_consecutive)
348  { // Time to advance to next
349  m_currentConsecutive = 0;
350  m_current += m_increment->GetValue ();
351  if (m_current >= m_max)
352  {
353  m_current = m_min + (m_current - m_max);
354  }
355  }
356  return r;
357 }

715 double
716 NormalRandomVariable::GetValue (double mean, double variance, double bound)
717 {
718  NS_LOG_FUNCTION (this << mean << variance << bound);
719  if (m_nextValid)
720  { // use previously generated
721  m_nextValid = false;
722  return m_next;
723  }
724  while (1)
725  { // See Simulation Modeling and Analysis p. 466 (Averill Law)
726  // for algorithm; basically a Box-Muller transform:
727  // http://en.wikipedia.org/wiki/Box-Muller_transform
728  double u1 = Peek ()->RandU01 ();
729  double u2 = Peek ()->RandU01 ();
730  if (IsAntithetic ())
731  {
732  u1 = (1 - u1);
733  u2 = (1 - u2);
734  }
735  double v1 = 2 * u1 - 1;
736  double v2 = 2 * u2 - 1;
737  double w = v1 * v1 + v2 * v2;
738  if (w <= 1.0)
739  { // Got good pair
740  double y = std::sqrt ((-2 * std::log (w)) / w);
741  m_next = mean + v2 * y * std::sqrt (variance);
742  // if next is in bounds, it is valid
743  m_nextValid = std::fabs (m_next - mean) <= bound;
744  double x1 = mean + v1 * y * std::sqrt (variance);
745  // if x1 is in bounds, return it
746  if (std::fabs (x1 - mean) <= bound)
747  {
748  return x1;
749  }
750  // otherwise try and return m_next if it is valid
751  else if (m_nextValid)
752  {
753  m_nextValid = false;
754  return m_next;
755  }
756  // otherwise, just run this loop again
757  }
758  }

945  tools/ranvar.cc
946 
947  Originally the algorithm was devised by Marsaglia in 2000:
948  G. Marsaglia, W. W. Tsang: A simple method for generating Gamma variables
949  ACM Transactions on mathematical software, Vol. 26, No. 3, Sept. 2000
950 
951  The Gamma distribution density function has the form
952 
953  x^(alpha-1) * exp(-x/beta)
954  p(x; alpha, beta) = ----------------------------
955  beta^alpha * Gamma(alpha)
956 
957  for x > 0.
958 */
959 double
960 GammaRandomVariable::GetValue (double alpha, double beta)
961 {
962  NS_LOG_FUNCTION (this << alpha << beta);
963  if (alpha < 1)
964  {
965  double u = Peek ()->RandU01 ();
966  if (IsAntithetic ())
967  {
968  u = (1 - u);
969  }
970  return GetValue (1.0 + alpha, beta) * std::pow (u, 1.0 / alpha);
971  }
972 
973  double x, v, u;
974  double d = alpha - 1.0 / 3.0;
975  double c = (1.0 / 3.0) / std::sqrt (d);
976 
977  while (1)
978  {
979  do
980  {
981  // Get a value from a normal distribution that has mean
982  // zero, variance 1, and no bound.
983  double mean = 0.0;
984  double variance = 1.0;
985  double bound = NormalRandomVariable::INFINITE_VALUE;
986  x = GetNormalValue (mean, variance, bound);
987 
988  v = 1.0 + c * x;
989  }
990  while (v <= 0);
991 
992  v = v * v * v;
993  u = Peek ()->RandU01 ();
994  if (IsAntithetic ())
995  {
996  u = (1 - u);
997  }
998  if (u < 1 - 0.0331 * x * x * x * x)
999  {
1000  break;
1001  }
1002  if (std::log (u) < 0.5 * x * x + d * (1 - v + std::log (v)))
1003  {
1004  break;
1005  }
1006  }
1007

1009 }

1033 {
1034  NS_LOG_FUNCTION (this << mean << variance << bound);
1035  if (m_nextValid)
1036  { // use previously generated
1037  m_nextValid = false;
1038  return m_next;
1039  }
1040  while (1)
1041  { // See Simulation Modeling and Analysis p. 466 (Averill Law)
1042  // for algorithm; basically a Box-Muller transform:
1043  // http://en.wikipedia.org/wiki/Box-Muller_transform
1044  double u1 = Peek ()->RandU01 ();
1045  double u2 = Peek ()->RandU01 ();
1046  if (IsAntithetic ())
1047  {
1048  u1 = (1 - u1);
1049  u2 = (1 - u2);
1050  }
1051  double v1 = 2 * u1 - 1;
1052  double v2 = 2 * u2 - 1;
1053  double w = v1 * v1 + v2 * v2;
1054  if (w <= 1.0)
1055  { // Got good pair
1056  double y = std::sqrt ((-2 * std::log (w)) / w);
1057  m_next = mean + v2 * y * std::sqrt (variance);
1058  // if next is in bounds, it is valid
1059  m_nextValid = std::fabs (m_next - mean) <= bound;
1060  double x1 = mean + v1 * y * std::sqrt (variance);
1061  // if x1 is in bounds, return it
1062  if (std::fabs (x1 - mean) <= bound)
1063  {
1064  return x1;
1065  }
1066  // otherwise try and return m_next if it is valid
1067  else if (m_nextValid)
1068  {
1069  m_nextValid = false;
1070  return m_next;
1071  }
1072  // otherwise, just run this loop again
1073  }
1074  }

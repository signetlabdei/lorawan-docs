109 void
110 LoraDeviceAddress::Serialize (uint8_t buf[4]) const
111 {
112  NS_LOG_FUNCTION (this << &buf);
113 
114  uint32_t address = Get ();
115 
116  buf[0] = (address >> 24) & 0xff;
117  buf[1] = (address >> 16) & 0xff;
118  buf[2] = (address >> 8) & 0xff;
119  buf[3] = (address >> 0) & 0xff;
120 }

122 LoraDeviceAddress
123 LoraDeviceAddress::Deserialize (const uint8_t buf[4])
124 {
125  NS_LOG_FUNCTION (&buf);
126 
127  // Craft the address from the buffer
128  uint32_t address = 0;
129  address |= buf[0];
130  address <<= 8;
131  address |= buf[1];
132  address <<= 8;
133  address |= buf[2];
134  address <<= 8;
135  address |= buf[3];
136 
137  return LoraDeviceAddress (address);
138 }

175 LoraDeviceAddress::Get (void) const
176 {
177  NS_LOG_FUNCTION_NOARGS ();
178 
179  uint32_t address = 0;

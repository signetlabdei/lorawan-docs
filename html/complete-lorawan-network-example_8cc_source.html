70 void
71 CheckReceptionByAllGWsComplete (std::map<Ptr<Packet const>, PacketStatus>::iterator it)
72 {
73  // Check whether this packet is received by all gateways
74  if ((*it).second.outcomeNumber == nGateways)
75  {
76  // Update the statistics
77  PacketStatus status = (*it).second;
78  for (int j = 0; j < nGateways; j++)
79  {
80  switch (status.outcomes.at (j))
81  {
82  case RECEIVED:
83  {
84  received += 1;
85  break;
86  }
87  case UNDER_SENSITIVITY:
88  {
89  underSensitivity += 1;
90  break;
91  }
92  case NO_MORE_RECEIVERS:
93  {
94  noMoreReceivers += 1;
95  break;
96  }
97  case INTERFERED:
98  {
99  interfered += 1;
100  break;
101  }
102  case UNSET:
103  {
104  break;
105  }
106  }
107  }
108  // Remove the packet from the tracker
109  packetTracker.erase (it);
110  }
111 }

114 TransmissionCallback (Ptr<Packet const> packet, uint32_t systemId)
115 {
116  // NS_LOG_INFO ("Transmitted a packet from device " << systemId);
117  // Create a packetStatus
118  PacketStatus status;

121  status.outcomeNumber = 0;
122  status.outcomes = std::vector<enum PacketOutcome> (nGateways, UNSET);
123 
124  packetTracker.insert (std::pair<Ptr<Packet const>, PacketStatus> (packet, status));
125 }

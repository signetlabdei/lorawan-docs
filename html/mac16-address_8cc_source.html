66 Mac16Address::Mac16Address (const char *str)
67 {
68  NS_LOG_FUNCTION (this << str);
69  int i = 0;
70  while (*str != 0 && i < 2)
71  {
72  uint8_t byte = 0;
73  while (*str != ASCII_COLON && *str != 0)
74  {
75  byte <<= 4;
76  char low = AsciiToLowCase (*str);
77  if (low >= ASCII_a)
78  {
79  byte |= low - ASCII_a + 10;
80  }
81  else
82  {
83  byte |= low - ASCII_ZERO;
84  }
85  str++;
86  }
87  m_address[i] = byte;
88  i++;
89  if (*str == 0)
90  {
91  break;
92  }
93  str++;
94  }
95  NS_ASSERT (i == 2);
96 }

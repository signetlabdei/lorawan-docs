166 void OnOffApplication::StartApplication () // Called at time specified by Start
167 {
168  NS_LOG_FUNCTION (this);
169 
170  // Create the socket if not already
171  if (!m_socket)
172  {
173  m_socket = Socket::CreateSocket (GetNode (), m_tid);
174  int ret = -1;
175 
176  if (! m_local.IsInvalid())
177  {
178  NS_ABORT_MSG_IF ((Inet6SocketAddress::IsMatchingType (m_peer) && InetSocketAddress::IsMatchingType (m_local)) ||
179  (InetSocketAddress::IsMatchingType (m_peer) && Inet6SocketAddress::IsMatchingType (m_local)),
180  "Incompatible peer and local address IP version");
181  ret = m_socket->Bind (m_local);
182  }
183  else
184  {
185  if (Inet6SocketAddress::IsMatchingType (m_peer))
186  {
187  ret = m_socket->Bind6 ();
188  }
189  else if (InetSocketAddress::IsMatchingType (m_peer) ||
190  PacketSocketAddress::IsMatchingType (m_peer))
191  {
192  ret = m_socket->Bind ();
193  }
194  }
195 
196  if (ret == -1)
197  {
198  NS_FATAL_ERROR ("Failed to bind socket");
199  }
200 
201  m_socket->Connect (m_peer);
202  m_socket->SetAllowBroadcast (true);
203  m_socket->ShutdownRecv ();
204 
205  m_socket->SetConnectCallback (
206  MakeCallback (&OnOffApplication::ConnectionSucceeded, this),
207  MakeCallback (&OnOffApplication::ConnectionFailed, this));
208  }
209  m_cbrRateFailSafe = m_cbrRate;
210 
211  // Insure no pending event
212  CancelEvents ();
213  // If we are not yet connected, there is nothing to do here
214  // The ConnectionComplete upcall will start timers at that time
215  //if (!m_connected) return;
216  ScheduleStartEvent ();
217 }

234 void OnOffApplication::CancelEvents ()
235 {
236  NS_LOG_FUNCTION (this);
237 
238  if (m_sendEvent.IsRunning () && m_cbrRateFailSafe == m_cbrRate )
239  { // Cancel the pending send packet event
240  // Calculate residual bits since last packet sent
241  Time delta (Simulator::Now () - m_lastStartTime);
242  int64x64_t bits = delta.To (Time::S) * m_cbrRate.GetBitRate ();
243  m_residualBits += bits.GetHigh ();
244  }
245  m_cbrRateFailSafe = m_cbrRate;
246  Simulator::Cancel (m_sendEvent);
247  Simulator::Cancel (m_startStopEvent);
248 }

267 // Private helpers
268 void OnOffApplication::ScheduleNextTx ()
269 {
270  NS_LOG_FUNCTION (this);
271 
272  if (m_maxBytes == 0 || m_totBytes < m_maxBytes)
273  {
274  uint32_t bits = m_pktSize * 8 - m_residualBits;
275  NS_LOG_LOGIC ("bits = " << bits);
276  Time nextTime (Seconds (bits /
277  static_cast<double>(m_cbrRate.GetBitRate ()))); // Time till next packet
278  NS_LOG_LOGIC ("nextTime = " << nextTime);
279  m_sendEvent = Simulator::Schedule (nextTime,
280  &OnOffApplication::SendPacket, this);
281  }
282  else
283  { // All done, cancel any pending events
284  StopApplication ();
285  }
286 }

307 void OnOffApplication::SendPacket ()
308 {
309  NS_LOG_FUNCTION (this);
310 
311  NS_ASSERT (m_sendEvent.IsExpired ());
312 
313  Ptr<Packet> packet;
314  if (m_enableE2EStats)
315  {
316  Address from, to;
317  m_socket->GetSockName (from);
318  m_socket->GetPeerName (to);
319  E2eStatsHeader header;
320  header.SetSeq (m_seq++);
321  header.SetSize (m_pktSize);
322  NS_ABORT_IF (m_pktSize < header.GetSerializedSize ());
323  packet = Create<Packet> (m_pktSize - header.GetSerializedSize ());
324  packet->AddHeader (header);
325  m_txTraceWithStats (packet, from, to, header);
326  }
327  else
328  {
329  packet = Create<Packet> (m_pktSize);
330  }
331 
332  m_txTrace (packet);
333  m_socket->Send (packet);
334  m_totBytes += m_pktSize;
335  Address localAddress;
336  m_socket->GetSockName (localAddress);
337  if (InetSocketAddress::IsMatchingType (m_peer))
338  {
339  NS_LOG_INFO ("At time " << Simulator::Now ().GetSeconds ()
340  << "s on-off application sent "
341  << packet->GetSize () << " bytes to "
342  << InetSocketAddress::ConvertFrom(m_peer).GetIpv4 ()
343  << " port " << InetSocketAddress::ConvertFrom (m_peer).GetPort ()
344  << " total Tx " << m_totBytes << " bytes");
345  m_txTraceWithAddresses (packet, localAddress, InetSocketAddress::ConvertFrom (m_peer));
346  }
347  else if (Inet6SocketAddress::IsMatchingType (m_peer))
348  {
349  NS_LOG_INFO ("At time " << Simulator::Now ().GetSeconds ()
350  << "s on-off application sent "
351  << packet->GetSize () << " bytes to "
352  << Inet6SocketAddress::ConvertFrom(m_peer).GetIpv6 ()
353  << " port " << Inet6SocketAddress::ConvertFrom (m_peer).GetPort ()
354  << " total Tx " << m_totBytes << " bytes");
355  m_txTraceWithAddresses (packet, localAddress, Inet6SocketAddress::ConvertFrom(m_peer));
356  }
357  m_lastStartTime = Simulator::Now ();
358  m_residualBits = 0;
359  ScheduleNextTx ();
360 }

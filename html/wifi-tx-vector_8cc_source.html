209 bool
210 WifiTxVector::IsValid (void) const
211 {
212  if (!GetModeInitialized ())
213  {
214  return false;
215  }
216  std::string modeName = m_mode.GetUniqueName ();
217  if (m_channelWidth == 20)
218  {
219  if (m_nss != 3 && m_nss != 6)
220  {
221  return (modeName != "VhtMcs9");
222  }
223  }
224  else if (m_channelWidth == 80)
225  {
226  if (m_nss == 3 || m_nss == 7)
227  {
228  return (modeName != "VhtMcs6");
229  }
230  else if (m_nss == 6)
231  {
232  return (modeName != "VhtMcs9");
233  }
234  }
235  else if (m_channelWidth == 160)
236  {
237  if (m_nss == 3)
238  {
239  return (modeName != "VhtMcs9");
240  }
241  }
242  return true;
243 }

59 static void
60 DoBeamforming (Ptr<NetDevice> thisDevice, Ptr<ThreeGppAntennaArrayModel> thisAntenna, Ptr<NetDevice> otherDevice)
61 {
62  ThreeGppAntennaArrayModel::ComplexVector antennaWeights;
63 
64  // retrieve the position of the two devices
65  Vector aPos = thisDevice->GetNode ()->GetObject<MobilityModel> ()->GetPosition ();
66  Vector bPos = otherDevice->GetNode ()->GetObject<MobilityModel> ()->GetPosition ();
67 
68  // compute the azimuth and the elevation angles
69  Angles completeAngle (bPos,aPos);
70 
71  double posX = bPos.x - aPos.x;
72  double phiAngle = atan ((bPos.y - aPos.y) / posX);
73 
74  if (posX < 0)
75  {
76  phiAngle = phiAngle + M_PI;
77  }
78  if (phiAngle < 0)
79  {
80  phiAngle = phiAngle + 2 * M_PI;
81  }
82 
83  double hAngleRadian = fmod ((phiAngle + M_PI),2 * M_PI - M_PI); // the azimuth angle
84  double vAngleRadian = completeAngle.theta; // the elevation angle
85 
86  // retrieve the number of antenna elements
87  int totNoArrayElements = thisAntenna->GetNumberOfElements ();
88 
89  // the total power is divided equally among the antenna elements
90  double power = 1 / sqrt (totNoArrayElements);
91 
92  // compute the antenna weights
93  for (int ind = 0; ind < totNoArrayElements; ind++)
94  {
95  Vector loc = thisAntenna->GetElementLocation (ind);
96  double phase = -2 * M_PI * (sin (vAngleRadian) * cos (hAngleRadian) * loc.x
97  + sin (vAngleRadian) * sin (hAngleRadian) * loc.y
98  + cos (vAngleRadian) * loc.z);
99  antennaWeights.push_back (exp (std::complex<double> (0, phase)) * power);
100  }
101 
102  // store the antenna weights
103  thisAntenna->SetBeamformingVector (antennaWeights);
104 }

113 static void
114 ComputeSnr (Ptr<MobilityModel> txMob, Ptr<MobilityModel> rxMob, double txPow, double noiseFigure)
115 {
116  // Create the tx PSD using the LteSpectrumValueHelper
117  // 100 RBs corresponds to 18 MHz (1 RB = 180 kHz)
118  // EARFCN 100 corresponds to 2125.00 MHz
119  std::vector<int> activeRbs0 (100);
120  for (int i = 0; i < 100 ; i++)
121  {
122  activeRbs0[i] = i;
123  }
124  Ptr<SpectrumValue> txPsd = LteSpectrumValueHelper::CreateTxPowerSpectralDensity (2100, 100, txPow, activeRbs0);
125  Ptr<SpectrumValue> rxPsd = txPsd->Copy ();
126  NS_LOG_DEBUG ("Average tx power " << 10*log10(Sum (*txPsd) * 180e3) << " dB");
127 
128  // create the noise PSD
129  Ptr<SpectrumValue> noisePsd = LteSpectrumValueHelper::CreateNoisePowerSpectralDensity (2100, 100, noiseFigure);
130  NS_LOG_DEBUG ("Average noise power " << 10*log10 (Sum (*noisePsd) * 180e3) << " dB");
131 
132  // apply the pathloss
133  double propagationGainDb = m_propagationLossModel->CalcRxPower (0, txMob, rxMob);
134  NS_LOG_DEBUG ("Pathloss " << -propagationGainDb << " dB");
135  double propagationGainLinear = std::pow (10.0, (propagationGainDb) / 10.0);
136  *(rxPsd) *= propagationGainLinear;
137 
138  // apply the fast fading and the beamforming gain
139  rxPsd = m_spectrumLossModel->CalcRxPowerSpectralDensity (rxPsd, txMob, rxMob);
140  NS_LOG_DEBUG ("Average rx power " << 10*log10 (Sum (*rxPsd) * 180e3) << " dB");
141 
142  // compute the SNR
143  NS_LOG_DEBUG ("Average SNR " << 10 * log10 (Sum (*rxPsd) / Sum (*noisePsd)) << " dB");
144 
145  // print the SNR and pathloss values in the snr-trace.txt file
146  std::ofstream f;
147  f.open ("snr-trace.txt", std::ios::out | std::ios::app);
148  f << Simulator::Now ().GetSeconds () << " " << 10 * log10 (Sum (*rxPsd) / Sum (*noisePsd)) << " " << propagationGainDb << std::endl;
149  f.close ();
150 }

152 int
153 main (int argc, char *argv[])
154 {
155  double frequency = 2125.0e6; // operating frequency in Hz (corresponds to EARFCN 2100)
156  double txPow = 49.0; // tx power in dBm
157  double noiseFigure = 9.0; // noise figure in dB
158  double distance = 10.0; // distance between tx and rx nodes in meters
159  uint32_t simTime = 10000; // simulation time in milliseconds
160  uint32_t timeRes = 10; // time resolution in milliseconds
161  std::string scenario = "UMa"; // 3GPP propagation scenario
162 
163  Config::SetDefault ("ns3::ThreeGppChannelModel::UpdatePeriod", TimeValue(MilliSeconds (1))); // update the channel at each iteration
164  Config::SetDefault ("ns3::ThreeGppChannelConditionModel::UpdatePeriod", TimeValue(MilliSeconds (0.0))); // do not update the channel condition
165 
166  RngSeedManager::SetSeed(1);
167  RngSeedManager::SetRun(1);
168 
169  // create and configure the factories for the channel condition and propagation loss models
170  ObjectFactory propagationLossModelFactory;
171  ObjectFactory channelConditionModelFactory;
172  if (scenario == "RMa")
173  {
174  propagationLossModelFactory.SetTypeId (ThreeGppRmaPropagationLossModel::GetTypeId ());
175  channelConditionModelFactory.SetTypeId (ThreeGppRmaChannelConditionModel::GetTypeId ());
176  }
177  else if (scenario == "UMa")
178  {
179  propagationLossModelFactory.SetTypeId (ThreeGppUmaPropagationLossModel::GetTypeId ());
180  channelConditionModelFactory.SetTypeId (ThreeGppUmaChannelConditionModel::GetTypeId ());
181  }
182  else if (scenario == "UMi-StreetCanyon")
183  {
184  propagationLossModelFactory.SetTypeId (ThreeGppUmiStreetCanyonPropagationLossModel::GetTypeId ());
185  channelConditionModelFactory.SetTypeId (ThreeGppUmiStreetCanyonChannelConditionModel::GetTypeId ());
186  }
187  else if (scenario == "InH-OfficeOpen")
188  {
189  propagationLossModelFactory.SetTypeId (ThreeGppIndoorOfficePropagationLossModel::GetTypeId ());
190  channelConditionModelFactory.SetTypeId (ThreeGppIndoorOpenOfficeChannelConditionModel::GetTypeId ());
191  }
192  else if (scenario == "InH-OfficeMixed")
193  {
194  propagationLossModelFactory.SetTypeId (ThreeGppIndoorOfficePropagationLossModel::GetTypeId ());
195  channelConditionModelFactory.SetTypeId (ThreeGppIndoorMixedOfficeChannelConditionModel::GetTypeId ());
196  }
197  else
198  {
199  NS_FATAL_ERROR ("Unknown scenario");
200  }
201 
202  // create the propagation loss model
203  m_propagationLossModel = propagationLossModelFactory.Create<ThreeGppPropagationLossModel> ();
204  m_propagationLossModel->SetAttribute ("Frequency", DoubleValue (frequency));
205  m_propagationLossModel->SetAttribute ("ShadowingEnabled", BooleanValue (false));
206 
207  // create the spectrum propagation loss model
208  m_spectrumLossModel = CreateObject<ThreeGppSpectrumPropagationLossModel> ();
209  m_spectrumLossModel->SetChannelModelAttribute ("Frequency", DoubleValue (frequency));
210  m_spectrumLossModel->SetChannelModelAttribute ("Scenario", StringValue (scenario));
211 
212  // create the channel condition model and associate it with the spectrum and
213  // propagation loss model
214  Ptr<ChannelConditionModel> condModel = channelConditionModelFactory.Create<ThreeGppChannelConditionModel> ();
215  m_spectrumLossModel->SetChannelModelAttribute ("ChannelConditionModel", PointerValue (condModel));
216  m_propagationLossModel->SetChannelConditionModel (condModel);
217 
218  // create the tx and rx nodes
219  NodeContainer nodes;
220  nodes.Create (2);
221 
222  // create the tx and rx devices
223  Ptr<SimpleNetDevice> txDev = CreateObject<SimpleNetDevice> ();
224  Ptr<SimpleNetDevice> rxDev = CreateObject<SimpleNetDevice> ();
225 
226  // associate the nodes and the devices
227  nodes.Get (0)->AddDevice (txDev);
228  txDev->SetNode (nodes.Get (0));
229  nodes.Get (1)->AddDevice (rxDev);
230  rxDev->SetNode (nodes.Get (1));
231 
232  // create the tx and rx mobility models, set the positions
233  Ptr<MobilityModel> txMob = CreateObject<ConstantPositionMobilityModel> ();
234  txMob->SetPosition (Vector (0.0,0.0,10.0));
235  Ptr<MobilityModel> rxMob = CreateObject<ConstantPositionMobilityModel> ();
236  rxMob->SetPosition (Vector (distance,0.0,1.6));
237 
238  // assign the mobility models to the nodes
239  nodes.Get (0)->AggregateObject (txMob);
240  nodes.Get (1)->AggregateObject (rxMob);
241 
242  // create the antenna objects and set their dimensions
243  Ptr<ThreeGppAntennaArrayModel> txAntenna = CreateObjectWithAttributes<ThreeGppAntennaArrayModel> ("NumColumns", UintegerValue (2), "NumRows", UintegerValue (2));
244  Ptr<ThreeGppAntennaArrayModel> rxAntenna = CreateObjectWithAttributes<ThreeGppAntennaArrayModel> ("NumColumns", UintegerValue (2), "NumRows", UintegerValue (2));
245 
246  // initialize the devices in the ThreeGppSpectrumPropagationLossModel
247  m_spectrumLossModel->AddDevice (txDev, txAntenna);
248  m_spectrumLossModel->AddDevice (rxDev, rxAntenna);
249 
250  // set the beamforming vectors
251  DoBeamforming (txDev, txAntenna, rxDev);
252  DoBeamforming (rxDev, rxAntenna, txDev);
253 
254  for (int i = 0; i < floor (simTime / timeRes); i++)
255  {
256  Simulator::Schedule (MilliSeconds (timeRes*i), &ComputeSnr, txMob, rxMob, txPow, noiseFigure);
257  }
258 
259  Simulator::Run ();
260  Simulator::Destroy ();
261  return 0;
262 }

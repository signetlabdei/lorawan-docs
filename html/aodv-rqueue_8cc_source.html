138 RequestQueue::Purge ()
139 {
140  IsExpired pred;
141  for (std::vector<QueueEntry>::iterator i = m_queue.begin (); i
142  != m_queue.end (); ++i)
143  {
144  if (pred (*i))
145  {
146  Drop (*i, "Drop outdated packet ");
147  }
148  }
149  m_queue.erase (std::remove_if (m_queue.begin (), m_queue.end (), pred),
150  m_queue.end ());

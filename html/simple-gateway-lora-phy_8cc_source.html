58 void
59 SimpleGatewayLoraPhy::Send (Ptr<Packet> packet, LoraTxParameters txParams,
60  double frequencyMHz, double txPowerDbm)
61 {
62  NS_LOG_FUNCTION (this << packet << frequencyMHz << txPowerDbm);
63 
64  // Get the time a packet with these parameters will take to be transmitted
65  Time duration = GetOnAirTime (packet, txParams);
66 
67  NS_LOG_DEBUG ("Duration of packet: " << duration << ", SF" <<
68  unsigned(txParams.sf));
69 
70  // Interrupt all receive operations
71  std::list<Ptr<SimpleGatewayLoraPhy::ReceptionPath> >::iterator it;
72  for (it = m_receptionPaths.begin (); it != m_receptionPaths.end (); ++it)
73  {
74 
75  Ptr<SimpleGatewayLoraPhy::ReceptionPath> currentPath = *it;
76 
77  if (!currentPath->IsAvailable ()) // Reception path is occupied
78  {
79  // Call the callback for reception interrupted by transmission
80  // Fire the trace source
81  if (m_device)
82  {
83  m_noReceptionBecauseTransmitting (currentPath->GetEvent ()->GetPacket (),
84  m_device->GetNode ()->GetId ());
85 
86  }
87  else
88  {
89  m_noReceptionBecauseTransmitting (currentPath->GetEvent ()->GetPacket (), 0);
90  }
91 
92  // Cancel the scheduled EndReceive call
93  Simulator::Cancel (currentPath->GetEndReceive ());
94 
95  // Free it
96  // This also resets all parameters like packet and endReceive call
97  currentPath->Free ();
98  }
99  }
100 
101  // Send the packet in the channel
102  m_channel->Send (this, packet, txPowerDbm, txParams, duration, frequencyMHz);
103 
104  Simulator::Schedule (duration, &SimpleGatewayLoraPhy::TxFinished, this, packet);
105 
106  m_isTransmitting = true;
107 
108  // Fire the trace source
109  if (m_device)
110  {
111  m_startSending (packet, m_device->GetNode ()->GetId ());
112  }
113  else
114  {
115  m_startSending (packet, 0);
116  }
117 }

119 void
120 SimpleGatewayLoraPhy::StartReceive (Ptr<Packet> packet, double rxPowerDbm,
121  uint8_t sf, Time duration, double frequencyMHz)
122 {
123  NS_LOG_FUNCTION (this << packet << rxPowerDbm << duration << frequencyMHz);
124 
125  // Fire the trace source
126  m_phyRxBeginTrace (packet);
127 
128  if (m_isTransmitting)
129  {
130  // If we get to this point, there are no demodulators we can use
131  NS_LOG_INFO ("Dropping packet reception of packet with sf = "
132  << unsigned(sf) <<
133  " because we are in TX mode");
134 
135  m_phyRxEndTrace (packet);
136 
137  // Fire the trace source
138  if (m_device)
139  {
140  m_noReceptionBecauseTransmitting (packet, m_device->GetNode ()->GetId ());
141  }
142  else
143  {
144  m_noReceptionBecauseTransmitting (packet, 0);
145  }
146 
147  return;
148  }
149 
150  // Add the event to the LoraInterferenceHelper
151  Ptr<LoraInterferenceHelper::Event> event;
152  event = m_interference.Add (duration, rxPowerDbm, sf, packet, frequencyMHz);
153 
154  // Cycle over the receive paths to check availability to receive the packet
155  std::list<Ptr<SimpleGatewayLoraPhy::ReceptionPath> >::iterator it;
156 
157  for (it = m_receptionPaths.begin (); it != m_receptionPaths.end (); ++it)
158  {
159  Ptr<SimpleGatewayLoraPhy::ReceptionPath> currentPath = *it;
160 
161  NS_LOG_DEBUG ("Current ReceptionPath is centered on frequency = " <<
162  currentPath->GetFrequency ());
163 
164  // If the receive path is available and listening on the channel of
165  // interest, we have a candidate
166  if (currentPath->GetFrequency () == frequencyMHz
167  && currentPath->IsAvailable ())
168  {
169  // See whether the reception power is above or below the sensitivity
170  // for that spreading factor
171  double sensitivity = SimpleGatewayLoraPhy::sensitivity[unsigned(sf) - 7];
172 
173  if (rxPowerDbm < sensitivity) // Packet arrived below sensitivity
174  {
175  NS_LOG_INFO ("Dropping packet reception of packet with sf = "
176  << unsigned(sf) <<
177  " because under the sensitivity of "
178  << sensitivity << " dBm");
179 
180  if (m_device)
181  {
182  m_underSensitivity (packet, m_device->GetNode ()->GetId ());
183  }
184  else
185  {
186  m_underSensitivity (packet, 0);
187  }
188 
189  // Since the packet is below sensitivity, it makes no sense to
190  // search for another ReceivePath
191  return;
192  }
193  else // We have sufficient sensitivity to start receiving
194  {
195  NS_LOG_INFO ("Scheduling reception of a packet, " <<
196  "occupying one demodulator");
197 
198  // Block this resource
199  currentPath->LockOnEvent (event);
200  m_occupiedReceptionPaths++;
201 
202  // Schedule the end of the reception of the packet
203  EventId endReceiveEventId = Simulator::Schedule (duration,
204  &LoraPhy::EndReceive,
205  this, packet,
206  event);
207 
208  currentPath->SetEndReceive (endReceiveEventId);
209 
210  // Make sure we don't go on searching for other ReceivePaths
211  return;
212  }
213  }
214  }
215  // If we get to this point, there are no demodulators we can use
216  NS_LOG_INFO ("Dropping packet reception of packet with sf = "
217  << unsigned(sf) <<
218  " because no suitable demodulator was found");
219 
220  // Fire the trace source
221  if (m_device)
222  {
223  m_noMoreDemodulators (packet, m_device->GetNode ()->GetId ());
224  }
225  else
226  {
227  m_noMoreDemodulators (packet, 0);
228  }
229 }

231 void
232 SimpleGatewayLoraPhy::EndReceive (Ptr<Packet> packet,
233  Ptr<LoraInterferenceHelper::Event> event)
234 {
235  NS_LOG_FUNCTION (this << packet << *event);
236 
237  // Call the trace source
238  m_phyRxEndTrace (packet);
239 
240  // Call the LoraInterferenceHelper to determine whether there was
241  // destructive interference. If the packet is correctly received, this
242  // method returns a 0.
243  uint8_t packetDestroyed = 0;
244  packetDestroyed = m_interference.IsDestroyedByInterference (event);
245 
246  // Check whether the packet was destroyed
247  if (packetDestroyed != uint8_t (0))
248  {
249  NS_LOG_DEBUG ("packetDestroyed by " << unsigned(packetDestroyed));
250 
251  // Update the packet's LoraTag
252  LoraTag tag;
253  packet->RemovePacketTag (tag);
254  tag.SetDestroyedBy (packetDestroyed);
255  packet->AddPacketTag (tag);
256 
257  // Fire the trace source
258  if (m_device)
259  {
260  m_interferedPacket (packet, m_device->GetNode ()->GetId ());
261  }
262  else
263  {
264  m_interferedPacket (packet, 0);
265  }
266  }
267  else // Reception was correct
268  {
269  NS_LOG_INFO ("Packet with SF " <<
270  unsigned(event->GetSpreadingFactor ()) <<
271  " received correctly");
272 
273  // Fire the trace source
274  if (m_device)
275  {
276  m_successfullyReceivedPacket (packet, m_device->GetNode ()->GetId ());
277  }
278  else
279  {
280  m_successfullyReceivedPacket (packet, 0);
281  }
282 
283  // Forward the packet to the upper layer
284  if (!m_rxOkCallback.IsNull ())
285  {
286  // Make a copy of the packet
287  // Ptr<Packet> packetCopy = packet->Copy ();
288 
289  // Set the receive power and frequency of this packet in the LoraTag: this
290  // information can be useful for upper layers trying to control link
291  // quality.
292  LoraTag tag;
293  packet->RemovePacketTag (tag);
294  tag.SetReceivePower (event->GetRxPowerdBm ());
295  tag.SetFrequency (event->GetFrequency ());
296  packet->AddPacketTag (tag);
297 
298  m_rxOkCallback (packet);
299  }
300 
301  }
302 
303  // Search for the demodulator that was locked on this event to free it.
304 
305  std::list< Ptr< SimpleGatewayLoraPhy::ReceptionPath > >::iterator it;
306 
307  for (it = m_receptionPaths.begin (); it != m_receptionPaths.end (); ++it)
308  {
309  Ptr<SimpleGatewayLoraPhy::ReceptionPath> currentPath = *it;
310 
311  if (currentPath->GetEvent () == event)
312  {
313  currentPath->Free ();
314  m_occupiedReceptionPaths--;
315  return;
316  }
317  }
318 }
319 
320 }

61 void
62 LoraMacHeader::Serialize (Buffer::Iterator start) const
63 {
64  NS_LOG_FUNCTION_NOARGS ();
65 
66  // The header we need to fill
67  uint8_t header = 0;
68 
69  // The MType
70  header |= m_mtype << 5;
71 
72  // Do nothing for the bits that are RFU
73 
74  // The major version bits
75  header |= m_major;
76 
77  // Write the byte
78  start.WriteU8 (header);
79 
80  NS_LOG_DEBUG ("Serialization of MAC header: " << std::bitset<8>(header));
81 }

83 uint32_t
84 LoraMacHeader::Deserialize (Buffer::Iterator start)
85 {
86  NS_LOG_FUNCTION_NOARGS ();
87 
88  // Save the byte on a temporary variable
89  uint8_t byte;
90  byte = start.ReadU8 ();
91 
92  // Get the 2 least significant bits to have the Major
93  m_major = byte & 0b11;
94 
95  // Move the three most significant bits to the least significant positions
96  // to get the MType
97  m_mtype = byte >> 5;
98 
99  return 1; // the number of bytes consumed.
100 }

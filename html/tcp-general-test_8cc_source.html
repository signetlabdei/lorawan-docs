284 Ptr<TcpSocketMsgBase>
285 TcpGeneralTest::CreateSocket (Ptr<Node> node, TypeId socketType,
286  TypeId congControl, TypeId recoveryAlgorithm)
287 {
288  ObjectFactory rttFactory;
289  ObjectFactory congestionAlgorithmFactory;
290  ObjectFactory recoveryAlgorithmFactory;
291  ObjectFactory socketFactory;
292 
293  rttFactory.SetTypeId (RttMeanDeviation::GetTypeId ());
294  congestionAlgorithmFactory.SetTypeId (congControl);
295  recoveryAlgorithmFactory.SetTypeId (recoveryAlgorithm);
296  socketFactory.SetTypeId (socketType);
297 
298  Ptr<RttEstimator> rtt = rttFactory.Create<RttEstimator> ();
299  Ptr<TcpSocketMsgBase> socket = DynamicCast<TcpSocketMsgBase> (socketFactory.Create ());
300  Ptr<TcpCongestionOps> algo = congestionAlgorithmFactory.Create<TcpCongestionOps> ();
301  Ptr<TcpRecoveryOps> recovery = recoveryAlgorithmFactory.Create<TcpRecoveryOps> ();
302 
303  socket->SetNode (node);
304  socket->SetTcp (node->GetObject<TcpL4Protocol> ());
305  socket->SetRtt (rtt);
306  socket->SetCongestionControlAlgorithm (algo);
307  socket->SetRecoveryAlgorithm (recovery);
308  return socket;
309 }

1104 /*
1105  * Send empty packet, copied/pasted from TcpSocketBase
1106  *
1107  * The rationale for copying/pasting is that we need to edit a little the
1108  * code inside. Since there isn't a well-defined division of duties,
1109  * we are forced to do this.
1110  */
1111 void
1112 TcpSocketSmallAcks::SendEmptyPacket (uint8_t flags)
1113 {
1114  Ptr<Packet> p = Create<Packet> ();
1115  TcpHeader header;
1116  SequenceNumber32 s = m_tcb->m_nextTxSequence;
1117 
1118  /*
1119  * Add tags for each socket option.
1120  * Note that currently the socket adds both IPv4 tag and IPv6 tag
1121  * if both options are set. Once the packet got to layer three, only
1122  * the corresponding tags will be read.
1123  */
1124  if (GetIpTos ())
1125  {
1126  SocketIpTosTag ipTosTag;
1127  ipTosTag.SetTos (GetIpTos ());
1128  p->AddPacketTag (ipTosTag);
1129  }
1130 
1131  if (IsManualIpv6Tclass ())
1132  {
1133  SocketIpv6TclassTag ipTclassTag;
1134  ipTclassTag.SetTclass (GetIpv6Tclass ());
1135  p->AddPacketTag (ipTclassTag);
1136  }
1137 
1138  if (IsManualIpTtl ())
1139  {
1140  SocketIpTtlTag ipTtlTag;
1141  ipTtlTag.SetTtl (GetIpTtl ());
1142  p->AddPacketTag (ipTtlTag);
1143  }
1144 
1145  if (IsManualIpv6HopLimit ())
1146  {
1147  SocketIpv6HopLimitTag ipHopLimitTag;
1148  ipHopLimitTag.SetHopLimit (GetIpv6HopLimit ());
1149  p->AddPacketTag (ipHopLimitTag);
1150  }
1151 
1152  if (m_endPoint == nullptr && m_endPoint6 == nullptr)
1153  {
1154  NS_LOG_WARN ("Failed to send empty packet due to null endpoint");
1155  return;
1156  }
1157  if (flags & TcpHeader::FIN)
1158  {
1159  flags |= TcpHeader::ACK;
1160  }
1161  else if (m_state == FIN_WAIT_1 || m_state == LAST_ACK || m_state == CLOSING)
1162  {
1163  ++s;
1164  }
1165 
1166  bool hasSyn = flags & TcpHeader::SYN;
1167  bool hasFin = flags & TcpHeader::FIN;
1168  bool isAck = flags == TcpHeader::ACK;
1169 
1170  header.SetFlags (flags);
1171  header.SetSequenceNumber (s);
1172 
1173  // Actual division in small acks.
1174  if (hasSyn || hasFin)
1175  {
1176  header.SetAckNumber (m_tcb->m_rxBuffer->NextRxSequence ());
1177  }
1178  else
1179  {
1180  SequenceNumber32 ackSeq;
1181 
1182  ackSeq = m_lastAckedSeq + m_bytesToAck;
1183 
1184  if (m_bytesLeftToBeAcked == 0 && m_tcb->m_rxBuffer->NextRxSequence () > m_lastAckedSeq)
1185  {
1186  m_bytesLeftToBeAcked = m_tcb->m_rxBuffer->NextRxSequence ().GetValue () - 1 - m_bytesToAck;
1187  }
1188  else if (m_bytesLeftToBeAcked > 0 && m_tcb->m_rxBuffer->NextRxSequence () > m_lastAckedSeq)
1189  {
1190  m_bytesLeftToBeAcked -= m_bytesToAck;
1191  }
1192 
1193  NS_LOG_LOGIC ("Acking up to " << ackSeq << " remaining bytes: " << m_bytesLeftToBeAcked);
1194 
1195  header.SetAckNumber (ackSeq);
1196  m_lastAckedSeq = ackSeq;
1197  }
1198 
1199  // end of division in small acks
1200 
1201  if (m_endPoint != nullptr)
1202  {
1203  header.SetSourcePort (m_endPoint->GetLocalPort ());
1204  header.SetDestinationPort (m_endPoint->GetPeerPort ());
1205  }
1206  else
1207  {
1208  header.SetSourcePort (m_endPoint6->GetLocalPort ());
1209  header.SetDestinationPort (m_endPoint6->GetPeerPort ());
1210  }
1211  AddOptions (header);
1212  header.SetWindowSize (AdvertisedWindowSize ());
1213 
1214  // RFC 6298, clause 2.4
1215  m_rto = Max (m_rtt->GetEstimate () + Max (m_clockGranularity, m_rtt->GetVariation () * 4), m_minRto);
1216 
1217  if (hasSyn)
1218  {
1219  if (m_synCount == 0)
1220  { // No more connection retries, give up
1221  NS_LOG_LOGIC ("Connection failed.");
1222  m_rtt->Reset (); //According to recommendation -> RFC 6298
1223  CloseAndNotify ();
1224  return;
1225  }
1226  else
1227  { // Exponential backoff of connection time out
1228  int backoffCount = 0x1 << (m_synRetries - m_synCount);
1229  m_rto = m_cnTimeout * backoffCount;
1230  m_synCount--;
1231  }
1232  }
1233  if (m_endPoint != nullptr)
1234  {
1235  m_tcp->SendPacket (p, header, m_endPoint->GetLocalAddress (),
1236  m_endPoint->GetPeerAddress (), m_boundnetdevice);
1237  }
1238  else
1239  {
1240  m_tcp->SendPacket (p, header, m_endPoint6->GetLocalAddress (),
1241  m_endPoint6->GetPeerAddress (), m_boundnetdevice);
1242  }
1243 
1244  m_txTrace (p, header, this);
1245 
1246  if (flags & TcpHeader::ACK)
1247  { // If sending an ACK, cancel the delay ACK as well
1248  m_delAckEvent.Cancel ();
1249  m_delAckCount = 0;
1250  }
1251  if (m_retxEvent.IsExpired () && (hasSyn || hasFin) && !isAck )
1252  { // Retransmit SYN / SYN+ACK / FIN / FIN+ACK to guard against lost
1253  NS_LOG_LOGIC ("Schedule retransmission timeout at time "
1254  << Simulator::Now ().GetSeconds () << " to expire at time "
1255  << (Simulator::Now () + m_rto.Get ()).GetSeconds ());
1256  m_retxEvent = Simulator::Schedule (m_rto, &TcpSocketSmallAcks::SendEmptyPacket, this, flags);
1257  }
1258 
1259  // send another ACK if bytes remain
1260  if (m_bytesLeftToBeAcked > 0 && m_tcb->m_rxBuffer->NextRxSequence () > m_lastAckedSeq)
1261  {
1262  SendEmptyPacket (flags);
1263  }
1264 }

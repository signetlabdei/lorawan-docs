53 FriisSpectrumPropagationLossModel::DoCalcRxPowerSpectralDensity (Ptr<const SpectrumValue> txPsd,

56 {

60 
61  NS_ASSERT (a);

63 
64  double d = a->GetDistanceFrom (b);
65 
66  while (vit != rxPsd->ValuesEnd ())

77 double
78 FriisSpectrumPropagationLossModel::CalculateLoss (double f, double d) const
79 {
80  NS_ASSERT (d >= 0);
81 
82  if (d == 0)
83  {
84  return 1;
85  }
86 
87  NS_ASSERT (f > 0);
88  double loss_sqrt = (4 * M_PI * f * d) / 3e8;
89  double loss = loss_sqrt * loss_sqrt;
90 
91  if (loss < 1)
92  {
93  loss = 1;
94  }
95  return loss;
96 }

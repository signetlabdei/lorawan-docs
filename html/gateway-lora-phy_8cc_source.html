186 bool
187 GatewayLoraPhy::IsOnFrequency (double frequencyMHz)
188 {
189  NS_LOG_FUNCTION (this << frequencyMHz);
190 
191  // Search every demodulator to see whether there's one listening on this
192  // frequency.
193  std::list< Ptr< GatewayLoraPhy::ReceptionPath > >::iterator it;
194 
195  for (it = m_receptionPaths.begin (); it != m_receptionPaths.end (); ++it)
196  {
197  Ptr<GatewayLoraPhy::ReceptionPath> currentPath = *it;
198 
199  NS_LOG_DEBUG ("Current reception path is on frequency " <<
200  currentPath->GetFrequency ());
201 
202  if (currentPath->GetFrequency () == frequencyMHz)
203  {
204  return true;
205  }
206  }
207  return false;
208 }
209 }

111 void
112 UdpClient::StartApplication (void)
113 {
114  NS_LOG_FUNCTION (this);
115 
116  if (m_socket == 0)
117  {
118  TypeId tid = TypeId::LookupByName ("ns3::UdpSocketFactory");
119  m_socket = Socket::CreateSocket (GetNode (), tid);
120  if (Ipv4Address::IsMatchingType(m_peerAddress) == true)
121  {
122  if (m_socket->Bind () == -1)
123  {
124  NS_FATAL_ERROR ("Failed to bind socket");
125  }
126  m_socket->Connect (InetSocketAddress (Ipv4Address::ConvertFrom(m_peerAddress), m_peerPort));
127  }
128  else if (Ipv6Address::IsMatchingType(m_peerAddress) == true)
129  {
130  if (m_socket->Bind6 () == -1)
131  {
132  NS_FATAL_ERROR ("Failed to bind socket");
133  }
134  m_socket->Connect (Inet6SocketAddress (Ipv6Address::ConvertFrom(m_peerAddress), m_peerPort));
135  }
136  else if (InetSocketAddress::IsMatchingType (m_peerAddress) == true)
137  {
138  if (m_socket->Bind () == -1)
139  {
140  NS_FATAL_ERROR ("Failed to bind socket");
141  }
142  m_socket->Connect (m_peerAddress);
143  }
144  else if (Inet6SocketAddress::IsMatchingType (m_peerAddress) == true)
145  {
146  if (m_socket->Bind6 () == -1)
147  {
148  NS_FATAL_ERROR ("Failed to bind socket");
149  }
150  m_socket->Connect (m_peerAddress);
151  }
152  else
153  {
154  NS_ASSERT_MSG (false, "Incompatible address type: " << m_peerAddress);
155  }
156  }
157 
158  m_socket->SetRecvCallback (MakeNullCallback<void, Ptr<Socket> > ());
159  m_socket->SetAllowBroadcast (true);
160  m_sendEvent = Simulator::Schedule (Seconds (0.0), &UdpClient::Send, this);
161 }

170 void
171 UdpClient::Send (void)
172 {
173  NS_LOG_FUNCTION (this);
174  NS_ASSERT (m_sendEvent.IsExpired ());
175  SeqTsHeader seqTs;
176  seqTs.SetSeq (m_sent);
177  Ptr<Packet> p = Create<Packet> (m_size-(8+4)); // 8+4 : the size of the seqTs header
178  p->AddHeader (seqTs);
179 
180  std::stringstream peerAddressStringStream;
181  if (Ipv4Address::IsMatchingType (m_peerAddress))
182  {
183  peerAddressStringStream << Ipv4Address::ConvertFrom (m_peerAddress);
184  }
185  else if (Ipv6Address::IsMatchingType (m_peerAddress))
186  {
187  peerAddressStringStream << Ipv6Address::ConvertFrom (m_peerAddress);
188  }
189 
190  if ((m_socket->Send (p)) >= 0)
191  {
192  ++m_sent;
193  NS_LOG_INFO ("TraceDelay TX " << m_size << " bytes to "
194  << peerAddressStringStream.str () << " Uid: "
195  << p->GetUid () << " Time: "
196  << (Simulator::Now ()).GetSeconds ());
197 
198  }
199  else
200  {
201  NS_LOG_INFO ("Error while sending " << m_size << " bytes to "
202  << peerAddressStringStream.str ());
203  }
204 
205  if (m_sent < m_count)
206  {
207  m_sendEvent = Simulator::Schedule (m_interval, &UdpClient::Send, this);
208  }
209 }

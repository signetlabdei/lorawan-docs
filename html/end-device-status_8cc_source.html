109 Ptr<Packet>
110 EndDeviceStatus::GetCompleteReplyPacket (void)
111 {
112  NS_LOG_FUNCTION_NOARGS ();
113 
114  // Start from reply payload
115  Ptr<Packet> replyPacket;
116  if (m_reply.payload) // If it has APP data to send
117  {
118  NS_LOG_DEBUG ("Crafting reply packet from existing payload");
119  replyPacket = m_reply.payload->Copy ();
120  }
121  else // If no APP data needs to be sent, use an empty payload
122  {
123  NS_LOG_DEBUG ("Crafting reply packet using an empty payload");
124  replyPacket = Create<Packet> (0);
125  }
126 
127  // Add headers
128  m_reply.frameHeader.SetAddress (m_endDeviceAddress);
129  Ptr<Packet> lastPacket = GetLastPacketReceivedFromDevice ()->Copy ();
130  LorawanMacHeader mHdr;
131  LoraFrameHeader fHdr;
132  fHdr.SetAsUplink ();
133  lastPacket->RemoveHeader (mHdr);
134  lastPacket->RemoveHeader (fHdr);
135  m_reply.frameHeader.SetFCnt (fHdr.GetFCnt ());
136  m_reply.macHeader.SetMType (LorawanMacHeader::UNCONFIRMED_DATA_DOWN);
137  replyPacket->AddHeader (m_reply.frameHeader);
138  replyPacket->AddHeader (m_reply.macHeader);
139 
140  NS_LOG_DEBUG ("Added MAC header" << m_reply.macHeader);
141  NS_LOG_DEBUG ("Added frame header" << m_reply.frameHeader);
142 
143  return replyPacket;
144 }

241 void
242 EndDeviceStatus::InsertReceivedPacket (Ptr<Packet const> receivedPacket, const Address &gwAddress)
243 {
244  NS_LOG_FUNCTION_NOARGS ();
245 
246  // Create a copy of the packet
247  Ptr<Packet> myPacket = receivedPacket->Copy ();
248 
249  // Extract the headers
250  LorawanMacHeader macHdr;
251  myPacket->RemoveHeader (macHdr);
252 
253  LoraFrameHeader frameHdr;
254  frameHdr.SetAsUplink ();
255  myPacket->RemoveHeader (frameHdr);
256 
257  // Update current parameters
258  LoraTag tag;
259  myPacket->RemovePacketTag (tag);
260  SetFirstReceiveWindowSpreadingFactor (tag.GetSpreadingFactor ());
261  SetFirstReceiveWindowFrequency (tag.GetFrequency ());
262 
263  // Update Information on the received packet
264  ReceivedPacketInfo info;
265  info.sf = tag.GetSpreadingFactor ();
266  info.frequency = tag.GetFrequency ();
267  info.packet = receivedPacket;
268 
269  double rcvPower = tag.GetReceivePower ();
270 
271  // Perform insertion in list, also checking that the packet isn't already in
272  // the list (it could have been received by another GW already)
273 
274  // Start searching from the end
275  auto it = m_receivedPacketList.rbegin ();
276  for (; it != m_receivedPacketList.rend (); it++)
277  {
278  // Get the frame counter of the current packet to compare it with the
279  // newly received one
280  Ptr<Packet> packetCopy = ((*it).first)->Copy ();
281  LorawanMacHeader currentMacHdr;
282  packetCopy->RemoveHeader (currentMacHdr);
283  LoraFrameHeader currentFrameHdr;
284  frameHdr.SetAsUplink ();
285  packetCopy->RemoveHeader (currentFrameHdr);
286 
287  NS_LOG_DEBUG ("Received packet's frame counter: " << unsigned(frameHdr.GetFCnt ())
288  << "\nCurrent packet's frame counter: "
289  << unsigned(currentFrameHdr.GetFCnt ()));
290 
291  if (frameHdr.GetFCnt () == currentFrameHdr.GetFCnt ())
292  {
293  NS_LOG_INFO ("Packet was already received by another gateway");
294 
295  // This packet had already been received from another gateway:
296  // add this gateway's reception information.
297  GatewayList &gwList = it->second.gwList;
298 
299  PacketInfoPerGw gwInfo;
300  gwInfo.receivedTime = Simulator::Now ();
301  gwInfo.rxPower = rcvPower;
302  gwInfo.gwAddress = gwAddress;
303  gwList.insert (std::pair<Address, PacketInfoPerGw> (gwAddress, gwInfo));
304 
305  NS_LOG_DEBUG ("Size of gateway list: " << gwList.size ());
306 
307  break; // Exit from the cycle
308  }
309  }
310  if (it == m_receivedPacketList.rend ())
311  {
312  NS_LOG_INFO ("Packet was received for the first time");
313  PacketInfoPerGw gwInfo;
314  gwInfo.receivedTime = Simulator::Now ();
315  gwInfo.rxPower = rcvPower;
316  gwInfo.gwAddress = gwAddress;
317  info.gwList.insert (std::pair<Address, PacketInfoPerGw> (gwAddress, gwInfo));
318  m_receivedPacketList.push_back (
319  std::pair<Ptr<Packet const>, ReceivedPacketInfo> (receivedPacket, info));
320  }
321  NS_LOG_DEBUG (*this);
322 }

368 std::map<double, Address>
369 EndDeviceStatus::GetPowerGatewayMap (void)
370 {
371  // Create a map of the gateways
372  // Key: received power
373  // Value: address of the corresponding gateway
374  ReceivedPacketInfo info = m_receivedPacketList.back ().second;
375  GatewayList gwList = info.gwList;
376 
377  std::map<double, Address> gatewayPowers;
378 
379  for (auto it = gwList.begin (); it != gwList.end (); it++)
380  {
381  Address currentGwAddress = (*it).first;
382  double currentRxPower = (*it).second.rxPower;
383  gatewayPowers.insert (std::pair<double, Address> (currentRxPower, currentGwAddress));
384  }
385 
386  return gatewayPowers;
387 }

88 uint8_t Ipv6Extension::ProcessOptions (Ptr<Packet>& packet,
89  uint8_t offset,
90  uint8_t length,
91  Ipv6Header const& ipv6Header,
92  Ipv6Address dst,
93  uint8_t *nextHeader,
94  bool& stopProcessing,
95  bool& isDropped,
96  Ipv6L3Protocol::DropReason& dropReason)
97 {
98  NS_LOG_FUNCTION (this << packet << offset << length << ipv6Header << dst << nextHeader << isDropped);
99 
100  // For ICMPv6 Error packets
101  Ptr<Packet> malformedPacket = packet->Copy ();
102  malformedPacket->AddHeader (ipv6Header);
103  Ptr<Icmpv6L4Protocol> icmpv6 = GetNode ()->GetObject<Ipv6L3Protocol> ()->GetIcmpv6 ();
104 
105  Ptr<Packet> p = packet->Copy ();
106  p->RemoveAtStart (offset);
107 
108  Ptr<Ipv6OptionDemux> ipv6OptionDemux = GetNode ()->GetObject<Ipv6OptionDemux> ();
109  Ptr<Ipv6Option> ipv6Option;
110 
111  uint8_t processedSize = 0;
112  uint32_t size = p->GetSize ();
113  uint8_t *data = new uint8_t[size];
114  p->CopyData (data, size);
115 
116  uint8_t optionType = 0;
117  uint8_t optionLength = 0;
118 
119  while (length > processedSize && !isDropped)
120  {
121  optionType = *(data + processedSize);
122  ipv6Option = ipv6OptionDemux->GetOption (optionType);
123 
124  if (ipv6Option == 0)
125  {
126  optionType >>= 6;
127  switch (optionType)
128  {
129  case 0:
130  optionLength = *(data + processedSize + 1) + 2;
131  break;
132 
133  case 1:
134  NS_LOG_LOGIC ("Unknown Option. Drop!");
135  optionLength = 0;
136  isDropped = true;
137  stopProcessing = true;
138  dropReason = Ipv6L3Protocol::DROP_UNKNOWN_OPTION;
139  break;
140 
141  case 2:
142  NS_LOG_LOGIC ("Unknown Option. Drop!");
143  icmpv6->SendErrorParameterError (malformedPacket, ipv6Header.GetSourceAddress (), Icmpv6Header::ICMPV6_UNKNOWN_OPTION, offset + processedSize);
144  optionLength = 0;
145  isDropped = true;
146  stopProcessing = true;
147  dropReason = Ipv6L3Protocol::DROP_UNKNOWN_OPTION;
148  break;
149 
150  case 3:
151  NS_LOG_LOGIC ("Unknown Option. Drop!");
152 
153  if (!ipv6Header.GetDestinationAddress ().IsMulticast ())
154  {
155  icmpv6->SendErrorParameterError (malformedPacket, ipv6Header.GetSourceAddress (), Icmpv6Header::ICMPV6_UNKNOWN_OPTION, offset + processedSize);
156  }
157  optionLength = 0;
158  isDropped = true;
159  stopProcessing = true;
160  dropReason = Ipv6L3Protocol::DROP_UNKNOWN_OPTION;
161  break;
162 
163  default:
164  break;
165  }
166 
167  }
168  else
169  {
170  optionLength = ipv6Option->Process (packet, offset + processedSize, ipv6Header, isDropped);
171  }
172 
173  processedSize += optionLength;
174  p->RemoveAtStart (optionLength);
175  }
176 
177  delete [] data;
178 
179  return processedSize;
180 }

363 uint8_t Ipv6ExtensionFragment::Process (Ptr<Packet>& packet,
364  uint8_t offset,
365  Ipv6Header const& ipv6Header,
366  Ipv6Address dst,
367  uint8_t *nextHeader,
368  bool& stopProcessing,
369  bool& isDropped,
370  Ipv6L3Protocol::DropReason& dropReason)
371 {
372  NS_LOG_FUNCTION (this << packet << offset << ipv6Header << dst << nextHeader << isDropped);
373 
374  Ptr<Packet> p = packet->Copy ();
375  p->RemoveAtStart (offset);
376 
377  Ipv6ExtensionFragmentHeader fragmentHeader;
378  p->RemoveHeader (fragmentHeader);
379 
380  if (nextHeader)
381  {
382  *nextHeader = fragmentHeader.GetNextHeader ();
383  }
384 
385  bool moreFragment = fragmentHeader.GetMoreFragment ();
386  uint16_t fragmentOffset = fragmentHeader.GetOffset ();
387  uint32_t identification = fragmentHeader.GetIdentification ();
388  Ipv6Address src = ipv6Header.GetSourceAddress ();
389 
390  FragmentKey_t fragmentsId = FragmentKey_t (src, identification);
391  Ptr<Fragments> fragments;
392 
393  Ipv6Header ipHeader = ipv6Header;
394  ipHeader.SetNextHeader (fragmentHeader.GetNextHeader ());
395 
396  MapFragments_t::iterator it = m_fragments.find (fragmentsId);
397  if (it == m_fragments.end ())
398  {
399  fragments = Create<Fragments> ();
400  m_fragments.insert (std::make_pair (fragmentsId, fragments));
401  FragmentsTimeoutsListI_t iter = SetTimeout (fragmentsId, ipHeader);
402  fragments->SetTimeoutIter (iter);
403  }
404  else
405  {
406  fragments = it->second;
407  }
408 
409  if (fragmentOffset == 0)
410  {
411  Ptr<Packet> unfragmentablePart = packet->Copy ();
412  unfragmentablePart->RemoveAtEnd (packet->GetSize () - offset);
413  fragments->SetUnfragmentablePart (unfragmentablePart);
414  }
415 
416  fragments->AddFragment (p, fragmentOffset, moreFragment);
417 
418  if (fragments->IsEntire ())
419  {
420  packet = fragments->GetPacket ();
421 
422  m_timeoutEventList.erase (fragments->GetTimeoutIter ());
423  m_fragments.erase (fragmentsId);
424  stopProcessing = false;
425  }
426  else
427  {
428  stopProcessing = true;
429  }
430 
431  return 0;
432 }

434 void Ipv6ExtensionFragment::GetFragments (Ptr<Packet> packet, Ipv6Header ipv6Header, uint32_t maxFragmentSize, std::list<Ipv6PayloadHeaderPair>& listFragments)
435 {
436  Ptr<Packet> p = packet->Copy ();
437 
438  uint8_t nextHeader = ipv6Header.GetNextHeader ();
439  uint8_t ipv6HeaderSize = ipv6Header.GetSerializedSize ();
440 
441  uint8_t type;
442  p->CopyData (&type, sizeof(type));
443 
444  bool moreHeader = true;
445  if (!(nextHeader == Ipv6Header::IPV6_EXT_HOP_BY_HOP || nextHeader == Ipv6Header::IPV6_EXT_ROUTING
446  || (nextHeader == Ipv6Header::IPV6_EXT_DESTINATION && type == Ipv6Header::IPV6_EXT_ROUTING)))
447  {
448  moreHeader = false;
449  ipv6Header.SetNextHeader (Ipv6Header::IPV6_EXT_FRAGMENTATION);
450  }
451 
452  std::list<std::pair<Ipv6ExtensionHeader *, uint8_t> > unfragmentablePart;
453  uint32_t unfragmentablePartSize = 0;
454 
455  Ptr<Ipv6ExtensionDemux> extensionDemux = GetNode ()->GetObject<Ipv6ExtensionDemux> ();
456  Ptr<Ipv6Extension> extension = extensionDemux->GetExtension (nextHeader);
457  uint8_t extensionHeaderLength;
458 
459  while (moreHeader)
460  {
461  if (nextHeader == Ipv6Header::IPV6_EXT_HOP_BY_HOP)
462  {
463  Ipv6ExtensionHopByHopHeader *hopbyhopHeader = new Ipv6ExtensionHopByHopHeader ();
464  p->RemoveHeader (*hopbyhopHeader);
465 
466  nextHeader = hopbyhopHeader->GetNextHeader ();
467  extensionHeaderLength = hopbyhopHeader->GetLength ();
468 
469  uint8_t type;
470  p->CopyData (&type, sizeof(type));
471 
472  if (!(nextHeader == Ipv6Header::IPV6_EXT_HOP_BY_HOP || nextHeader == Ipv6Header::IPV6_EXT_ROUTING
473  || (nextHeader == Ipv6Header::IPV6_EXT_DESTINATION && type == Ipv6Header::IPV6_EXT_ROUTING)))
474  {
475  moreHeader = false;
476  hopbyhopHeader->SetNextHeader (Ipv6Header::IPV6_EXT_FRAGMENTATION);
477  }
478 
479  unfragmentablePart.emplace_back (hopbyhopHeader, Ipv6Header::IPV6_EXT_HOP_BY_HOP);
480  unfragmentablePartSize += extensionHeaderLength;
481  }
482  else if (nextHeader == Ipv6Header::IPV6_EXT_ROUTING)
483  {
484  uint8_t buf[2];
485  p->CopyData (buf, sizeof(buf));
486  uint8_t numberAddress = buf[1] / 2;
487  Ipv6ExtensionLooseRoutingHeader *routingHeader = new Ipv6ExtensionLooseRoutingHeader ();
488  routingHeader->SetNumberAddress (numberAddress);
489  p->RemoveHeader (*routingHeader);
490 
491  nextHeader = routingHeader->GetNextHeader ();
492  extensionHeaderLength = routingHeader->GetLength ();
493 
494  uint8_t type;
495  p->CopyData (&type, sizeof(type));
496  if (!(nextHeader == Ipv6Header::IPV6_EXT_HOP_BY_HOP || nextHeader == Ipv6Header::IPV6_EXT_ROUTING
497  || (nextHeader == Ipv6Header::IPV6_EXT_DESTINATION && type == Ipv6Header::IPV6_EXT_ROUTING)))
498  {
499  moreHeader = false;
500  routingHeader->SetNextHeader (Ipv6Header::IPV6_EXT_FRAGMENTATION);
501  }
502 
503  unfragmentablePart.emplace_back (routingHeader, Ipv6Header::IPV6_EXT_ROUTING);
504  unfragmentablePartSize += extensionHeaderLength;
505  }
506  else if (nextHeader == Ipv6Header::IPV6_EXT_DESTINATION)
507  {
508  Ipv6ExtensionDestinationHeader *destinationHeader = new Ipv6ExtensionDestinationHeader ();
509  p->RemoveHeader (*destinationHeader);
510 
511  nextHeader = destinationHeader->GetNextHeader ();
512  extensionHeaderLength = destinationHeader->GetLength ();
513 
514  uint8_t type;
515  p->CopyData (&type, sizeof(type));
516  if (!(nextHeader == Ipv6Header::IPV6_EXT_HOP_BY_HOP || nextHeader == Ipv6Header::IPV6_EXT_ROUTING
517  || (nextHeader == Ipv6Header::IPV6_EXT_DESTINATION && type == Ipv6Header::IPV6_EXT_ROUTING)))
518  {
519  moreHeader = false;
520  destinationHeader->SetNextHeader (Ipv6Header::IPV6_EXT_FRAGMENTATION);
521  }
522 
523  unfragmentablePart.emplace_back (destinationHeader, Ipv6Header::IPV6_EXT_DESTINATION);
524  unfragmentablePartSize += extensionHeaderLength;
525  }
526  }
527 
528  Ipv6ExtensionFragmentHeader fragmentHeader;
529  uint8_t fragmentHeaderSize = fragmentHeader.GetSerializedSize ();
530 
531  uint32_t maxFragmentablePartSize = maxFragmentSize - ipv6HeaderSize - unfragmentablePartSize - fragmentHeaderSize;
532  uint32_t currentFragmentablePartSize = 0;
533 
534  bool moreFragment = true;
535  uint32_t identification = (uint32_t) m_uvar->GetValue (0, (uint32_t)-1);
536  uint16_t offset = 0;
537 
538  do
539  {
540  if (p->GetSize () > offset + maxFragmentablePartSize)
541  {
542  moreFragment = true;
543  currentFragmentablePartSize = maxFragmentablePartSize;
544  currentFragmentablePartSize -= currentFragmentablePartSize % 8;
545  }
546  else
547  {
548  moreFragment = false;
549  currentFragmentablePartSize = p->GetSize () - offset;
550  }
551 
552 
553  fragmentHeader.SetNextHeader (nextHeader);
554  fragmentHeader.SetOffset (offset);
555  fragmentHeader.SetMoreFragment (moreFragment);
556  fragmentHeader.SetIdentification (identification);
557 
558  Ptr<Packet> fragment = p->CreateFragment (offset, currentFragmentablePartSize);
559  offset += currentFragmentablePartSize;
560 
561  fragment->AddHeader (fragmentHeader);
562 
563  for (std::list<std::pair<Ipv6ExtensionHeader *, uint8_t> >::iterator it = unfragmentablePart.begin (); it != unfragmentablePart.end (); it++)
564  {
565  if (it->second == Ipv6Header::IPV6_EXT_HOP_BY_HOP)
566  {
567  Ipv6ExtensionHopByHopHeader * p =
568  dynamic_cast<Ipv6ExtensionHopByHopHeader *> (it->first);
569  NS_ASSERT (p != 0);
570  fragment->AddHeader (*p);
571  }
572  else if (it->second == Ipv6Header::IPV6_EXT_ROUTING)
573  {
574  Ipv6ExtensionLooseRoutingHeader * p =
575  dynamic_cast<Ipv6ExtensionLooseRoutingHeader *> (it->first);
576  NS_ASSERT (p != 0);
577  fragment->AddHeader (*p);
578  }
579  else if (it->second == Ipv6Header::IPV6_EXT_DESTINATION)
580  {
581  Ipv6ExtensionDestinationHeader * p =
582  dynamic_cast<Ipv6ExtensionDestinationHeader *> (it->first);
583  NS_ASSERT (p != 0);
584  fragment->AddHeader (*p);
585  }
586  }
587 
588  ipv6Header.SetPayloadLength (fragment->GetSize ());
589 
590  std::ostringstream oss;
591  oss << ipv6Header;
592  fragment->Print (oss);
593 
594  listFragments.emplace_back (fragment, ipv6Header);
595  }
596  while (moreFragment);
597 
598  for (std::list<std::pair<Ipv6ExtensionHeader *, uint8_t> >::iterator it = unfragmentablePart.begin (); it != unfragmentablePart.end (); it++)
599  {
600  delete it->first;
601  }
602 
603  unfragmentablePart.clear ();
604 }

607 void Ipv6ExtensionFragment::HandleFragmentsTimeout (FragmentKey_t fragmentsId,
608  Ipv6Header ipHeader)
609 {
610  Ptr<Fragments> fragments;
611 
612  MapFragments_t::iterator it = m_fragments.find (fragmentsId);
613  NS_ASSERT_MSG(it != m_fragments.end (), "IPv6 Fragment timeout reached for non-existent fragment");
614  fragments = it->second;
615 
616  Ptr<Packet> packet = fragments->GetPartialPacket ();
617 
618  // if we have at least 8 bytes, we can send an ICMP.
619  if ( packet->GetSize () > 8 )
620  {
621  Ptr<Packet> p = packet->Copy ();
622  p->AddHeader (ipHeader);
623  Ptr<Icmpv6L4Protocol> icmp = GetNode ()->GetObject<Icmpv6L4Protocol> ();
624  icmp->SendErrorTimeExceeded (p, ipHeader.GetSourceAddress (), Icmpv6Header::ICMPV6_FRAGTIME);
625  }
626 
627  Ptr<Ipv6L3Protocol> ipL3 = GetNode ()->GetObject<Ipv6L3Protocol> ();
628  ipL3->ReportDrop (ipHeader, packet, Ipv6L3Protocol::DROP_FRAGMENT_TIMEOUT);
629 
630  // clear the buffers
631  m_fragments.erase (fragmentsId);
632 }

648 void Ipv6ExtensionFragment::HandleTimeout (void)
649 {
650  Time now = Simulator::Now ();
651 
652  while (!m_timeoutEventList.empty () && std::get<0> (*m_timeoutEventList.begin ()) == now)

654  HandleFragmentsTimeout (std::get<1> (*m_timeoutEventList.begin ()),

967 uint8_t Ipv6ExtensionLooseRouting::Process (Ptr<Packet>& packet,
968  uint8_t offset,
969  Ipv6Header const& ipv6Header,
970  Ipv6Address dst,
971  uint8_t *nextHeader,
972  bool& stopProcessing,
973  bool& isDropped,
974  Ipv6L3Protocol::DropReason& dropReason)
975 {
976  NS_LOG_FUNCTION (this << packet << offset << ipv6Header << dst << nextHeader << isDropped);
977 
978  // For ICMPv6 Error packets
979  Ptr<Packet> malformedPacket = packet->Copy ();
980  malformedPacket->AddHeader (ipv6Header);
981 
982  Ptr<Packet> p = packet->Copy ();
983  p->RemoveAtStart (offset);
984 
985  // Copy IPv6 Header : ipv6Header -> ipv6header
986  Buffer tmp;
987  tmp.AddAtStart (ipv6Header.GetSerializedSize ());
988  Buffer::Iterator it = tmp.Begin ();
989  Ipv6Header ipv6header;
990  ipv6Header.Serialize (it);
991  ipv6header.Deserialize (it);
992 
993  // Get the number of routers' address field
994  uint8_t buf[2];
995  p->CopyData (buf, sizeof(buf));
996  uint8_t numberAddress = buf[1] / 2;
997  Ipv6ExtensionLooseRoutingHeader routingHeader;
998  routingHeader.SetNumberAddress (numberAddress);
999  p->RemoveHeader (routingHeader);
1000 
1001  if (nextHeader)
1002  {
1003  *nextHeader = routingHeader.GetNextHeader ();
1004  }
1005 
1006  Ptr<Icmpv6L4Protocol> icmpv6 = GetNode ()->GetObject<Ipv6L3Protocol> ()->GetIcmpv6 ();
1007 
1008  Ipv6Address srcAddress = ipv6header.GetSourceAddress ();
1009  Ipv6Address destAddress = ipv6header.GetDestinationAddress ();
1010  uint8_t hopLimit = ipv6header.GetHopLimit ();
1011  uint8_t segmentsLeft = routingHeader.GetSegmentsLeft ();
1012  uint8_t length = (routingHeader.GetLength () >> 3) - 1;
1013  uint8_t nbAddress = length / 2;
1014  uint8_t nextAddressIndex;
1015  Ipv6Address nextAddress;
1016 
1017  if (segmentsLeft == 0)
1018  {
1019  isDropped = false;
1020  return routingHeader.GetSerializedSize ();
1021  }
1022 
1023  if (length % 2 != 0)
1024  {
1025  NS_LOG_LOGIC ("Malformed header. Drop!");
1026  icmpv6->SendErrorParameterError (malformedPacket, srcAddress, Icmpv6Header::ICMPV6_MALFORMED_HEADER, offset + 1);
1027  dropReason = Ipv6L3Protocol::DROP_MALFORMED_HEADER;
1028  isDropped = true;
1029  stopProcessing = true;
1030  return routingHeader.GetSerializedSize ();
1031  }
1032 
1033  if (segmentsLeft > nbAddress)
1034  {
1035  NS_LOG_LOGIC ("Malformed header. Drop!");
1036  icmpv6->SendErrorParameterError (malformedPacket, srcAddress, Icmpv6Header::ICMPV6_MALFORMED_HEADER, offset + 3);
1037  dropReason = Ipv6L3Protocol::DROP_MALFORMED_HEADER;
1038  isDropped = true;
1039  stopProcessing = true;
1040  return routingHeader.GetSerializedSize ();
1041  }
1042 
1043  routingHeader.SetSegmentsLeft (segmentsLeft - 1);
1044  nextAddressIndex = nbAddress - segmentsLeft;
1045  nextAddress = routingHeader.GetRouterAddress (nextAddressIndex);
1046 
1047  if (nextAddress.IsMulticast () || destAddress.IsMulticast ())
1048  {
1049  dropReason = Ipv6L3Protocol::DROP_MALFORMED_HEADER;
1050  isDropped = true;
1051  stopProcessing = true;
1052  return routingHeader.GetSerializedSize ();
1053  }
1054 
1055  routingHeader.SetRouterAddress (nextAddressIndex, destAddress);
1056  ipv6header.SetDestinationAddress (nextAddress);
1057 
1058  if (hopLimit <= 1)
1059  {
1060  NS_LOG_LOGIC ("Time Exceeded : Hop Limit <= 1. Drop!");
1061  icmpv6->SendErrorTimeExceeded (malformedPacket, srcAddress, Icmpv6Header::ICMPV6_HOPLIMIT);
1062  dropReason = Ipv6L3Protocol::DROP_MALFORMED_HEADER;
1063  isDropped = true;
1064  stopProcessing = true;
1065  return routingHeader.GetSerializedSize ();
1066  }
1067 
1068  ipv6header.SetHopLimit (hopLimit - 1);
1069  p->AddHeader (routingHeader);
1070 
1071  /* short-circuiting routing stuff
1072  *
1073  * If we process this option,
1074  * the packet was for us so we resend it to
1075  * the new destination (modified in the header above).
1076  */
1077 
1078  Ptr<Ipv6L3Protocol> ipv6 = GetNode ()->GetObject<Ipv6L3Protocol> ();
1079  Ptr<Ipv6RoutingProtocol> ipv6rp = ipv6->GetRoutingProtocol ();
1080  Socket::SocketErrno err;
1081  NS_ASSERT (ipv6rp);
1082 
1083  Ptr<Ipv6Route> rtentry = ipv6rp->RouteOutput (p, ipv6header, 0, err);
1084 
1085  if (rtentry)
1086  {
1087  /* we know a route exists so send packet now */
1088  ipv6->SendRealOut (rtentry, p, ipv6header);
1089  }
1090  else
1091  {
1092  NS_LOG_INFO ("No route for next router");
1093  }
1094 
1095  /* as we directly send packet, mark it as dropped */
1096  isDropped = true;
1097 
1098  return routingHeader.GetSerializedSize ();
1099 }

81 void
82 NetworkServer::AddGateway (Ptr<Node> gateway, Ptr<NetDevice> netDevice)
83 {
84  NS_LOG_FUNCTION (this << gateway);
85 
86  // Get the PointToPointNetDevice
87  Ptr<PointToPointNetDevice> p2pNetDevice;
88  for (uint32_t i = 0; i < gateway->GetNDevices (); i++)
89  {
90  p2pNetDevice = gateway->GetDevice (i)->GetObject<PointToPointNetDevice> ();
91  if (p2pNetDevice != 0)
92  {
93  // We found a p2pNetDevice on the gateway
94  break;
95  }
96  }
97 
98  // Get the gateway's LoRa MAC layer (assumes gateway's MAC is configured as first device)
99  Ptr<GatewayLorawanMac> gwMac = gateway->GetDevice (0)->GetObject<LoraNetDevice> ()->
100  GetMac ()->GetObject<GatewayLorawanMac> ();
101  NS_ASSERT (gwMac != 0);
102 
103  // Get the Address
104  Address gatewayAddress = p2pNetDevice->GetAddress ();
105 
106  // Create new gatewayStatus
107  Ptr<GatewayStatus> gwStatus = Create<GatewayStatus> (gatewayAddress,
108  netDevice,
109  gwMac);
110 
111  m_status->AddGateway (gatewayAddress, gwStatus);
112 }

152 bool
153 NetworkServer::Receive (Ptr<NetDevice> device, Ptr<const Packet> packet,
154  uint16_t protocol, const Address& address)
155 {
156  NS_LOG_FUNCTION (this << packet << protocol << address);
157 
158  // Create a copy of the packet
159  Ptr<Packet> myPacket = packet->Copy ();
160 
161  // Fire the trace source
162  m_receivedPacket (packet);
163 
164  // Inform the scheduler of the newly arrived packet
165  m_scheduler->OnReceivedPacket (packet);
166 
167  // Inform the status of the newly arrived packet
168  m_status->OnReceivedPacket (packet, address);
169 
170  // Inform the controller of the newly arrived packet
171  m_controller->OnNewPacket (packet);
172 
173  return true;
174 }

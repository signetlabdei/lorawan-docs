92 
93  // Find the received packet in the m_macPacketTracker
94  auto it = m_macPacketTracker.find (packet);

220 std::vector<int>
221 LoraPacketTracker::CountPhyPacketsPerGw (Time startTime, Time stopTime,
222  int gwId)
223 {
224  // Vector packetCounts will contain - for the interval given in the input of
225  // the function, the following fields: totPacketsSent receivedPackets
226  // interferedPackets noMoreGwPackets underSensitivityPackets
227 
228  std::vector<int> packetCounts (6, 0);
229 
230  for (auto itPhy = m_packetTracker.begin ();
231  itPhy != m_packetTracker.end ();
232  ++itPhy)
233  {
234  if ((*itPhy).second.sendTime >= startTime && (*itPhy).second.sendTime <= stopTime)
235  {
236  packetCounts.at (0)++;
237 
238  NS_LOG_DEBUG ("Dealing with packet " << (*itPhy).second.packet);
239  NS_LOG_DEBUG ("This packet was received by " <<
240  (*itPhy).second.outcomes.size () << " gateways");
241 
242  if ((*itPhy).second.outcomes.count (gwId) > 0)
243  {
244  switch ((*itPhy).second.outcomes.at (gwId))
245  {
246  case RECEIVED:
247  {
248  packetCounts.at (1)++;
249  break;
250  }
251  case INTERFERED:
252  {
253  packetCounts.at (2)++;
254  break;
255  }
256  case NO_MORE_RECEIVERS:
257  {
258  packetCounts.at (3)++;
259  break;
260  }
261  case UNDER_SENSITIVITY:
262  {
263  packetCounts.at (4)++;
264  break;
265  }
266  case LOST_BECAUSE_TX:
267  {
268  packetCounts.at (5)++;
269  break;
270  }
271  case UNSET:
272  {
273  break;
274  }
275  }
276  }
277  }
278  }
279 
280  return packetCounts;
281 }

352  LoraPacketTracker::CountMacPacketsGlobally (Time startTime, Time stopTime)
353  {
354  NS_LOG_FUNCTION (this << startTime << stopTime);
355 
356  double sent = 0;

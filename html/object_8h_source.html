462 template <typename T>
463 Ptr<T>
464 Object::GetObject () const
465 {
466  // This is an optimization: if the cast works (which is likely),
467  // things will be pretty fast.
468  T *result = dynamic_cast<T *> (m_aggregates->buffer[0]);
469  if (result != 0)
470  {
471  return Ptr<T> (result);
472  }
473  // if the cast does not work, we try to do a full type check.
474  Ptr<Object> found = DoGetObject (T::GetTypeId ());
475  if (found != 0)
476  {
477  return Ptr<T> (static_cast<T *> (PeekPointer (found)));
478  }
479  return 0;
480 }

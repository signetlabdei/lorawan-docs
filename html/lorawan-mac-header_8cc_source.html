62 void
63 LorawanMacHeader::Serialize (Buffer::Iterator start) const
64 {
65  NS_LOG_FUNCTION_NOARGS ();
66 
67  // The header we need to fill
68  uint8_t header = 0;
69 
70  // The MType
71  header |= m_mtype << 5;
72 
73  // Do nothing for the bits that are RFU
74 
75  // The major version bits
76  header |= m_major;
77 
78  // Write the byte
79  start.WriteU8 (header);
80 
81  NS_LOG_DEBUG ("Serialization of MAC header: " << std::bitset<8> (header));
82 }

84 uint32_t
85 LorawanMacHeader::Deserialize (Buffer::Iterator start)
86 {
87  NS_LOG_FUNCTION_NOARGS ();
88 
89  // Save the byte on a temporary variable
90  uint8_t byte;
91  byte = start.ReadU8 ();
92 
93  // Get the 2 least significant bits to have the Major
94  m_major = byte & 0b11;
95 
96  // Move the three most significant bits to the least significant positions
97  // to get the MType
98  m_mtype = byte >> 5;
99 
100  return 1; // the number of bytes consumed.
101 }

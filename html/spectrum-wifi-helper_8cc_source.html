67  Ptr<ErrorRateModel> error = m_errorRateModel.Create<ErrorRateModel> ();
68  phy->SetErrorRateModel (error);
69  if (m_frameCaptureModel.IsTypeIdSet ())
70  {
71  Ptr<FrameCaptureModel> capture = m_frameCaptureModel.Create<FrameCaptureModel> ();
72  phy->SetFrameCaptureModel (capture);
73  }
74  if (m_preambleDetectionModel.IsTypeIdSet ())
75  {
76  Ptr<PreambleDetectionModel> capture = m_preambleDetectionModel.Create<PreambleDetectionModel> ();

78  }
79  phy->SetChannel (m_channel);
80  phy->SetDevice (device);
81  phy->SetMobility (node->GetObject<MobilityModel> ());

160 EthernetHeader::Serialize (Buffer::Iterator start) const
161 {
162  NS_LOG_FUNCTION (this << &start);
163  Buffer::Iterator i = start;
164 
165  if (m_enPreambleSfd)

167  i.WriteU64 (m_preambleSfd);

183 
184  ReadFrom (i, m_destination);

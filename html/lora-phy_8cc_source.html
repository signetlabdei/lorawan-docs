158 Time
159 LoraPhy::GetOnAirTime (Ptr<Packet> packet, LoraTxParameters txParams)
160 {
161 
162  NS_LOG_FUNCTION (packet << txParams);
163 
164  // The contents of this function are based on [1].
165  // [1] SX1272 LoRa modem designer's guide.
166 
167  // Compute the symbol duration
168  // Bandwidth is in Hz
169  double tSym = pow (2, int(txParams.sf)) / (txParams.bandwidthHz);
170 
171  // Compute the preamble duration
172  double tPreamble = (double(txParams.nPreamble) + 4.25) * tSym;
173 
174  // Payload size
175  uint32_t pl = packet->GetSize (); // Size in bytes
176  NS_LOG_DEBUG ("Packet of size " << pl << " bytes");
177 
178  // This step is needed since the formula deals with double values.
179  // de = 1 when the low data rate optimization is enabled, 0 otherwise
180  // h = 1 when header is implicit, 0 otherwise
181  double de = txParams.lowDataRateOptimizationEnabled ? 1 : 0;
182  double h = txParams.headerDisabled ? 1 : 0;
183  double crc = txParams.crcEnabled ? 1 : 0;
184 
185  // num and den refer to numerator and denominator of the time on air formula
186  double num = 8 * pl - 4 * txParams.sf + 28 + 16 * crc - 20 * h;
187  double den = 4 * (txParams.sf - 2 * de);
188  double payloadSymbNb = 8 + std::max (std::ceil (num / den) *
189  (txParams.codingRate + 4), double(0));
190 
191  // Time to transmit the payload
192  double tPayload = payloadSymbNb * tSym;
193 
194  NS_LOG_DEBUG ("Time computation: num = " << num << ", den = " << den <<
195  ", payloadSymbNb = " << payloadSymbNb << ", tSym = " << tSym);
196  NS_LOG_DEBUG ("tPreamble = " << tPreamble);
197  NS_LOG_DEBUG ("tPayload = " << tPayload);
198  NS_LOG_DEBUG ("Total time = " << tPreamble + tPayload);
199 
200  // Compute and return the total packet on-air time
201  return Seconds (tPreamble + tPayload);
202 }

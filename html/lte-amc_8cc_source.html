329 std::vector<int>
330 LteAmc::CreateCqiFeedbacks (const SpectrumValue& sinr, uint8_t rbgSize)
331 {
332  NS_LOG_FUNCTION (this);
333 
334  std::vector<int> cqi;
335  Values::const_iterator it;
336 
337  if (m_amcModel == PiroEW2010)
338  {
339 
340  for (it = sinr.ConstValuesBegin (); it != sinr.ConstValuesEnd (); it++)
341  {
342  double sinr_ = (*it);
343  if (sinr_ == 0.0)
344  {
345  cqi.push_back (-1); // SINR == 0 (linear units) means no signal in this RB
346  }
347  else
348  {
349  /*
350  * Compute the spectral efficiency from the SINR
351  * SINR
352  * spectralEfficiency = log2 (1 + -------------------- )
353  * -ln(5*BER)/1.5
354  * NB: SINR must be expressed in linear units
355  */
356 
357  double s = log2 ( 1 + ( sinr_ / ( (-std::log (5.0 * m_ber )) / 1.5) ));
358 
359  int cqi_ = GetCqiFromSpectralEfficiency (s);
360 
361  NS_LOG_LOGIC (" PRB =" << cqi.size ()
362  << ", sinr = " << sinr_
363  << " (=" << 10 * std::log10 (sinr_) << " dB)"
364  << ", spectral efficiency =" << s
365  << ", CQI = " << cqi_ << ", BER = " << m_ber);
366 
367  cqi.push_back (cqi_);
368  }
369  }
370  }
371  else if (m_amcModel == MiErrorModel)
372  {
373  NS_LOG_DEBUG (this << " AMC-VIENNA RBG size " << (uint16_t)rbgSize);
374  NS_ASSERT_MSG (rbgSize > 0, " LteAmc-Vienna: RBG size must be greater than 0");
375  std::vector <int> rbgMap;
376  int rbId = 0;
377  for (it = sinr.ConstValuesBegin (); it != sinr.ConstValuesEnd (); it++)
378  {
379  rbgMap.push_back (rbId++);
380  if ((rbId % rbgSize == 0)||((it+1)==sinr.ConstValuesEnd ()))
381  {
382  uint8_t mcs = 0;
383  TbStats_t tbStats;
384  while (mcs <= 28)
385  {
386  HarqProcessInfoList_t harqInfoList;
387  tbStats = LteMiErrorModel::GetTbDecodificationStats (sinr, rbgMap, (uint16_t)GetDlTbSizeFromMcs (mcs, rbgSize) / 8, mcs, harqInfoList);
388  if (tbStats.tbler > 0.1)
389  {
390  break;
391  }
392  mcs++;
393 
394  }
395  if (mcs > 0)
396  {
397  mcs--;
398  }
399  NS_LOG_DEBUG (this << "\t RBG " << rbId << " MCS " << (uint16_t)mcs << " TBLER " << tbStats.tbler);
400  int rbgCqi = 0;
401  if ((tbStats.tbler > 0.1)&&(mcs==0))
402  {
403  rbgCqi = 0; // any MCS can guarantee the 10 % of BER
404  }
405  else if (mcs == 28)
406  {
407  rbgCqi = 15; // all MCSs can guarantee the 10 % of BER
408  }
409  else
410  {
411  double s = SpectralEfficiencyForMcs[mcs];
412  rbgCqi = 0;
413  while ((rbgCqi < 15) && (SpectralEfficiencyForCqi[rbgCqi + 1] < s))
414  {
415  ++rbgCqi;
416  }
417  }
418  NS_LOG_DEBUG (this << "\t MCS " << (uint16_t)mcs << "-> CQI " << rbgCqi);
419  // fill the cqi vector (per RB basis)
420  for (uint8_t j = 0; j < rbgSize; j++)
421  {
422  cqi.push_back (rbgCqi);
423  }
424  rbgMap.clear ();
425  }
426 
427  }
428 
429  }
430 
431  return cqi;
432 }

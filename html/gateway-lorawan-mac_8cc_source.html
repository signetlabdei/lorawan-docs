54 void
55 GatewayLorawanMac::Send (Ptr<Packet> packet)
56 {
57  NS_LOG_FUNCTION (this << packet);
58 
59  // Get DataRate to send this packet with
60  LoraTag tag;
61  packet->RemovePacketTag (tag);
62  uint8_t dataRate = tag.GetDataRate ();
63  double frequency = tag.GetFrequency ();
64  NS_LOG_DEBUG ("DR: " << unsigned (dataRate));
65  NS_LOG_DEBUG ("SF: " << unsigned (GetSfFromDataRate (dataRate)));
66  NS_LOG_DEBUG ("BW: " << GetBandwidthFromDataRate (dataRate));
67  NS_LOG_DEBUG ("Freq: " << frequency << " MHz");
68  packet->AddPacketTag (tag);
69 
70  // Make sure we can transmit this packet
71  if (m_channelHelper.GetWaitingTime(CreateObject<LogicalLoraChannel> (frequency)) > 0)
72  {
73  // We cannot send now!
74  NS_LOG_WARN ("Trying to send a packet but Duty Cycle won't allow it. Aborting.");
75  return;
76  }
77 
78  LoraTxParameters params;
79  params.sf = GetSfFromDataRate (dataRate);
80  params.headerDisabled = false;
81  params.codingRate = 1;
82  params.bandwidthHz = GetBandwidthFromDataRate (dataRate);
83  params.nPreamble = 8;
84  params.crcEnabled = 1;
85  params.lowDataRateOptimizationEnabled = 0;
86 
87  // Get the duration
88  Time duration = m_phy->GetOnAirTime (packet, params);
89 
90  NS_LOG_DEBUG ("Duration: " << duration.GetSeconds ());
91 
92  // Find the channel with the desired frequency
93  double sendingPower = m_channelHelper.GetTxPowerForChannel
94  (CreateObject<LogicalLoraChannel> (frequency));
95 
96  // Add the event to the channelHelper to keep track of duty cycle
97  m_channelHelper.AddEvent (duration, CreateObject<LogicalLoraChannel>
98  (frequency));
99 
100  // Send the packet to the PHY layer to send it on the channel
101  m_phy->Send (packet, params, frequency, sendingPower);
102 
103  m_sentNewPacket (packet);
104 }

112 void
113 GatewayLorawanMac::Receive (Ptr<Packet const> packet)
114 {
115  NS_LOG_FUNCTION (this << packet);
116 
117  // Make a copy of the packet to work on
118  Ptr<Packet> packetCopy = packet->Copy ();
119 
120  // Only forward the packet if it's uplink
121  LorawanMacHeader macHdr;
122  packetCopy->PeekHeader (macHdr);
123 
124  if (macHdr.IsUplink ())
125  {
126  m_device->GetObject<LoraNetDevice> ()->Receive (packetCopy);
127 
128  NS_LOG_DEBUG ("Received packet: " << packet);
129 
130  m_receivedPacket (packet);
131  }
132  else
133  {
134  NS_LOG_DEBUG ("Not forwarding downlink message to NetDevice");
135  }
136 }

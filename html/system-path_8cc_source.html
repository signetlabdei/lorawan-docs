98 std::string FindSelfDirectory (void)
99 {
109  NS_LOG_FUNCTION_NOARGS ();
110  std::string filename;
111 #if defined(__linux__)
112  {
113  ssize_t size = 1024;
114  char *buffer = (char*)malloc (size);
115  memset (buffer, 0, size);
116  int status;
117  while (true)
118  {
119  status = readlink ("/proc/self/exe", buffer, size);
120  if (status != 1 || (status == -1 && errno != ENAMETOOLONG))
121  {
122  break;
123  }
124  size *= 2;
125  free (buffer);
126  buffer = (char*)malloc (size);
127  memset (buffer, 0, size);
128  }
129  if (status == -1)
130  {
131  NS_FATAL_ERROR ("Oops, could not find self directory.");
132  }
133  filename = buffer;
134  free (buffer);
135  }
136 #elif defined (__win32__)
137  {
141  DWORD size = 1024;
142  LPTSTR lpFilename = (LPTSTR) malloc (sizeof(TCHAR) * size);
143  DWORD status = GetModuleFilename (0, lpFilename, size);
144  while (status == size)
145  {
146  size = size * 2;
147  free (lpFilename);
148  lpFilename = (LPTSTR) malloc (sizeof(TCHAR) * size);
149  status = GetModuleFilename (0, lpFilename, size);
150  }
151  NS_ASSERT (status != 0);
152  filename = lpFilename;
153  free (lpFilename);
154  }
155 #elif defined (__APPLE__)
156  {
157  uint32_t bufsize = 1024;
158  char *buffer = (char *) malloc (bufsize);
159  NS_ASSERT (buffer != 0);
160  int status = _NSGetExecutablePath (buffer, &bufsize);
161  if (status == -1)
162  {
163  free (buffer);
164  buffer = (char *) malloc (bufsize);
165  status = _NSGetExecutablePath (buffer, &bufsize);
166  }
167  NS_ASSERT (status == 0);
168  filename = buffer;
169  free (buffer);
170  }
171 #elif defined (__FreeBSD__)
172  {
173  int mib[4];
174  std::size_t bufSize = 1024;
175  char *buf = (char *) malloc (bufSize);
176 
177  mib[0] = CTL_KERN;
178  mib[1] = KERN_PROC;
179  mib[2] = KERN_PROC_PATHNAME;
180  mib[3] = -1;
181 
182  sysctl (mib, 4, buf, &bufSize, NULL, 0);
183  filename = buf;
184  }
185 #endif
186  return Dirname (filename);
187 }

189 std::string Append (std::string left, std::string right)
190 {
191  // removing trailing separators from 'left'
192  NS_LOG_FUNCTION (left << right);
193  while (true)
194  {
195  std::string::size_type lastSep = left.rfind (SYSTEM_PATH_SEP);
196  if (lastSep != left.size () - 1)
197  {
198  break;
199  }
200  left = left.substr (0, left.size () - 1);
201  }
202  std::string retval = left + SYSTEM_PATH_SEP + right;
203  return retval;
204 }

206 std::list<std::string> Split (std::string path)
207 {
208  NS_LOG_FUNCTION (path);
209  std::list<std::string> retval;
210  std::string::size_type current = 0, next = 0;
211  next = path.find (SYSTEM_PATH_SEP, current);
212  while (next != std::string::npos)
213  {
214  std::string item = path.substr (current, next - current);
215  retval.push_back (item);
216  current = next + 1;
217  next = path.find (SYSTEM_PATH_SEP, current);
218  }
219  std::string item = path.substr (current, next - current);
220  retval.push_back (item);
221  return retval;
222 }

224 std::string Join (std::list<std::string>::const_iterator begin,
225  std::list<std::string>::const_iterator end)
226 {
227  NS_LOG_FUNCTION (&begin << &end);
228  std::string retval = "";
229  for (std::list<std::string>::const_iterator i = begin; i != end; i++)
230  {
231  if (i == begin)
232  {
233  retval = *i;
234  }
235  else
236  {
237  retval = retval + SYSTEM_PATH_SEP + *i;
238  }
239  }
240  return retval;

243 std::list<std::string> ReadFiles (std::string path)
244 {
245  NS_LOG_FUNCTION (path);
246  std::list<std::string> files;
247 #if defined HAVE_OPENDIR
248  DIR *dp = opendir (path.c_str ());
249  if (dp == NULL)
250  {
251  NS_FATAL_ERROR ("Could not open directory=" << path);
252  }
253  struct dirent *de = readdir (dp);
254  while (de != 0)
255  {
256  files.push_back (de->d_name);
257  de = readdir (dp);
258  }
259  closedir (dp);
260 #elif defined (HAVE_FIND_FIRST_FILE)
261 
262  HANDLE hFind;
263  WIN32_FIND_DATA fileData;
264 
265  hFind = FindFirstFile (path.c_str (), &FindFileData);
266  if (hFind == INVALID_HANDLE_VALUE)
267  {
268  NS_FATAL_ERROR ("Could not open directory=" << path);
269  }
270  do
271  {
272  files.push_back (fileData.cFileName);
273  }
274  while (FindNextFile (hFind, &fileData));
275  FindClose (hFind);
276 #else
277 #error "No support for reading a directory on this platform"
278 #endif
279  return files;
280 }

282 std::string
283 MakeTemporaryDirectoryName (void)
284 {
285  NS_LOG_FUNCTION_NOARGS ();
286  char *path = NULL;
287 
288  path = std::getenv ("TMP");
289  if (path == NULL || std::strlen (path) == 0)
290  {
291  path = std::getenv ("TEMP");
292  if (path == NULL || std::strlen (path) == 0)
293  {
294  path = const_cast<char *> ("/tmp");
295  }
296  }
297 
298  //
299  // Just in case the user wants to go back and find the output, we give
300  // a hint as to which dir we created by including a time hint.
301  //
302  time_t now = time (NULL);
303  struct tm *tm_now = localtime (&now);
304  //
305  // But we also randomize the name in case there are multiple users doing
306  // this at the same time
307  //
308  srand (time (0));
309  long int n = rand ();
310 
311  //
312  // The final path to the directory is going to look something like
313  //
314  // /tmp/ns3.14.30.29.32767
315  //
316  // The first segment comes from one of the temporary directory env
317  // variables or /tmp if not found. The directory name starts with an
318  // identifier telling folks who is making all of the temp directories
319  // and then the local time (in this case 14.30.29 -- which is 2:30 and
320  // 29 seconds PM).
321  //
322  std::ostringstream oss;
323  oss << path << SYSTEM_PATH_SEP << "ns-3." << tm_now->tm_hour << "."
324  << tm_now->tm_min << "." << tm_now->tm_sec << "." << n;
325 
326  return oss.str ();
327 }

331 {
332  NS_LOG_FUNCTION (path);
333 
334  // Make sure all directories on the path exist
335  std::list<std::string> elements = Split (path);
336  auto i = elements.begin ();
337  while (i != elements.end ())
338  {
339  if (*i == "")
340  {
341  NS_LOG_LOGIC ("skipping empty directory name");
342  ++i;
343  continue;
344  }
345  NS_LOG_LOGIC ("creating directory " << *i);
346  ++i; // Now points to one past the directory we want to create
347  std::string tmp = Join (elements.begin (), i);
348  bool makeDirErr = false;
349 
350 #if defined(HAVE_MKDIR_H)
351  makeDirErr = mkdir (tmp.c_str (), S_IRWXU);
352 #endif
353 
354  if (makeDirErr)
355  {
356  NS_LOG_ERROR ("failed creating directory " << tmp);
357  }
358  }

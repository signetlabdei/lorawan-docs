160 NodeListPriv::DoDispose (void)
161 {
162  NS_LOG_FUNCTION (this);
163  for (std::vector<Ptr<Node> >::iterator i = m_nodes.begin ();
164  i != m_nodes.end (); i++)
165  {
166  Ptr<Node> node = *i;
167  node->Dispose ();
168  *i = 0;
169  }
170  m_nodes.erase (m_nodes.begin (), m_nodes.end ());
171  Object::DoDispose ();
172 }

108 void
109 LoraChannel::Send (Ptr< LoraPhy > sender, Ptr< Packet > packet,
110  double txPowerDbm, LoraTxParameters txParams,
111  Time duration, double frequencyMHz) const
112 {
113  NS_LOG_FUNCTION (this << sender << packet << txPowerDbm << txParams <<
114  duration << frequencyMHz);
115 
116  // Get the mobility model of the sender
117  Ptr<MobilityModel> senderMobility = sender->GetMobility ()->GetObject<MobilityModel> ();
118 
119  NS_ASSERT (senderMobility != 0); // Make sure it's available
120 
121  NS_LOG_INFO ("Starting cycle over all " << m_phyList.size () << " PHYs");
122  NS_LOG_INFO ("Sender mobility: " << senderMobility->GetPosition ());
123 
124  // Cycle over all registered PHYs
125  uint32_t j = 0;
126  std::vector<Ptr<LoraPhy> >::const_iterator i;
127  for (i = m_phyList.begin (); i != m_phyList.end (); i++, j++)
128  {
129  // Do not deliver to the sender (*i is the current PHY)
130  if (sender != (*i))
131  {
132  // Get the receiver's mobility model
133  Ptr<MobilityModel> receiverMobility = (*i)->GetMobility ()->
134  GetObject<MobilityModel> ();
135 
136  NS_LOG_INFO ("Receiver mobility: " <<
137  receiverMobility->GetPosition ());
138 
139  // Compute delay using the delay model
140  Time delay = m_delay->GetDelay (senderMobility, receiverMobility);
141 
142  // Compute received power using the loss model
143  double rxPowerDbm = GetRxPower (txPowerDbm, senderMobility,
144  receiverMobility);
145 
146  NS_LOG_DEBUG ("Propagation: txPower=" << txPowerDbm <<
147  "dbm, rxPower=" << rxPowerDbm << "dbm, " <<
148  "distance=" << senderMobility->GetDistanceFrom (receiverMobility) <<
149  "m, delay=" << delay);
150 
151  // Get the id of the destination PHY to correctly format the context
152  Ptr<NetDevice> dstNetDevice = m_phyList[j]->GetDevice ();
153  uint32_t dstNode = 0;
154  if (dstNetDevice != 0)
155  {
156  NS_LOG_INFO ("Getting node index from NetDevice, since it exists");
157  dstNode = dstNetDevice->GetNode ()->GetId ();
158  NS_LOG_DEBUG ("dstNode = " << dstNode);
159  }
160  else
161  {
162  NS_LOG_INFO ("No net device connected to the PHY, using context 0");
163  }
164 
165  // Create the parameters object based on the calculations above
166  LoraChannelParameters parameters;
167  parameters.rxPowerDbm = rxPowerDbm;
168  parameters.sf = txParams.sf;
169  parameters.duration = duration;
170  parameters.frequencyMHz = frequencyMHz;
171 
172  // Schedule the receive event
173  NS_LOG_INFO ("Scheduling reception of the packet");
174  Simulator::ScheduleWithContext (dstNode, delay, &LoraChannel::Receive,
175  this, j, packet, parameters);
176 
177  // Fire the trace source for sent packet
178  m_packetSent (packet);
179  }
180  }
181 }

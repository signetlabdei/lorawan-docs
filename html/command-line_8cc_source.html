136 {

139  PrintDoxygenUsage ();

172 bool
173 CommandLine::HandleOption (const std::string & param) const
174 {
175  // remove leading "--" or "-"
176  std::string arg = param;
177  std::string::size_type cur = arg.find ("--");
178  if (cur == 0)
179  {
180  arg = arg.substr (2, arg.size () - 2);
181  }
182  else
183  {
184  cur = arg.find ("-");
185  if (cur == 0)
186  {
187  arg = arg.substr (1, arg.size () - 1);
188  }
189  else
190  {
191  // non-option argument?
192  return false;
193  }
194  }
195  // find any value following '='
196  cur = arg.find ("=");
197  std::string name, value;
198  if (cur == std::string::npos)
199  {
200  name = arg;
201  value = "";
202  }
203  else
204  {
205  name = arg.substr (0, cur);
206  value = arg.substr (cur + 1, arg.size () - (cur + 1));
207  }
208  HandleArgument (name, value);
209 
210  return true;
211 }

213 bool
214 CommandLine::HandleNonOption (const std::string &value)
215 {
216  NS_LOG_FUNCTION (this << value);
217 
218  if (m_nonOptionCount == m_nonOptions.size ())
219  {
220  // Add an unspecified non-option as a string
221  NS_LOG_LOGIC ("adding StringItem, NOCount:" << m_nonOptionCount
222  << ", NOSize:" << m_nonOptions.size ());
223  StringItem * item = new StringItem;
224  item->m_name = "extra-non-option-argument";
225  item->m_help = "Extra non-option argument encountered.";
226  item->m_value = value;
227  m_nonOptions.push_back (item);
228  }
229 
230  auto i = m_nonOptions[m_nonOptionCount];
231  if (!i->Parse (value))
232  {
233  std::cerr << "Invalid non-option argument value "
234  << value << " for " << i->m_name
235  << std::endl;
236  PrintHelp (std::cerr);
237  std::exit (1);
238  }
239  ++m_nonOptionCount;
240  return true;
241 }

571 void
572 CommandLine::HandleArgument (const std::string &name, const std::string &value) const
573 {
574  NS_LOG_FUNCTION (this << name << value);
575 
576  NS_LOG_DEBUG ("Handle arg name=" << name << " value=" << value);
577 
578  // Hard-coded options
579  if (name == "PrintHelp" || name == "help")
580  {
581  // method below never returns.
582  PrintHelp (std::cout);
583  std::exit (0);
584  }
585  else if (name == "PrintGroups")
586  {
587  // method below never returns.
588  PrintGroups (std::cout);
589  std::exit (0);
590  }
591  else if (name == "PrintTypeIds")
592  {
593  // method below never returns.
594  PrintTypeIds (std::cout);
595  std::exit (0);
596  }
597  else if (name == "PrintGlobals")
598  {
599  // method below never returns.
600  PrintGlobals (std::cout);
601  std::exit (0);
602  }
603  else if (name == "PrintGroup")
604  {
605  // method below never returns.
606  PrintGroup (std::cout, value);
607  std::exit (0);
608  }
609  else if (name == "PrintAttributes")
610  {
611  // method below never returns.
612  PrintAttributes (std::cout, value);
613  std::exit (0);
614  }
615  else
616  {
617  for (auto i : m_options)
618  {
619  if (i->m_name == name)
620  {
621  if (!i->Parse (value))
622  {
623  std::cerr << "Invalid argument value: "
624  << name << "=" << value << std::endl;
625  PrintHelp (std::cerr);
626  std::exit (1);
627  }
628  else
629  {
630  return;
631  }
632  }
633  }
634  }
635  // Global or ConfigPath options
636  if (!Config::SetGlobalFailSafe (name, StringValue (value))
637  && !Config::SetDefaultFailSafe (name, StringValue (value)))
638  {
639  std::cerr << "Invalid command-line arguments: --"
640  << name << "=" << value << std::endl;
641  PrintHelp (std::cerr);
642  std::exit (1);
643  }
644 }

667 void
668 CommandLine::AddValue (const std::string &name,
669  const std::string &attributePath)
670 {
671  NS_LOG_FUNCTION (this << name << attributePath);
672  // Attribute name is last token
673  std::size_t colon = attributePath.rfind ("::");
674  const std::string typeName = attributePath.substr (0, colon);
675  NS_LOG_DEBUG ("typeName: '" << typeName << "', colon: " << colon);
676 
677  TypeId tid;
678  if (!TypeId::LookupByNameFailSafe (typeName, &tid))
679  {
680  NS_FATAL_ERROR ("Unknown type=" << typeName);
681  }
682 
683  const std::string attrName = attributePath.substr (colon + 2);
684  struct TypeId::AttributeInformation info;
685  if (!tid.LookupAttributeByName (attrName, &info))
686  {
687  NS_FATAL_ERROR ("Attribute not found: " << attributePath);
688  }
689 
690  std::stringstream ss;
691  ss << info.help
692  << " (" << attributePath << ") ["
693  << info.initialValue->SerializeToString (info.checker) << "]";
694 
695  AddValue (name, ss.str (),
696  MakeBoundCallback (CommandLine::HandleAttribute, attributePath));

784 template <>
785 bool
786 CommandLineHelper::UserItemParse<bool> (const std::string value, bool & val)
787 {
788  std::string src = value;
789  std::transform (src.begin (), src.end (), src.begin (),
790  [](char c) {return static_cast<char> (std::tolower (c)); });
791  if (src.length () == 0)
792  {
793  val = !val;
794  return true;
795  }
796  else if ( (src == "true") || (src == "t") )
797  {
798  val = true;
799  return true;
800  }
801  else if ( (src == "false") || (src == "f"))
802  {
803  val = false;
804  return true;
805  }
806  else
807  {
808  std::istringstream iss;
809  iss.str (src);
810  iss >> val;
811  return !iss.bad () && !iss.fail ();
812  }

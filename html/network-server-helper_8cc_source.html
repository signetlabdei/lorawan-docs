83 Ptr<Application>
84 NetworkServerHelper::InstallPriv (Ptr<Node> node)
85 {
86  NS_LOG_FUNCTION (this << node);
87 
88  Ptr<NetworkServer> app = m_factory.Create<NetworkServer> ();
89 
90  app->SetNode (node);
91  node->AddApplication (app);
92 
93  // Cycle on each gateway
94  for (NodeContainer::Iterator i = m_gateways.Begin ();
95  i != m_gateways.End ();
96  i++)
97  {
98  // Add the connections with the gateway
99  // Create a PointToPoint link between gateway and NS
100  NetDeviceContainer container = p2pHelper.Install (node, *i);
101 
102  // Add the gateway to the NS list
103  app->AddGateway (*i, container.Get (0));
104  }
105 
106  // Link the NetworkServer to its NetDevices
107  for (uint32_t i = 0; i < node->GetNDevices (); i++)
108  {
109  Ptr<NetDevice> currentNetDevice = node->GetDevice (i);
110  currentNetDevice->SetReceiveCallback (MakeCallback
111  (&NetworkServer::Receive,
112  app));
113  }
114 
115  // Add the end devices
116  app->AddNodes (m_endDevices);
117 
118  // Add components to the NetworkServer
119  InstallComponents (app);
120 
121  return app;
122 }

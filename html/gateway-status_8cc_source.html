93 bool
94 GatewayStatus::IsAvailableForTransmission (double frequency)
95 {
96  // We can't send multiple packets at once, see SX1301 V2.01 page 29
97 
98  // Check that the gateway was not already "booked"
99  if (m_nextTransmissionTime > Simulator::Now () - MilliSeconds (1))
100  {
101  NS_LOG_INFO ("This gateway is already booked for a transmission");
102  return false;
103  }
104 
105  // Check that the gateway is not already in TX mode
106  if (m_gatewayMac->IsTransmitting ())
107  {
108  NS_LOG_INFO ("This gateway is currently transmitting");
109  return false;
110  }
111 
112  // Check that the gateway is not constrained by the duty cycle
113  Time waitingTime = m_gatewayMac->GetWaitingTime (frequency);
114  if (waitingTime > Seconds (0))
115  {
116  NS_LOG_INFO ("Gateway cannot be used because of duty cycle");
117  NS_LOG_INFO ("Waiting time at current GW: " << waitingTime.GetSeconds ()
118  << " seconds");
119 
120  return false;
121  }
122 
123  return true;
124 }

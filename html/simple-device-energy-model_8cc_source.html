84 double
85 SimpleDeviceEnergyModel::GetTotalEnergyConsumption (void) const
86 {
87  NS_LOG_FUNCTION (this);
88  Time duration = Simulator::Now () - m_lastUpdateTime;
89 
90  double energyToDecrease = 0.0;
91  double supplyVoltage = m_source->GetSupplyVoltage ();
92  energyToDecrease = duration.GetSeconds () * m_actualCurrentA * supplyVoltage;
93 
94  m_source->UpdateEnergySource ();
95 
96  return m_totalEnergyConsumption + energyToDecrease;
97 }

99 void
100 SimpleDeviceEnergyModel::SetCurrentA (double current)
101 {
102  NS_LOG_FUNCTION (this << current);
103  Time duration = Simulator::Now () - m_lastUpdateTime;
104 
105  double energyToDecrease = 0.0;
106  double supplyVoltage = m_source->GetSupplyVoltage ();
107  energyToDecrease = duration.GetSeconds () * current * supplyVoltage;
108 
109  // update total energy consumption
110  m_totalEnergyConsumption += energyToDecrease;
111  // update last update time stamp
112  m_lastUpdateTime = Simulator::Now ();
113  // notify energy source
114  m_source->UpdateEnergySource ();
115  // update the current drain
116  m_actualCurrentA = current;
117 }

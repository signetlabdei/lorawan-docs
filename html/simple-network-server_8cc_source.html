68 SimpleNetworkServer::AddGateway (Ptr<Node> gateway, Ptr<NetDevice> netDevice)
69 {
70  NS_LOG_FUNCTION (this << gateway);
71 
72  // Get the PointToPointNetDevice
73  Ptr<PointToPointNetDevice> p2pNetDevice;
74  for (uint32_t i = 0; i < gateway->GetNDevices (); i++)
75  {
76  p2pNetDevice = gateway->GetDevice (i)->GetObject<PointToPointNetDevice> ();
77  if (p2pNetDevice != 0)
78  {
79  // We found a p2pNetDevice on the gateway
80  break;
81  }
82  }
83 
84  // Get the gateway's LoRa MAC layer (assumes gateway's MAC is configured as first device)
85  Ptr<GatewayLoraMac> gwMac = gateway->GetDevice (0)->GetObject<LoraNetDevice> ()->
86  GetMac ()->GetObject<GatewayLoraMac> ();
87  NS_ASSERT (gwMac != 0);
88 
89  // Get the Address
90  Address gatewayAddress = p2pNetDevice->GetAddress ();
91 
92  // Check whether this device already exists
93  if (m_gatewayStatuses.find (gatewayAddress) == m_gatewayStatuses.end ())
94  {
95  // The device doesn't exist
96 
97  // Create new gatewayStatus
98  GatewayStatus gwStatus = GatewayStatus (gatewayAddress, netDevice, gwMac);
99  // Add it to the map
100  m_gatewayStatuses.insert (std::pair<Address, GatewayStatus>
101  (gatewayAddress, gwStatus));
102  NS_LOG_DEBUG ("Added a gateway to the list");
103  }
104 }

155 bool
156 SimpleNetworkServer::Receive (Ptr<NetDevice> device, Ptr<const Packet> packet,
157  uint16_t protocol, const Address& address)
158 {
159  NS_LOG_FUNCTION (this << packet << protocol << address);
160 
161  // Create a copy of the packet
162  Ptr<Packet> myPacket = packet->Copy ();
163 
164  // Extract the headers
165  LoraMacHeader macHdr;
166  myPacket->RemoveHeader (macHdr);
167  LoraFrameHeader frameHdr;
168  frameHdr.SetAsUplink ();
169  myPacket->RemoveHeader (frameHdr);
170  LoraTag tag;
171  myPacket->RemovePacketTag (tag);
172 
173  // Register which gateway this packet came from
174  double rcvPower = tag.GetReceivePower ();
175  m_deviceStatuses.at (frameHdr.GetAddress ()).UpdateGatewayData (address,
176  rcvPower);
177 
178  // Determine whether the packet requires a reply
179  if (macHdr.GetMType () == LoraMacHeader::CONFIRMED_DATA_UP &&
180  !m_deviceStatuses.at (frameHdr.GetAddress ()).HasReply ())
181  {
182  NS_LOG_DEBUG ("Scheduling a reply for this device");
183 
184  DeviceStatus::Reply reply;
185  reply.hasReply = true;
186 
187  LoraMacHeader replyMacHdr = LoraMacHeader ();
188  replyMacHdr.SetMajor (0);
189  replyMacHdr.SetMType (LoraMacHeader::UNCONFIRMED_DATA_DOWN);
190  reply.macHeader = replyMacHdr;
191 
192  LoraFrameHeader replyFrameHdr = LoraFrameHeader ();
193  replyFrameHdr.SetAsDownlink ();
194  replyFrameHdr.SetAddress (frameHdr.GetAddress ());
195  replyFrameHdr.SetAck (true);
196  reply.frameHeader = replyFrameHdr;
197 
198  Ptr<Packet> replyPacket = Create<Packet> (10);
199  reply.packet = replyPacket;
200 
201  m_deviceStatuses.at (frameHdr.GetAddress ()).SetReply (reply);
202  m_deviceStatuses.at (frameHdr.GetAddress ()).SetFirstReceiveWindowFrequency (tag.GetFrequency ());
203 
204  // Schedule a reply on the first receive window
205  Simulator::Schedule (Seconds (1), &SimpleNetworkServer::SendOnFirstWindow,
206  this, frameHdr.GetAddress ());
207  }
208 
209  return true;
210 }

212 void
213 SimpleNetworkServer::SendOnFirstWindow (LoraDeviceAddress address)
214 {
215  NS_LOG_FUNCTION (this << address);
216 
217  // Decide on which gateway we'll transmit our reply
218  double firstReceiveWindowFrequency = m_deviceStatuses.at
219  (address).GetFirstReceiveWindowFrequency ();
220 
221  Address gatewayForReply = GetGatewayForReply (address,
222  firstReceiveWindowFrequency);
223 
224  if (gatewayForReply != Address ())
225  {
226  NS_LOG_FUNCTION ("Found a suitable GW!");
227 
228  // Get the packet to use in the reply
229  Ptr<Packet> replyPacket = m_deviceStatuses.at (address).GetReplyPacket ();
230  NS_LOG_DEBUG ("Packet size: " << replyPacket->GetSize ());
231 
232  // Tag the packet so that the Gateway sends it according to the first
233  // receive window parameters
234  LoraTag replyPacketTag;
235  uint8_t dataRate = m_deviceStatuses.at (address).GetFirstReceiveWindowDataRate ();
236  double frequency = m_deviceStatuses.at (address).GetFirstReceiveWindowFrequency ();
237  replyPacketTag.SetDataRate (dataRate);
238  replyPacketTag.SetFrequency (frequency);
239 
240  replyPacket->AddPacketTag (replyPacketTag);
241 
242  NS_LOG_INFO ("Sending reply through the gateway with address " << gatewayForReply);
243 
244  // Inform the gateway of the transmission
245  m_gatewayStatuses.find (gatewayForReply)->second.GetNetDevice ()->
246  Send (replyPacket, gatewayForReply, 0x0800);
247  }
248  else
249  {
250  NS_LOG_FUNCTION ("No suitable GW found, scheduling second window reply");
251 
252  // Schedule a reply on the second receive window
253  Simulator::Schedule (Seconds (1), &SimpleNetworkServer::SendOnSecondWindow, this,
254  address);
255  }

299 Address
300 SimpleNetworkServer::GetGatewayForReply (LoraDeviceAddress deviceAddress,
301  double frequency)
302 {
303  NS_LOG_FUNCTION (this);
304 
305  // Check which gateways can send this reply
306  // Go in the order suggested by the DeviceStatus
307  std::list<Address> addresses = m_deviceStatuses.at
308  (deviceAddress).GetSortedGatewayAddresses ();
309 
310  for (auto it = addresses.begin (); it != addresses.end (); ++it)
311  {
312  if (m_gatewayStatuses.at (*it).IsAvailableForTransmission (frequency))
313  {
314  m_gatewayStatuses.at (*it).SetNextTransmissionTime (Simulator::Now ());
315  return *it;
316  }
317  }
318 
319  return Address ();
320 }

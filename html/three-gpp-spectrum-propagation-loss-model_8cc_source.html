90 ThreeGppAntennaArrayModel::ComplexVector
91 ThreeGppSpectrumPropagationLossModel::CalcLongTerm (Ptr<const ThreeGppChannelModel::ThreeGppChannelMatrix> params,
92  const ThreeGppAntennaArrayModel::ComplexVector &sW,
93  const ThreeGppAntennaArrayModel::ComplexVector &uW) const
94 {
95  NS_LOG_FUNCTION (this);
96 
97  uint16_t sAntenna = static_cast<uint16_t> (sW.size ());
98  uint16_t uAntenna = static_cast<uint16_t> (uW.size ());
99 
100  NS_LOG_DEBUG ("CalcLongTerm with sAntenna " << sAntenna << " uAntenna " << uAntenna);
101  //store the long term part to reduce computation load
102  //only the small scale fading needs to be updated if the large scale parameters and antenna weights remain unchanged.
103  ThreeGppAntennaArrayModel::ComplexVector longTerm;
104  uint8_t numCluster = static_cast<uint8_t> (params->m_channel[0][0].size ());
105 
106  for (uint8_t cIndex = 0; cIndex < numCluster; cIndex++)
107  {
108  std::complex<double> txSum (0,0);
109  for (uint16_t sIndex = 0; sIndex < sAntenna; sIndex++)
110  {
111  std::complex<double> rxSum (0,0);
112  for (uint16_t uIndex = 0; uIndex < uAntenna; uIndex++)
113  {
114  rxSum = rxSum + uW[uIndex] * params->m_channel[uIndex][sIndex][cIndex];
115  }
116  txSum = txSum + sW[sIndex] * rxSum;
117  }
118  longTerm.push_back (txSum);
119  }
120  return longTerm;
121 }

123 Ptr<SpectrumValue>
124 ThreeGppSpectrumPropagationLossModel::CalcBeamformingGain (Ptr<SpectrumValue> txPsd,
125  ThreeGppAntennaArrayModel::ComplexVector longTerm,
126  Ptr<const ThreeGppChannelModel::ThreeGppChannelMatrix> params,
127  const ns3::Vector &sSpeed, const ns3::Vector &uSpeed) const
128 {
129  NS_LOG_FUNCTION (this);
130 
131  Ptr<SpectrumValue> tempPsd = Copy<SpectrumValue> (txPsd);
132 
133  //channel[rx][tx][cluster]
134  uint8_t numCluster = static_cast<uint8_t> (params->m_channel[0][0].size ());
135 
136  // compute the doppler term
137  // NOTE the update of Doppler is simplified by only taking the center angle of
138  // each cluster in to consideration.
139  double slotTime = Simulator::Now ().GetSeconds ();
140  ThreeGppAntennaArrayModel::ComplexVector doppler;
141  for (uint8_t cIndex = 0; cIndex < numCluster; cIndex++)
142  {
143  //cluster angle angle[direction][n],where, direction = 0(aoa), 1(zoa).
144  // TODO should I include the "alfa" term for the Doppler of delayed paths?
145  double temp_doppler = 2 * M_PI * ((sin (params->m_angle[ThreeGppChannelModel::ZOA_INDEX][cIndex] * M_PI / 180) * cos (params->m_angle[ThreeGppChannelModel::AOA_INDEX][cIndex] * M_PI / 180) * uSpeed.x
146  + sin (params->m_angle[ThreeGppChannelModel::ZOA_INDEX][cIndex] * M_PI / 180) * sin (params->m_angle[ThreeGppChannelModel::AOA_INDEX][cIndex] * M_PI / 180) * uSpeed.y
147  + cos (params->m_angle[ThreeGppChannelModel::ZOA_INDEX][cIndex] * M_PI / 180) * uSpeed.z)
148  + (sin (params->m_angle[ThreeGppChannelModel::ZOD_INDEX][cIndex] * M_PI / 180) * cos (params->m_angle[ThreeGppChannelModel::AOD_INDEX][cIndex] * M_PI / 180) * sSpeed.x
149  + sin (params->m_angle[ThreeGppChannelModel::ZOD_INDEX][cIndex] * M_PI / 180) * sin (params->m_angle[ThreeGppChannelModel::AOD_INDEX][cIndex] * M_PI / 180) * sSpeed.y
150  + cos (params->m_angle[ThreeGppChannelModel::ZOD_INDEX][cIndex] * M_PI / 180) * sSpeed.z))
151  * slotTime * GetFrequency () / 3e8;
152  doppler.push_back (exp (std::complex<double> (0, temp_doppler)));
153  }
154 
155  // apply the doppler term and the propagation delay to the long term component
156  // to obtain the beamforming gain
157  auto vit = tempPsd->ValuesBegin (); // psd iterator
158  auto sbit = tempPsd->ConstBandsBegin(); // band iterator
159  while (vit != tempPsd->ValuesEnd ())
160  {
161  std::complex<double> subsbandGain (0.0,0.0);
162  if ((*vit) != 0.00)
163  {
164  double fsb = (*sbit).fc; // center frequency of the sub-band
165  for (uint8_t cIndex = 0; cIndex < numCluster; cIndex++)
166  {
167  double delay = -2 * M_PI * fsb * (params->m_delay[cIndex]);
168  subsbandGain = subsbandGain + longTerm[cIndex] * doppler[cIndex] * exp (std::complex<double> (0, delay));
169  }
170  *vit = (*vit) * (norm (subsbandGain));
171  }
172  vit++;
173  sbit++;
174  }
175  return tempPsd;
176 }

178 ThreeGppAntennaArrayModel::ComplexVector
179 ThreeGppSpectrumPropagationLossModel::GetLongTerm (uint32_t aId, uint32_t bId,
180  Ptr<const ThreeGppChannelModel::ThreeGppChannelMatrix> channelMatrix,
181  const ThreeGppAntennaArrayModel::ComplexVector &aW,
182  const ThreeGppAntennaArrayModel::ComplexVector &bW) const
183 {
184  ThreeGppAntennaArrayModel::ComplexVector longTerm; // vector containing the long term component for each cluster
185 
186  // check if the channel matrix was generated considering a as the s-node and
187  // b as the u-node or viceversa
188  ThreeGppAntennaArrayModel::ComplexVector sW, uW;
189  if (!channelMatrix->IsReverse (aId, bId))
190  {
191  sW = aW;
192  uW = bW;
193  }
194  else
195  {
196  sW = bW;
197  uW = aW;
198  }
199 
200  // compute the long term key, the key is unique for each tx-rx pair
201  uint32_t x1 = std::min (aId, bId);
202  uint32_t x2 = std::max (aId, bId);
203  uint32_t longTermId = ThreeGppChannelModel::GetKey (x1, x2);
204 
205  bool update = false; // indicates whether the long term has to be updated
206  bool notFound = false; // indicates if the long term has not been computed yet
207 
208  // look for the long term in the map and check if it is valid
209  if (m_longTermMap.find (longTermId) != m_longTermMap.end ())
210  {
211  NS_LOG_DEBUG ("found the long term component in the map");
212  longTerm = m_longTermMap[longTermId]->m_longTerm;
213 
214  // check if the channel matrix has been updated
215  // or the s beam has been changed
216  // or the u beam has been changed
217  update = (m_longTermMap[longTermId]->m_channel->m_generatedTime != channelMatrix->m_generatedTime
218  || m_longTermMap[longTermId]->m_sW != sW
219  || m_longTermMap[longTermId]->m_uW != uW);
220 
221  }
222  else
223  {
224  NS_LOG_DEBUG ("long term component NOT found");
225  notFound = true;
226  }
227 
228  if (update || notFound)
229  {
230  NS_LOG_DEBUG ("compute the long term");
231  // compute the long term component
232  longTerm = CalcLongTerm (channelMatrix, sW, uW);
233 
234  // store the long term
235  Ptr<LongTerm> longTermItem = Create<LongTerm> ();
236  longTermItem->m_longTerm = longTerm;
237  longTermItem->m_channel = channelMatrix;
238  longTermItem->m_sW = sW;
239  longTermItem->m_uW = uW;
240 
241  m_longTermMap[longTermId] = longTermItem;
242  }
243 
244  return longTerm;
245 }

247 Ptr<SpectrumValue>
248 ThreeGppSpectrumPropagationLossModel::DoCalcRxPowerSpectralDensity (Ptr<const SpectrumValue> txPsd,
249  Ptr<const MobilityModel> a,
250  Ptr<const MobilityModel> b) const
251 {
252  NS_LOG_FUNCTION (this);
253  uint32_t aId = a->GetObject<Node> ()->GetId (); // id of the node a
254  uint32_t bId = b->GetObject<Node> ()->GetId (); // id of the node b
255 
256  NS_ASSERT (aId != bId);
257  NS_ASSERT_MSG (a->GetDistanceFrom (b) > 0.0, "The position of a and b devices cannot be the same");
258 
259  Ptr<SpectrumValue> rxPsd = Copy<SpectrumValue> (txPsd);
260 
261  // retrieve the antenna of device a
262  NS_ASSERT_MSG (m_deviceAntennaMap.find (aId) != m_deviceAntennaMap.end (), "Antenna not found for node " << aId);
263  Ptr<const ThreeGppAntennaArrayModel> aAntenna = m_deviceAntennaMap.at (aId);
264  NS_LOG_DEBUG ("a node " << a->GetObject<Node> () << " antenna " << aAntenna);
265 
266  // retrieve the antenna of the device b
267  NS_ASSERT_MSG (m_deviceAntennaMap.find (bId) != m_deviceAntennaMap.end (), "Antenna not found for device " << bId);
268  Ptr<const ThreeGppAntennaArrayModel> bAntenna = m_deviceAntennaMap.at (bId);
269  NS_LOG_DEBUG ("b node " << bId << " antenna " << bAntenna);
270 
271  if (aAntenna->IsOmniTx () || bAntenna->IsOmniTx () )
272  {
273  NS_LOG_LOGIC ("Omni transmission, do nothing.");
274  return rxPsd;
275  }
276 
277  Ptr<const ThreeGppChannelModel::ThreeGppChannelMatrix> channelMatrix = m_channelModel->GetChannel (a, b, aAntenna, bAntenna);
278 
279  // get the precoding and combining vectors
280  ThreeGppAntennaArrayModel::ComplexVector aW = aAntenna->GetBeamformingVector ();
281  ThreeGppAntennaArrayModel::ComplexVector bW = bAntenna->GetBeamformingVector ();
282 
283  // retrieve the long term component
284  ThreeGppAntennaArrayModel::ComplexVector longTerm = GetLongTerm (aId, bId, channelMatrix, aW, bW);
285 
286  // apply the beamforming gain
287  rxPsd = CalcBeamformingGain (rxPsd, longTerm, channelMatrix, a->GetVelocity (), b->GetVelocity ());
288 
289  return rxPsd;
290 }

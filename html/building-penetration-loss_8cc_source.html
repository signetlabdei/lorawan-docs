56 double
57 BuildingPenetrationLoss::DoCalcRxPower (double txPowerDbm,
58  Ptr<MobilityModel> a,
59  Ptr<MobilityModel> b) const
60 {
61  NS_LOG_FUNCTION (this << txPowerDbm << a << b);
62 
63  Ptr<MobilityBuildingInfo> a1 = a->GetObject<MobilityBuildingInfo> ();
64  Ptr<MobilityBuildingInfo> b1 = b->GetObject<MobilityBuildingInfo> ();
65 
66  // These are the components of the loss due to building penetration
67  double externalWallLoss = 0;
68  double tor1 = 0;
69  double tor3 = 0;
70  double gfh = 0;
71 
72  // Go through various cases in which a and b are indoors or outdoors
73  if ((b1->IsIndoor () && !a1->IsIndoor ()))
74  {
75  NS_LOG_INFO ("Tx is outdoors and Rx is indoors");
76 
77  externalWallLoss = GetWallLoss (b); // External wall loss due to b
78  tor1 = GetTor1 (b); // Internal wall loss due to b
79  tor3 = 0.6 * m_uniformRV->GetValue (0, 15);
80  gfh = 0;
81 
82  }
83  else if ((!b1->IsIndoor () && a1->IsIndoor ()))
84  {
85  NS_LOG_INFO ("Rx is outdoors and Tx is indoors");
86 
87  // These are the components of the loss due to building penetration
88  externalWallLoss = GetWallLoss (a);
89  tor1 = GetTor1 (a);
90  tor3 = 0.6 * m_uniformRV->GetValue (0, 15);
91  gfh = 0;
92 
93  }
94  else if (!a1->IsIndoor ()&& !b1->IsIndoor ())
95  {
96  NS_LOG_DEBUG ("No penetration loss since both devices are outside");
97  }
98  else if (a1->IsIndoor ()&& b1->IsIndoor ())
99  {
100  // They are in the same building
101  if (a1->GetBuilding () == b1->GetBuilding ())
102  {
103  NS_LOG_INFO ("Devices are in the same building");
104  // Only internal wall loss
105  tor1 = GetTor1 (b);
106  tor3 = 0.6 * m_uniformRV->GetValue (0, 15);
107  }
108  // They are in different buildings
109  else
110  {
111  // These are the components of the loss due to building penetration
112  externalWallLoss = GetWallLoss (b) + GetWallLoss (a);
113  tor1 = GetTor1 (b) + GetTor1 (a);
114  tor3 = 0.6 * m_uniformRV->GetValue (0, 15);
115  gfh = 0;
116  }
117  }
118 
119  NS_LOG_DEBUG ("Building penetration loss:" <<
120  " externalWallLoss = " << externalWallLoss <<
121  ", tor1 = " << tor1 <<
122  ", tor3 = " << tor3 <<
123  ", GFH = " << gfh);
124 
125  // Put together all the pieces
126  double loss = externalWallLoss + std::max (tor1, tor3) - gfh;
127 
128  NS_LOG_DEBUG ("Total loss due to building penetration: " << loss);
129 
130  return txPowerDbm - loss;
131 }

147 
148  // Distribution is specified in TR 45.820, page 482, first scenario

190 double
191 BuildingPenetrationLoss::GetWallLoss (Ptr<MobilityModel> b) const
192 {
193  NS_LOG_FUNCTION (this << b);
194 
195  std::map<Ptr<MobilityModel>, int>::const_iterator it;
196 
197  // Check whether the b device already has a wall loss value
198  it = m_wallLossMap.find (b);
199  if (it == m_wallLossMap.end ())
200  {
201  // Create a random value and insert it on the map
202  m_wallLossMap[b] = GetWallLossValue ();
203  NS_LOG_DEBUG ("Inserted a new wall loss value: " <<
204  m_wallLossMap.find (b)->second);
205  }
206 
207  switch (m_wallLossMap.find (b)->second)
208  {
209  case 0:
210  return m_uniformRV->GetValue (4, 11);
211  case 1:
212  return m_uniformRV->GetValue (11, 19);
213  case 2:
214  return m_uniformRV->GetValue (19, 23);
215  }
216 
217  // Case in which something goes wrong
218  return 0;
219 }

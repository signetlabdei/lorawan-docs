98 static SimulatorImpl * GetImpl (void)
99 {
100  SimulatorImpl **pimpl = PeekImpl ();
101  /* Please, don't include any calls to logging macros in this function
102  * or pay the price, that is, stack explosions.
103  */
104  if (*pimpl == 0)
105  {
106  {
107  ObjectFactory factory;
108  StringValue s;
109 
110  g_simTypeImpl.GetValue (s);
111  factory.SetTypeId (s.Get ());
112  *pimpl = GetPointer (factory.Create<SimulatorImpl> ());
113  }
114  {
115  ObjectFactory factory;
116  StringValue s;
117  g_schedTypeImpl.GetValue (s);
118  factory.SetTypeId (s.Get ());
119  (*pimpl)->SetScheduler (factory);
120  }
121 
122 //
123 // Note: we call LogSetTimePrinter _after_ creating the implementation
124 // object because the act of creation can trigger calls to the logging
125 // framework which would call the TimePrinter function which would call
126 // Simulator::Now which would call Simulator::GetImpl, and, thus, get us
127 // in an infinite recursion until the stack explodes.
128 //
129  LogSetTimePrinter (&DefaultTimePrinter);
130  LogSetNodePrinter (&DefaultNodePrinter);
131  }
132  return *pimpl;
133 }

135 void
136 Simulator::Destroy (void)
137 {
138  NS_LOG_FUNCTION_NOARGS ();
139 
140  SimulatorImpl **pimpl = PeekImpl ();
141  if (*pimpl == 0)
142  {
143  return;
144  }
145  /* Note: we have to call LogSetTimePrinter (0) below because if we do not do
146  * this, and restart a simulation after this call to Destroy, (which is
147  * legal), Simulator::GetImpl will trigger again an infinite recursion until
148  * the stack explodes.
149  */
150  LogSetTimePrinter (0);
151  LogSetNodePrinter (0);
152  (*pimpl)->Destroy ();
153  (*pimpl)->Unref ();
154  *pimpl = 0;
155 }

350 void
351 Simulator::SetImplementation (Ptr<SimulatorImpl> impl)
352 {
353  NS_LOG_FUNCTION (impl);
354  if (*PeekImpl () != 0)
355  {
356  NS_FATAL_ERROR ("It is not possible to set the implementation after calling any Simulator:: function. Call Simulator::SetImplementation earlier or after Simulator::Destroy.");
357  }
358  *PeekImpl () = GetPointer (impl);
359  // Set the default scheduler
360  ObjectFactory factory;
361  StringValue s;
362  g_schedTypeImpl.GetValue (s);
363  factory.SetTypeId (s.Get ());
364  impl->SetScheduler (factory);
365 //
366 // Note: we call LogSetTimePrinter _after_ creating the implementation
367 // object because the act of creation can trigger calls to the logging
368 // framework which would call the TimePrinter function which would call
369 // Simulator::Now which would call Simulator::GetImpl, and, thus, get us
370 // in an infinite recursion until the stack explodes.
371 //
372  LogSetTimePrinter (&DefaultTimePrinter);
373  LogSetNodePrinter (&DefaultNodePrinter);
374 }

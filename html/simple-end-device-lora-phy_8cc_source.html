55 void
56 SimpleEndDeviceLoraPhy::Send (Ptr<Packet> packet, LoraTxParameters txParams,
57  double frequencyMHz, double txPowerDbm)
58 {
59  NS_LOG_FUNCTION (this << packet << txParams << frequencyMHz << txPowerDbm);
60 
61  NS_LOG_INFO ("Current state: " << m_state);
62 
63  // We must be either in STANDBY or SLEEP mode to send a packet
64  if (m_state != STANDBY && m_state != SLEEP)
65  {
66  NS_LOG_INFO ("Cannot send because device is currently not in STANDBY or SLEEP mode");
67  return;
68  }
69 
70  // Compute the duration of the transmission
71  Time duration = GetOnAirTime (packet, txParams);
72 
73  // We can send the packet: switch to the TX state
74  SwitchToTx (txPowerDbm);
75 
76  // Tag the packet with information about its Spreading Factor
77  LoraTag tag;
78  packet->RemovePacketTag (tag);
79  tag.SetSpreadingFactor (txParams.sf);
80  packet->AddPacketTag (tag);
81 
82  // Send the packet over the channel
83  NS_LOG_INFO ("Sending the packet in the channel");
84  m_channel->Send (this, packet, txPowerDbm, txParams, duration, frequencyMHz);
85 
86  // Schedule the switch back to STANDBY mode.
87  // For reference see SX1272 datasheet, section 4.1.6
88  Simulator::Schedule (duration, &EndDeviceLoraPhy::SwitchToStandby, this);
89 
90  // Schedule the txFinished callback, if it was set
91  // The call is scheduled just after the switch to standby in case the upper
92  // layer wishes to change the state. This ensures that it will find a PHY in
93  // STANDBY mode.
94  if (!m_txFinishedCallback.IsNull ())
95  {
96  Simulator::Schedule (duration + NanoSeconds (10),
97  &SimpleEndDeviceLoraPhy::m_txFinishedCallback, this,
98  packet);
99  }
100 
101 
102  // Call the trace source
103  if (m_device)
104  {
105  m_startSending (packet, m_device->GetNode ()->GetId ());
106  }
107  else
108  {
109  m_startSending (packet, 0);
110  }
111 }

113 void
114 SimpleEndDeviceLoraPhy::StartReceive (Ptr<Packet> packet, double rxPowerDbm,
115  uint8_t sf, Time duration, double frequencyMHz)
116 {
117 
118  NS_LOG_FUNCTION (this << packet << rxPowerDbm << unsigned (sf) << duration <<
119  frequencyMHz);
120 
121  // Notify the LoraInterferenceHelper of the impinging signal, and remember
122  // the event it creates. This will be used then to correctly handle the end
123  // of reception event.
124  //
125  // We need to do this regardless of our state or frequency, since these could
126  // change (and making the interference relevant) while the interference is
127  // still incoming.
128 
129  Ptr<LoraInterferenceHelper::Event> event;
130  event = m_interference.Add (duration, rxPowerDbm, sf, packet, frequencyMHz);
131 
132  // Switch on the current PHY state
133  switch (m_state)
134  {
135  // In the SLEEP, TX and RX cases we cannot receive the packet: we only add
136  // it to the list of interferers and do not schedule an EndReceive event for
137  // it.
138  case SLEEP:
139  {
140  NS_LOG_INFO ("Dropping packet because device is in SLEEP state");
141  break;
142  }
143  case TX:
144  {
145  NS_LOG_INFO ("Dropping packet because device is in TX state");
146  break;
147  }
148  case RX:
149  {
150  NS_LOG_INFO ("Dropping packet because device is already in RX state");
151  break;
152  }
153  // If we are in STANDBY mode, we can potentially lock on the currently
154  // incoming transmission
155  case STANDBY:
156  {
157  // There are a series of properties the packet needs to respect in order
158  // for us to be able to lock on it:
159  // - It's on frequency we are listening on
160  // - It uses the SF we are configured to look for
161  // - Its receive power is above the device sensitivity for that SF
162 
163  // Flag to signal whether we can receive the packet or not
164  bool canLockOnPacket = true;
165 
166  // Save needed sensitivity
167  double sensitivity = EndDeviceLoraPhy::sensitivity[unsigned(sf) - 7];
168 
169  // Check frequency
171  if (!IsOnFrequency (frequencyMHz))
172  {
173  NS_LOG_INFO ("Packet lost because it's on frequency " <<
174  frequencyMHz << " MHz and we are listening at " <<
175  m_frequency << " MHz");
176 
177  // Fire the trace source for this event.
178  if (m_device)
179  {
180  m_wrongFrequency (packet, m_device->GetNode ()->GetId ());
181  }
182  else
183  {
184  m_wrongFrequency (packet, 0);
185  }
186 
187  canLockOnPacket = false;
188  }
189 
190  // Check Spreading Factor
192  if (sf != m_sf)
193  {
194  NS_LOG_INFO ("Packet lost because it's using SF" << unsigned(sf) <<
195  ", while we are listening for SF" << unsigned(m_sf));
196 
197  // Fire the trace source for this event.
198  if (m_device)
199  {
200  m_wrongSf (packet, m_device->GetNode ()->GetId ());
201  }
202  else
203  {
204  m_wrongSf (packet, 0);
205  }
206 
207  canLockOnPacket = false;
208  }
209 
210  // Check Sensitivity
212  if (rxPowerDbm < sensitivity)
213  {
214  NS_LOG_INFO ("Dropping packet reception of packet with sf = " <<
215  unsigned(sf) << " because under the sensitivity of " <<
216  sensitivity << " dBm");
217 
218  // Fire the trace source for this event.
219  if (m_device)
220  {
221  m_underSensitivity (packet, m_device->GetNode ()->GetId ());
222  }
223  else
224  {
225  m_underSensitivity (packet, 0);
226  }
227 
228  canLockOnPacket = false;
229  }
230 
231  // Check if one of the above failed
233  if (canLockOnPacket)
234  {
235  // Switch to RX state
236  // EndReceive will handle the switch back to STANDBY state
237  SwitchToRx ();
238 
239  // Schedule the end of the reception of the packet
240  NS_LOG_INFO ("Scheduling reception of a packet. End in " <<
241  duration.GetSeconds () << " seconds");
242 
243  Simulator::Schedule (duration, &LoraPhy::EndReceive, this, packet,
244  event);
245 
246  // Fire the beginning of reception trace source
247  m_phyRxBeginTrace (packet);
248  }
249  }
250  }

253 void
254 SimpleEndDeviceLoraPhy::EndReceive (Ptr<Packet> packet,
255  Ptr<LoraInterferenceHelper::Event> event)
256 {
257  NS_LOG_FUNCTION (this << packet << event);
258 
259  // Automatically switch to Standby in either case
260  SwitchToStandby ();
261 
262  // Fire the trace source
263  m_phyRxEndTrace (packet);
264 
265  // Call the LoraInterferenceHelper to determine whether there was destructive
266  // interference on this event.
267  bool packetDestroyed = m_interference.IsDestroyedByInterference (event);
268 
269  // Fire the trace source if packet was destroyed
270  if (packetDestroyed)
271  {
272  NS_LOG_INFO ("Packet destroyed by interference");
273 
274  if (m_device)
275  {
276  m_interferedPacket (packet, m_device->GetNode ()->GetId ());
277  }
278  else
279  {
280  m_interferedPacket (packet, 0);
281  }
282 
283  // If there is one, perform the callback to inform the upper layer of the
284  // lost packet
285  if (!m_rxFailedCallback.IsNull ())
286  {
287  m_rxFailedCallback (packet);
288  }
289 
290  }
291  else
292  {
293  NS_LOG_INFO ("Packet received correctly");
294 
295  if (m_device)
296  {
297  m_successfullyReceivedPacket (packet, m_device->GetNode ()->GetId ());
298  }
299  else
300  {
301  m_successfullyReceivedPacket (packet, 0);
302  }
303 
304  // If there is one, perform the callback to inform the upper layer
305  if (!m_rxOkCallback.IsNull ())
306  {
307  m_rxOkCallback (packet);
308  }
309 
310  }
311 }

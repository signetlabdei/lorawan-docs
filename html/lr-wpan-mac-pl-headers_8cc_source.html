202 CommandPayloadHeader::MacCommand
203 CommandPayloadHeader::GetCommandFrameType (void) const
204 {
205  switch (m_cmdFrameId)
206  {
207  case 0x01:
208  return ASSOCIATION_REQ;
209  break;
210  case 0x02:
211  return ASSOCIATION_RESP;
212  break;
213  case 0x03:
214  return DISASSOCIATION_NOTIF;
215  break;
216  case 0x04:
217  return DATA_REQ;
218  break;
219  case 0x05:
220  return PANID_CONFLICT;
221  break;
222  case 0x06:
223  return ORPHAN_NOTIF;
224  break;
225  case 0x07:
226  return BEACON_REQ;
227  break;
228  case 0x08:
229  return COOR_REALIGN;
230  break;
231  case 0x09:
232  return GTS_REQ;
233  break;
234  default:
235  return CMD_RESERVED;
236  }

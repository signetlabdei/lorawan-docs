110 double
111 ThreeGppPropagationLossModel::DoCalcRxPower (double txPowerDbm,
112  Ptr<MobilityModel> a,
113  Ptr<MobilityModel> b) const
114 {
115  NS_LOG_FUNCTION (this);
116 
117  // check if the model is initialized
118  NS_ASSERT_MSG (m_frequency != 0.0, "First set the centre frequency");
119 
120  // retrieve the channel condition
121  NS_ASSERT_MSG (m_channelConditionModel, "First set the channel condition model");
122  Ptr<ChannelCondition> cond = m_channelConditionModel->GetChannelCondition (a, b);
123 
124  // compute the 2D distance between a and b
125  double distance2d = Calculate2dDistance (a->GetPosition (), b->GetPosition ());
126 
127  // compute the 3D distance between a and b
128  double distance3d = CalculateDistance (a->GetPosition (), b->GetPosition ());
129 
130  // compute hUT and hBS
131  std::pair<double, double> heights = GetUtAndBsHeights (a->GetPosition ().z, b->GetPosition ().z);
132 
133  double rxPow = txPowerDbm;
134  if (cond->GetLosCondition () == ChannelCondition::LosConditionValue::LOS)
135  {
136  rxPow -= GetLossLos (distance2d, distance3d, heights.first, heights.second);
137  NS_LOG_DEBUG ("Channel codition is LOS, rxPower = " << rxPow);
138  }
139  else if (cond->GetLosCondition () == ChannelCondition::LosConditionValue::NLOS)
140  {
141  rxPow -= GetLossNlos (distance2d, distance3d, heights.first, heights.second);
142  NS_LOG_DEBUG ("Channel codition is NLOS, rxPower = " << rxPow);
143  }
144  else
145  {
146  NS_FATAL_ERROR ("Unknown channel condition");
147  }
148 
149  if (m_shadowingEnabled)
150  {
151  rxPow -= GetShadowing (a, b, cond->GetLosCondition ());
152  }
153 
154  return rxPow;
155 }

157 double
158 ThreeGppPropagationLossModel::GetShadowing (Ptr<MobilityModel> a, Ptr<MobilityModel> b, ChannelCondition::LosConditionValue cond) const
159 {
160  NS_LOG_FUNCTION (this);
161 
162  double shadowingValue;
163 
164  // compute the channel key
165  uint32_t key = GetKey (a, b);
166 
167  bool notFound = false; // indicates if the shadowing value has not been computed yet
168  bool newCondition = false; // indicates if the channel condition has changed
169  Vector newDistance; // the distance vector, that is not a distance but a difference
170  auto it = m_shadowingMap.end (); // the shadowing map iterator
171  if (m_shadowingMap.find (key) != m_shadowingMap.end ())
172  {
173  // found the shadowing value in the map
174  it = m_shadowingMap.find (key);
175  newDistance = GetVectorDifference (a, b);
176  newCondition = (it->second.m_condition != cond); // true if the condition changed
177  }
178  else
179  {
180  notFound = true;
181 
182  // add a new entry in the map and update the iterator
183  ShadowingMapItem newItem;
184  it = m_shadowingMap.insert (it, std::make_pair (key, newItem));
185  }
186 
187  if (notFound || newCondition)
188  {
189  // generate a new independent realization
190  shadowingValue = m_normRandomVariable->GetValue () * GetShadowingStd (a, b, cond);
191  }
192  else
193  {
194  // compute a new correlated shadowing loss
195  Vector2D displacement (newDistance.x - it->second.m_distance.x, newDistance.y - it->second.m_distance.y);
196  double R = exp (-1 * displacement.GetLength () / GetShadowingCorrelationDistance (cond));
197  shadowingValue = R * it->second.m_shadowing + sqrt (1 - R * R) * m_normRandomVariable->GetValue () * GetShadowingStd (a, b, cond);
198  }
199 
200  // update the entry in the map
201  it->second.m_shadowing = shadowingValue;
202  it->second.m_distance = newDistance; // Save the (0,0,0) vector in case it's the first time we are calculating this value
203  it->second.m_condition = cond;
204 
205  return shadowingValue;
206 }

238 uint32_t
239 ThreeGppPropagationLossModel::GetKey (Ptr<MobilityModel> a, Ptr<MobilityModel> b)
240 {
241  // use the nodes ids to obtain an unique key for the channel between a and b
242  // sort the nodes ids so that the key is reciprocal
243  uint32_t x1 = std::min (a->GetObject<Node> ()->GetId (), b->GetObject<Node> ()->GetId ());
244  uint32_t x2 = std::max (a->GetObject<Node> ()->GetId (), b->GetObject<Node> ()->GetId ());
245 
246  // use the cantor function to obtain the key
247  uint32_t key = (((x1 + x2) * (x1 + x2 + 1)) / 2) + x2;
248 
249  return key;
250 }

305 double
306 ThreeGppRmaPropagationLossModel::GetLossLos (double distance2D, double distance3D, double hUt, double hBs) const
307 {
308  NS_LOG_FUNCTION (this);
309  NS_ASSERT_MSG (m_frequency <= 30.0e9, "RMa scenario is valid for frequencies between 0.5 and 30 GHz.");
310 
311  // check if hBS and hUT are within the specified validity range
312  if (hUt < 1.0 || hUt > 10.0)
313  {
314  NS_LOG_WARN ("The height of the UT should be between 1 and 10 m (see TR 38.901, Table 7.4.1-1)");
315  }
316 
317  if (hBs < 10.0 || hBs > 150.0)
318  {
319  NS_LOG_WARN ("The height of the BS should be between 10 and 150 m (see TR 38.901, Table 7.4.1-1)");
320  }
321 
322  // NOTE The model is intended to be used for BS-UT links, however we may need to
323  // compute the pathloss between two BSs or UTs, e.g., to evaluate the
324  // interference. In order to apply the model, we need to retrieve the values of
325  // hBS and hUT, but in these cases one of the two falls outside the validity
326  // range and the warning message is printed (hBS for the UT-UT case and hUT
327  // for the BS-BS case).
328 
329  double distanceBp = GetBpDistance (m_frequency, hBs, hUt);
330  NS_LOG_DEBUG ("breakpoint distance " << distanceBp);
331 
332  // check if the distace is outside the validity range
333  if (distance2D < 10.0 || distance2D > 10.0e3)
334  {
335  NS_LOG_WARN ("The 2D distance is outside the validity range, the pathloss value may not be accurate");
336  }
337 
338  // compute the pathloss (see 3GPP TR 38.901, Table 7.4.1-1)
339  double loss = 0;
340  if (distance2D <= distanceBp)
341  {
342  // use PL1
343  loss = Pl1 (m_frequency, distance3D, m_h, m_w);
344  }
345  else
346  {
347  // use PL2
348  loss = Pl1 (m_frequency, distanceBp, m_h, m_w) + 40 * log10 (distance3D / distanceBp);
349  }
350 
351  NS_LOG_DEBUG ("Loss " << loss);
352 
353  return loss;
354 }

356 double
357 ThreeGppRmaPropagationLossModel::GetLossNlos (double distance2D, double distance3D, double hUt, double hBs) const
358 {
359  NS_LOG_FUNCTION (this);
360  NS_ASSERT_MSG (m_frequency <= 30.0e9, "RMa scenario is valid for frequencies between 0.5 and 30 GHz.");
361 
362  // check if hBs and hUt are within the validity range
363  if (hUt < 1.0 || hUt > 10.0)
364  {
365  NS_LOG_WARN ("The height of the UT should be between 1 and 10 m (see TR 38.901, Table 7.4.1-1)");
366  }
367 
368  if (hBs < 10.0 || hBs > 150.0)
369  {
370  NS_LOG_WARN ("The height of the BS should be between 10 and 150 m (see TR 38.901, Table 7.4.1-1)");
371  }
372 
373  // NOTE The model is intended to be used for BS-UT links, however we may need to
374  // compute the pathloss between two BSs or UTs, e.g., to evaluate the
375  // interference. In order to apply the model, we need to retrieve the values of
376  // hBS and hUT, but in these cases one of the two falls outside the validity
377  // range and the warning message is printed (hBS for the UT-UT case and hUT
378  // for the BS-BS case).
379 
380  // check if the distace is outside the validity range
381  if (distance2D < 10.0 || distance2D > 5.0e3)
382  {
383  NS_LOG_WARN ("The 2D distance is outside the validity range, the pathloss value may not be accurate");
384  }
385 
386  // compute the pathloss
387  double plNlos = 161.04 - 7.1 * log10 (m_w) + 7.5 * log10 (m_h) - (24.37 - 3.7 * pow ((m_h / hBs), 2)) * log10 (hBs) + (43.42 - 3.1 * log10 (hBs)) * (log10 (distance3D) - 3.0) + 20.0 * log10 (m_frequency / 1e9) - (3.2 * pow (log10 (11.75 * hUt), 2) - 4.97);
388 
389  double loss = std::max (GetLossLos (distance2D, distance3D, hUt, hBs), plNlos);
390 
391  NS_LOG_DEBUG ("Loss " << loss);
392 
393  return loss;
394 }

396 double
397 ThreeGppRmaPropagationLossModel::GetShadowingStd (Ptr<MobilityModel> a, Ptr<MobilityModel> b, ChannelCondition::LosConditionValue cond) const
398 {
399  NS_LOG_FUNCTION (this);
400  double shadowingStd;
401 
402  if (cond == ChannelCondition::LosConditionValue::LOS)
403  {
404  // compute the 2D distance between the two nodes
405  double distance2d = Calculate2dDistance (a->GetPosition (), b->GetPosition ());
406 
407  // compute the breakpoint distance (see 3GPP TR 38.901, Table 7.4.1-1, note 5)
408  double distanceBp = GetBpDistance (m_frequency, a->GetPosition ().z, b->GetPosition ().z);
409 
410  if (distance2d <= distanceBp)
411  {
412  shadowingStd = 4.0;
413  }
414  else
415  {
416  shadowingStd = 6.0;
417  }
418  }
419  else if (cond == ChannelCondition::LosConditionValue::NLOS)
420  {
421  shadowingStd = 8.0;
422  }
423  else
424  {
425  NS_FATAL_ERROR ("Unknown channel condition");
426  }
427 
428  return shadowingStd;
429 }

454 double
455 ThreeGppRmaPropagationLossModel::Pl1 (double frequency, double distance3D, double h, double w)
456 {
457  NS_UNUSED (w);
458  double loss = 20.0 * log10 (40.0 * M_PI * distance3D * frequency / 1e9 / 3.0) + std::min (0.03 * pow (h, 1.72), 10.0) * log10 (distance3D) - std::min (0.044 * pow (h, 1.72), 14.77) + 0.002 * log10 (h) * distance3D;
459  return loss;

499 double
500 ThreeGppUmaPropagationLossModel::GetBpDistance (double hUt, double hBs, double distance2D) const
501 {
502  NS_LOG_FUNCTION (this);
503 
504  // compute g (d2D) (see 3GPP TR 38.901, Table 7.4.1-1, Note 1)
505  double g = 0.0;
506  if (distance2D > 18.0)
507  {
508  g = 5.0 / 4.0 * pow (distance2D / 100.0, 3) * exp (-distance2D / 150.0);
509  }
510 
511  // compute C (hUt, d2D) (see 3GPP TR 38.901, Table 7.4.1-1, Note 1)
512  double c = 0.0;
513  if (hUt >= 13.0)
514  {
515  c = pow ((hUt - 13.0) / 10.0, 1.5) * g;
516  }
517 
518  // compute hE (see 3GPP TR 38.901, Table 7.4.1-1, Note 1)
519  double prob = 1.0 / (1.0 + c);
520  double hE = 0.0;
521  if (m_uniformVar->GetValue () < prob)
522  {
523  hE = 1.0;
524  }
525  else
526  {
527  int random = m_uniformVar->GetInteger (12, (int)(hUt - 1.5));
528  hE = (double)floor (random / 3.0) * 3.0;
529  }
530 
531  // compute dBP' (see 3GPP TR 38.901, Table 7.4.1-1, Note 1)
532  double distanceBp = 4 * (hBs - hE) * (hUt - hE) * m_frequency / M_C;
533 
534  return distanceBp;
535 }

537 double
538 ThreeGppUmaPropagationLossModel::GetLossLos (double distance2D, double distance3D, double hUt, double hBs) const
539 {
540  NS_LOG_FUNCTION (this);
541 
542  // check if hBS and hUT are within the validity range
543  if (hUt < 1.5 || hUt > 22.5)
544  {
545  NS_LOG_WARN ("The height of the UT should be between 1.5 and 22.5 m (see TR 38.901, Table 7.4.1-1)");
546  }
547 
548  if (hBs != 25.0)
549  {
550  NS_LOG_WARN ("The height of the BS should be equal to 25 m (see TR 38.901, Table 7.4.1-1)");
551  }
552 
553  // NOTE The model is intended to be used for BS-UT links, however we may need to
554  // compute the pathloss between two BSs or UTs, e.g., to evaluate the
555  // interference. In order to apply the model, we need to retrieve the values of
556  // hBS and hUT, but in these cases one of the two falls outside the validity
557  // range and the warning message is printed (hBS for the UT-UT case and hUT
558  // for the BS-BS case).
559 
560  // compute the breakpoint distance (see 3GPP TR 38.901, Table 7.4.1-1, note 1)
561  double distanceBp = GetBpDistance (hUt, hBs, distance2D);
562  NS_LOG_DEBUG ("breakpoint distance " << distanceBp);
563 
564  // check if the distace is outside the validity range
565  if (distance2D < 10.0 || distance2D > 5.0e3)
566  {
567  NS_LOG_WARN ("The 2D distance is outside the validity range, the pathloss value may not be accurate");
568  }
569 
570  // compute the pathloss (see 3GPP TR 38.901, Table 7.4.1-1)
571  double loss = 0;
572  if (distance2D <= distanceBp)
573  {
574  // use PL1
575  loss = 28.0 + 22.0 * log10 (distance3D) + 20.0 * log10 (m_frequency / 1e9);
576  }
577  else
578  {
579  // use PL2
580  loss = 28.0 + 40.0 * log10 (distance3D) + 20.0 * log10 (m_frequency / 1e9) - 9.0 * log10 (pow (distanceBp, 2) + pow (hBs - hUt, 2));
581  }
582 
583  NS_LOG_DEBUG ("Loss " << loss);
584 
585  return loss;
586 }

725 double
726 ThreeGppUmiStreetCanyonPropagationLossModel::GetLossLos (double distance2D, double distance3D, double hUt, double hBs) const
727 {
728  NS_LOG_FUNCTION (this);
729 
730  // check if hBS and hUT are within the validity range
731  if (hUt < 1.5 || hUt >= 10.0)
732  {
733  NS_LOG_WARN ("The height of the UT should be between 1.5 and 22.5 m (see TR 38.901, Table 7.4.1-1). We further assume hUT < hBS, then hUT is upper bounded by hBS, which should be 10 m");
734  }
735 
736  if (hBs != 10.0)
737  {
738  NS_LOG_WARN ("The height of the BS should be equal to 10 m (see TR 38.901, Table 7.4.1-1)");
739  }
740 
741  // NOTE The model is intended to be used for BS-UT links, however we may need to
742  // compute the pathloss between two BSs or UTs, e.g., to evaluate the
743  // interference. In order to apply the model, we need to retrieve the values of
744  // hBS and hUT, but in these cases one of the two falls outside the validity
745  // range and the warning message is printed (hBS for the UT-UT case and hUT
746  // for the BS-BS case).
747 
748  // compute the breakpoint distance (see 3GPP TR 38.901, Table 7.4.1-1, note 1)
749  double distanceBp = GetBpDistance (hUt, hBs, distance2D);
750  NS_LOG_DEBUG ("breakpoint distance " << distanceBp);
751 
752  // check if the distace is outside the validity range
753  if (distance2D < 10.0 || distance2D > 5.0e3)
754  {
755  NS_LOG_WARN ("The 2D distance is outside the validity range, the pathloss value may not be accurate");
756  }
757 
758  // compute the pathloss (see 3GPP TR 38.901, Table 7.4.1-1)
759  double loss = 0;
760  if (distance2D <= distanceBp)
761  {
762  // use PL1
763  loss = 32.4 + 21.0 * log10 (distance3D) + 20.0 * log10 (m_frequency / 1e9);
764  }
765  else
766  {
767  // use PL2
768  loss = 32.4 + 40.0 * log10 (distance3D) + 20.0 * log10 (m_frequency / 1e9) - 9.5 * log10 (pow (distanceBp, 2) + pow (hBs - hUt, 2));
769  }
770 
771  NS_LOG_DEBUG ("Loss " << loss);
772 
773  return loss;
774 }

814 ThreeGppUmiStreetCanyonPropagationLossModel::GetUtAndBsHeights (double za, double zb) const
815 {
816  NS_LOG_FUNCTION (this);
817  // TR 38.901 specifies hBS = 10 m and 1.5 <= hUT <= 22.5
818  double hBs, hUt;
819  if (za == 10.0)
820  {
821  // node A is the BS and node B is the UT
822  hBs = za;
823  hUt = zb;
824  }
825  else if (zb == 10.0)
826  {
827  // node B is the BS and node A is the UT
828  hBs = zb;
829  hUt = za;
830  }
831  else
832  {
833  // We cannot know who is the BS and who is the UT, we assume that the
834  // tallest node is the BS and the smallest is the UT
835  hBs = std::max (za, zb);
836  hUt = std::min (za, za);
837  }
838 
839  return std::pair<double, double> (hUt, hBs);
840 }

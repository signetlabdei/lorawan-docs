122 void
123 Bug772ChainTest::CreateDevices ()
124 {
125  int64_t streamsUsed = 0;
126  // 1. Setup WiFi
127  WifiMacHelper wifiMac;
128  wifiMac.SetType ("ns3::AdhocWifiMac");
129  YansWifiPhyHelper wifiPhy = YansWifiPhyHelper::Default ();
130  wifiPhy.DisablePreambleDetectionModel ();
131  // This test suite output was originally based on YansErrorRateModel
132  wifiPhy.SetErrorRateModel ("ns3::YansErrorRateModel");
133  YansWifiChannelHelper wifiChannel = YansWifiChannelHelper::Default ();
134  Ptr<YansWifiChannel> chan = wifiChannel.Create ();
135  wifiPhy.SetChannel (chan);
136  wifiPhy.Set ("TxGain", DoubleValue (1.0)); //this configuration should go away in future revision to the test
137  wifiPhy.Set ("RxGain", DoubleValue (1.0)); //this configuration should go away in future revision to the test
138  WifiHelper wifi;
139  wifi.SetRemoteStationManager ("ns3::ConstantRateWifiManager", "DataMode", StringValue ("OfdmRate6Mbps"), "RtsCtsThreshold", StringValue ("2200"), "MaxSlrc", UintegerValue (7));
140  NetDeviceContainer devices = wifi.Install (wifiPhy, wifiMac, *m_nodes);
141 
142  // Assign fixed stream numbers to wifi and channel random variables
143  streamsUsed += wifi.AssignStreams (devices, streamsUsed);
144  // Assign 6 streams per device
145  NS_TEST_ASSERT_MSG_EQ (streamsUsed, (devices.GetN () * 6), "Stream assignment mismatch");
146  streamsUsed += wifiChannel.AssignStreams (chan, streamsUsed);
147  // Assign 0 streams per channel for this configuration
148  NS_TEST_ASSERT_MSG_EQ (streamsUsed, (devices.GetN () * 6), "Stream assignment mismatch");
149 
150  // 2. Setup TCP/IP & AODV
151  AodvHelper aodv; // Use default parameters here
152  InternetStackHelper internetStack;
153  internetStack.SetRoutingHelper (aodv);
154  internetStack.Install (*m_nodes);
155  streamsUsed += internetStack.AssignStreams (*m_nodes, streamsUsed);
156  // Expect to use (3*m_size) more streams for internet stack random variables
157  NS_TEST_ASSERT_MSG_EQ (streamsUsed, ((devices.GetN () * 6) + (3 * m_size)), "Stream assignment mismatch");
158  streamsUsed += aodv.AssignStreams (*m_nodes, streamsUsed);
159  // Expect to use m_size more streams for AODV
160  NS_TEST_ASSERT_MSG_EQ (streamsUsed, ((devices.GetN () * 6) + (3 * m_size) + m_size), "Stream assignment mismatch");
161  Ipv4AddressHelper address;
162  address.SetBase ("10.1.1.0", "255.255.255.0");
163  Ipv4InterfaceContainer interfaces = address.Assign (devices);
164 
165  // 3. Setup UDP source and sink
166  m_sendSocket = Socket::CreateSocket (m_nodes->Get (0), TypeId::LookupByName (m_proto));
167  m_sendSocket->Bind ();
168  m_sendSocket->Connect (InetSocketAddress (interfaces.GetAddress (m_size - 1), m_port));
169  m_sendSocket->SetAllowBroadcast (true);
170  Simulator::ScheduleWithContext (m_sendSocket->GetNode ()->GetId (), Seconds (1.0),
171  &Bug772ChainTest::SendData, this, m_sendSocket);
172 
173  m_recvSocket = Socket::CreateSocket (m_nodes->Get (m_size - 1), TypeId::LookupByName (m_proto));
174  m_recvSocket->Bind (InetSocketAddress (Ipv4Address::GetAny (), m_port));
175  m_recvSocket->Listen ();
176  m_recvSocket->ShutdownSend ();
177  m_recvSocket->SetRecvCallback (MakeCallback (&Bug772ChainTest::HandleRead, this));
178 
179 }

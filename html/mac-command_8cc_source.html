69 uint8_t
70 MacCommand::GetCIDFromMacCommand (enum MacCommandType commandType)
71 {
72  NS_LOG_FUNCTION_NOARGS ();
73 
74  switch (commandType)
75  {
76  case (INVALID):
77  {
78  return 0x0;
79  }
80  case (LINK_CHECK_REQ):
81  case (LINK_CHECK_ANS):
82  {
83  return 0x02;
84  }
85  case (LINK_ADR_REQ):
86  case (LINK_ADR_ANS):
87  {
88  return 0x03;
89  }
90  case (DUTY_CYCLE_REQ):
91  case (DUTY_CYCLE_ANS):
92  {
93  return 0x04;
94  }
95  case (RX_PARAM_SETUP_REQ):
96  case (RX_PARAM_SETUP_ANS):
97  {
98  return 0x05;
99  }
100  case (DEV_STATUS_REQ):
101  case (DEV_STATUS_ANS):
102  {
103  return 0x06;
104  }
105  case (NEW_CHANNEL_REQ):
106  case (NEW_CHANNEL_ANS):
107  {
108  return 0x07;
109  }
110  case (RX_TIMING_SETUP_REQ):
111  case (RX_TIMING_SETUP_ANS):
112  {
113  return 0x08;
114  }
115  case (TX_PARAM_SETUP_REQ):
116  case (TX_PARAM_SETUP_ANS):
117  {
118  return 0x09;
119  }
120  case (DL_CHANNEL_REQ):
121  case (DL_CHANNEL_ANS):
122  {
123  return 0x0A;
124  }
125  }
126  return 0;
127 }

300 void
301 LinkAdrReq::Serialize (Buffer::Iterator &start) const
302 {
303  NS_LOG_FUNCTION_NOARGS ();
304 
305  // Write the CID
306  start.WriteU8 (GetCIDFromMacCommand (m_commandType));
307  start.WriteU8 (m_dataRate << 4 | (m_txPower & 0b1111));
308  start.WriteU16 (m_channelMask);
309  start.WriteU8 (m_chMaskCntl << 4 | (m_nbRep & 0b1111));
310 }

312 uint8_t
313 LinkAdrReq::Deserialize (Buffer::Iterator &start)
314 {
315  NS_LOG_FUNCTION_NOARGS ();
316 
317  // Consume the CID
318  start.ReadU8 ();
319  uint8_t firstByte = start.ReadU8 ();
320  m_dataRate = firstByte >> 4;
321  m_txPower = firstByte & 0b1111;
322  m_channelMask = start.ReadU16 ();
323  uint8_t fourthByte = start.ReadU8 ();
324  m_chMaskCntl = fourthByte >> 4;
325  m_nbRep = fourthByte & 0b1111;
326 
327  return m_serializedSize;
328 }

590 void
591 RxParamSetupReq::Serialize (Buffer::Iterator &start) const
592 {
593  NS_LOG_FUNCTION_NOARGS ();
594 
595  // Write the CID
596  start.WriteU8 (GetCIDFromMacCommand (m_commandType));
597  // Data serialization
598  start.WriteU8 ((m_rx1DrOffset & 0b111) << 4 | (m_rx2DataRate & 0b1111));
599  uint32_t encodedFrequency = uint32_t (m_frequency / 100);
600  NS_LOG_DEBUG (unsigned (encodedFrequency));
601  NS_LOG_DEBUG (std::bitset<32> (encodedFrequency));
602  start.WriteU8 ((encodedFrequency & 0xff0000) >> 16); // Most significant byte
603  start.WriteU8 ((encodedFrequency & 0xff00) >> 8); // Middle byte
604  start.WriteU8 (encodedFrequency & 0xff); // Least significant byte
605 }
606 
607 uint8_t
608 RxParamSetupReq::Deserialize (Buffer::Iterator &start)
609 {
610  NS_LOG_FUNCTION_NOARGS ();
611 
612  // Consume the CID
613  start.ReadU8 ();
614  // Data serialization
615  uint8_t firstByte = start.ReadU8 ();
616  m_rx1DrOffset = (firstByte & 0b1110000) >> 4;
617  m_rx2DataRate = firstByte & 0b1111;
618  uint32_t secondByte = start.ReadU8 ();
619  uint32_t thirdByte = start.ReadU8 ();
620  uint32_t fourthByte = start.ReadU8 ();
621  uint32_t encodedFrequency = (secondByte << 16) | (thirdByte << 8) | fourthByte;
622  NS_LOG_DEBUG (std::bitset<32> (encodedFrequency));
623  m_frequency = double(encodedFrequency) * 100;
624 
625  return m_serializedSize;
626 }

117 HwmpTag::Serialize (TagBuffer i) const
118 {
119  uint8_t address[6];
120  int j;
121  m_address.CopyTo (address);
122  i.WriteU8 (m_ttl);
123  i.WriteU32 (m_metric);
124  i.WriteU32 (m_seqno);

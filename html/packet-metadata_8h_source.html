694 PacketMetadata::PacketMetadata (uint64_t uid, uint32_t size)
695  : m_data (PacketMetadata::Create (10)),
696  m_head (0xffff),
697  m_tail (0xffff),
698  m_used (0),
699  m_packetUid (uid)
700 {
701  memset (m_data->m_data, 0xff, 4);
702  if (size > 0)
703  {
704  DoAddHeader (0, size);
705  }
706 }
707 PacketMetadata::PacketMetadata (PacketMetadata const &o)
708  : m_data (o.m_data),
709  m_head (o.m_head),
710  m_tail (o.m_tail),
711  m_used (o.m_used),
712  m_packetUid (o.m_packetUid)
713 {
714  NS_ASSERT (m_data != 0);
715  NS_ASSERT (m_data->m_count < std::numeric_limits<uint32_t>::max());
716  m_data->m_count++;
717 }
718 PacketMetadata &
719 PacketMetadata::operator = (PacketMetadata const& o)
720 {
721  if (m_data != o.m_data)
722  {
723  // not self assignment
724  NS_ASSERT (m_data != 0);
725  m_data->m_count--;
726  if (m_data->m_count == 0)
727  {
728  PacketMetadata::Recycle (m_data);
729  }
730  m_data = o.m_data;
731  NS_ASSERT (m_data != 0);
732  m_data->m_count++;
733  }
734  m_head = o.m_head;
735  m_tail = o.m_tail;
736  m_used = o.m_used;
737  m_packetUid = o.m_packetUid;
738  return *this;
739 }
740 PacketMetadata::~PacketMetadata ()
741 {
742  NS_ASSERT (m_data != 0);
743  m_data->m_count--;
744  if (m_data->m_count == 0)
745  {
746  PacketMetadata::Recycle (m_data);
747  }
748 }

102 uint32_t
103 PeerLinkOpenStart::Deserialize (Buffer::Iterator start)
104 {
105  Buffer::Iterator i = start;
106 
107  m_capability = i.ReadLsbtohU16 ();
108  i = m_rates.Deserialize (i);
109  i = m_rates.extended.DeserializeIfPresent (i);
110  uint8_t id = i.ReadU8 ();
111  uint8_t length = i.ReadU8 ();
112  m_meshId.DeserializeInformationField (i, length);
113  if ((m_meshId.ElementId () != (WifiInformationElementId) id) || (m_meshId.GetInformationFieldSize () != length))
114  {
115  NS_FATAL_ERROR ("Broken frame: Element ID does not match IE itself!");
116  }
117  i.Next (m_meshId.GetInformationFieldSize ());
118  id = i.ReadU8 ();
119  length = i.ReadU8 ();
120  m_config.DeserializeInformationField (i, length);
121  if ((m_config.ElementId () != (WifiInformationElementId) id) || (m_config.GetInformationFieldSize () != length))
122  {
123  NS_FATAL_ERROR ("Broken frame: Element ID does not match IE itself!");
124  }
125  i.Next (m_config.GetInformationFieldSize ());
126 
127  return i.GetDistanceFrom (start);
128 }

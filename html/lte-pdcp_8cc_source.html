176 void
177 LtePdcp::DoTransmitPdcpSdu (Ptr<Packet> p)
178 {
179  NS_LOG_FUNCTION (this << m_rnti << (uint32_t) m_lcid << p->GetSize ());
180 
181  // Sender timestamp
182  PdcpTag pdcpTag (Simulator::Now ());
183 
184  LtePdcpHeader pdcpHeader;
185  pdcpHeader.SetSequenceNumber (m_txSequenceNumber);
186 
187  m_txSequenceNumber++;
188  if (m_txSequenceNumber > m_maxPdcpSn)
189  {
190  m_txSequenceNumber = 0;
191  }
192 
193  pdcpHeader.SetDcBit (LtePdcpHeader::DATA_PDU);
194 
195  NS_LOG_LOGIC ("PDCP header: " << pdcpHeader);
196  p->AddHeader (pdcpHeader);
197  p->AddByteTag (pdcpTag, 1, pdcpHeader.GetSerializedSize ());
198 
199  m_txPdu (m_rnti, m_lcid, p->GetSize ());
200 
201  LteRlcSapProvider::TransmitPdcpPduParameters params;
202  params.rnti = m_rnti;
203  params.lcid = m_lcid;
204  params.pdcpPdu = p;
205 
206  m_rlcSapProvider->TransmitPdcpPdu (params);
207 }
208 
209 void
210 LtePdcp::DoReceivePdu (Ptr<Packet> p)
211 {
212  NS_LOG_FUNCTION (this << m_rnti << (uint32_t) m_lcid << p->GetSize ());
213 
214  // Receiver timestamp
215  PdcpTag pdcpTag;
216  Time delay;
217  p->FindFirstMatchingByteTag (pdcpTag);
218  delay = Simulator::Now() - pdcpTag.GetSenderTimestamp ();
219  m_rxPdu(m_rnti, m_lcid, p->GetSize (), delay.GetNanoSeconds ());
220 
221  LtePdcpHeader pdcpHeader;
222  p->RemoveHeader (pdcpHeader);
223  NS_LOG_LOGIC ("PDCP header: " << pdcpHeader);
224 
225  m_rxSequenceNumber = pdcpHeader.GetSequenceNumber () + 1;
226  if (m_rxSequenceNumber > m_maxPdcpSn)
227  {
228  m_rxSequenceNumber = 0;
229  }
230 
231  LtePdcpSapUser::ReceivePdcpSduParameters params;
232  params.pdcpSdu = p;
233  params.rnti = m_rnti;
234  params.lcid = m_lcid;
235  m_pdcpSapUser->ReceivePdcpSdu (params);
236 }

213 void
214 LoraRadioEnergyModel::ChangeState (int newState)
215 {
216  NS_LOG_FUNCTION (this << newState);
217 
218  Time duration = Simulator::Now () - m_lastUpdateTime;
219  NS_ASSERT (duration.GetNanoSeconds () >= 0); // check if duration is valid
220 
221  // energy to decrease = current * voltage * time
222  double energyToDecrease = 0.0;
223  double supplyVoltage = m_source->GetSupplyVoltage ();
224  switch (m_currentState)
225  {
226  case EndDeviceLoraPhy::STANDBY:
227  energyToDecrease = duration.GetSeconds () * m_idleCurrentA * supplyVoltage;
228  break;
229  case EndDeviceLoraPhy::TX:
230  energyToDecrease = duration.GetSeconds () * m_txCurrentA * supplyVoltage;
231  break;
232  case EndDeviceLoraPhy::RX:
233  energyToDecrease = duration.GetSeconds () * m_rxCurrentA * supplyVoltage;
234  break;
235  case EndDeviceLoraPhy::SLEEP:
236  energyToDecrease = duration.GetSeconds () * m_sleepCurrentA * supplyVoltage;
237  break;
238  default:
239  NS_FATAL_ERROR ("LoraRadioEnergyModel:Undefined radio state: " << m_currentState);
240  }
241 
242  // update total energy consumption
243  m_totalEnergyConsumption += energyToDecrease;
244 
245  // update last update time stamp
246  m_lastUpdateTime = Simulator::Now ();
247 
248  m_nPendingChangeState++;
249 
250  // notify energy source
251  m_source->UpdateEnergySource ();
252 
253  // in case the energy source is found to be depleted during the last update, a callback might be
254  // invoked that might cause a change in the Lora PHY state (e.g., the PHY is put into SLEEP mode).
255  // This in turn causes a new call to this member function, with the consequence that the previous
256  // instance is resumed after the termination of the new instance. In particular, the state set
257  // by the previous instance is erroneously the final state stored in m_currentState. The check below
258  // ensures that previous instances do not change m_currentState.
259 
260  if (!m_isSupersededChangeState)
261  {
262  // update current state & last update time stamp
263  SetLoraRadioState ((EndDeviceLoraPhy::State) newState);
264 
265  // some debug message
266  NS_LOG_DEBUG ("LoraRadioEnergyModel:Total energy consumption is " <<
267  m_totalEnergyConsumption << "J");
268  }
269 
270  m_isSupersededChangeState = (m_nPendingChangeState > 1);
271 
272  m_nPendingChangeState--;
273 }

325 double
326 LoraRadioEnergyModel::DoGetCurrentA (void) const
327 {
328  NS_LOG_FUNCTION (this);
329  switch (m_currentState)
330  {
331  case EndDeviceLoraPhy::STANDBY:
332  return m_idleCurrentA;
333  case EndDeviceLoraPhy::TX:
334  return m_txCurrentA;

336  return m_rxCurrentA;

341  }

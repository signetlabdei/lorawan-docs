357  void ReportExpectedCollisions () const
358  {
359  // Expected number of collisions
360  //
361  // Number of buckets = k = 2^bits
362  long double k32 = 0xFFFFFFFF;
363  long double k64 = static_cast<long double> (0xFFFFFFFFFFFFFFFFULL);
364 
365  long double n = m_nphrases;
366  long double Ec32 = n * (n - 1) / ( 2 * k32) * (1 - (n - 2) / (3 * k32));
367  long double Ec64 = n * (n - 1) / ( 2 * k64) * (1 - (n - 2) / (3 * k64));
368 
369  // Output collisions

371  std::cout << "Number of words or phrases: " << n << std::endl;
372  std::cout << "Expected number of collisions: (32-bit table) " << Ec32

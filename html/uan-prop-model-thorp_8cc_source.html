83 double
84 UanPropModelThorp::GetAttenDbKm (double freqKhz)
85 {
86 
87  double fsq = freqKhz * freqKhz;
88  double atten;
89 
90  if (freqKhz >= 0.4)
91  {
92  atten = 0.11 * fsq / (1 + fsq) + 44 * fsq / (4100 + fsq)
93  + 2.75 * 0.0001 * fsq + 0.003;
94  }
95  else
96  {
97  atten = 0.002 + 0.11 * (freqKhz / (1 + freqKhz)) + 0.011 * freqKhz;
98  }
99 
100  return atten;
101 }

53  CommandLine cmd;

55  cmd.Parse (argc, argv);
56 
57  // Set up logging
58  LogComponentEnable ("AlohaThroughput", LOG_LEVEL_ALL);
59 
60  // Make all devices use SF7 (i.e., DR5)
61  Config::SetDefault ("ns3::EndDeviceLorawanMac::DataRate", UintegerValue (5));
62  LoraInterferenceHelper::collisionMatrix = LoraInterferenceHelper::ALOHA;
63 
64  /***********
65  * Setup *
66  ***********/
67 
68  // Create the time value from the period
69  Time appPeriod = Seconds (appPeriodSeconds);
70 
71  // Mobility
72  MobilityHelper mobility;
73  mobility.SetPositionAllocator ("ns3::UniformDiscPositionAllocator", "rho", DoubleValue (radius),
74  "X", DoubleValue (0.0), "Y", DoubleValue (0.0));
75  mobility.SetMobilityModel ("ns3::ConstantPositionMobilityModel");
76 
77  /************************
78  * Create the channel *
79  ************************/
80 
81  // Create the lora channel object
82  Ptr<LogDistancePropagationLossModel> loss = CreateObject<LogDistancePropagationLossModel> ();
83  loss->SetPathLossExponent (3.76);
84  loss->SetReference (1, 7.7);
85 
86  if (realisticChannelModel)
87  {
88  // Create the correlated shadowing component
89  Ptr<CorrelatedShadowingPropagationLossModel> shadowing =
90  CreateObject<CorrelatedShadowingPropagationLossModel> ();
91 
92  // Aggregate shadowing to the logdistance loss
93  loss->SetNext (shadowing);
94 
95  // Add the effect to the channel propagation loss
96  Ptr<BuildingPenetrationLoss> buildingLoss = CreateObject<BuildingPenetrationLoss> ();
97 
98  shadowing->SetNext (buildingLoss);
99  }
100 
101  Ptr<PropagationDelayModel> delay = CreateObject<ConstantSpeedPropagationDelayModel> ();
102 
103  Ptr<LoraChannel> channel = CreateObject<LoraChannel> (loss, delay);
104 
105  /************************
106  * Create the helpers *
107  ************************/
108 
109  // Create the LoraPhyHelper
110  LoraPhyHelper phyHelper = LoraPhyHelper ();
111  phyHelper.SetChannel (channel);
112 
113  // Create the LorawanMacHelper
114  LorawanMacHelper macHelper = LorawanMacHelper ();
115  macHelper.SetRegion (LorawanMacHelper::ALOHA);
116 
117  // Create the LoraHelper
118  LoraHelper helper = LoraHelper ();
119  helper.EnablePacketTracking (); // Output filename
120 
121  //Create the NetworkServerHelper
122  NetworkServerHelper nsHelper = NetworkServerHelper ();
123 
124  //Create the ForwarderHelper
125  ForwarderHelper forHelper = ForwarderHelper ();
126 
127  /************************
128  * Create End Devices *
129  ************************/
130 
131  // Create a set of nodes
132  NodeContainer endDevices;
133  endDevices.Create (nDevices);
134 
135  // Assign a mobility model to each node
136  mobility.Install (endDevices);
137 
138  // Make it so that nodes are at a certain height > 0
139  for (NodeContainer::Iterator j = endDevices.Begin (); j != endDevices.End (); ++j)
140  {
141  Ptr<MobilityModel> mobility = (*j)->GetObject<MobilityModel> ();
142  Vector position = mobility->GetPosition ();
143  position.z = 1.2;
144  mobility->SetPosition (position);
145  }
146 
147  // Create the LoraNetDevices of the end devices
148  uint8_t nwkId = 54;
149  uint32_t nwkAddr = 1864;
150  Ptr<LoraDeviceAddressGenerator> addrGen =
151  CreateObject<LoraDeviceAddressGenerator> (nwkId, nwkAddr);
152 
153  // Create the LoraNetDevices of the end devices
154  macHelper.SetAddressGenerator (addrGen);
155  phyHelper.SetDeviceType (LoraPhyHelper::ED);
156  macHelper.SetDeviceType (LorawanMacHelper::ED_A);
157  helper.Install (phyHelper, macHelper, endDevices);
158 
159  // Now end devices are connected to the channel
160 
161  // Connect trace sources
162  for (NodeContainer::Iterator j = endDevices.Begin (); j != endDevices.End (); ++j)
163  {
164  Ptr<Node> node = *j;
165  Ptr<LoraNetDevice> loraNetDevice = node->GetDevice (0)->GetObject<LoraNetDevice> ();
166  Ptr<LoraPhy> phy = loraNetDevice->GetPhy ();
167  }
168 
169  /*********************
170  * Create Gateways *
171  *********************/
172 
173  // Create the gateway nodes (allocate them uniformely on the disc)
174  NodeContainer gateways;
175  gateways.Create (nGateways);
176 
177  Ptr<ListPositionAllocator> allocator = CreateObject<ListPositionAllocator> ();
178  // Make it so that nodes are at a certain height > 0
179  allocator->Add (Vector (0.0, 0.0, 15.0));
180  mobility.SetPositionAllocator (allocator);
181  mobility.Install (gateways);
182 
183  // Create a netdevice for each gateway
184  phyHelper.SetDeviceType (LoraPhyHelper::GW);
185  macHelper.SetDeviceType (LorawanMacHelper::GW);
186  helper.Install (phyHelper, macHelper, gateways);
187 
188  NS_LOG_DEBUG ("Completed configuration");
189 
190  /*********************************************
191  * Install applications on the end devices *
192  *********************************************/
193 
194  Time appStopTime = Seconds (simulationTime);
195  PeriodicSenderHelper appHelper = PeriodicSenderHelper ();
196  appHelper.SetPeriod (Seconds (appPeriodSeconds));
197  appHelper.SetPacketSize (150);
198  Ptr<RandomVariableStream> rv = CreateObjectWithAttributes<UniformRandomVariable> (
199  "Min", DoubleValue (0), "Max", DoubleValue (10));
200  ApplicationContainer appContainer = appHelper.Install (endDevices);
201 
202  appContainer.Start (Seconds (0));
203  appContainer.Stop (appStopTime);
204 
205  /**************************
206  * Create Network Server *
207  ***************************/
208 
209  // Create the NS node
210  NodeContainer networkServer;
211  networkServer.Create (1);
212 
213  // Create a NS for the network
214  nsHelper.SetEndDevices (endDevices);
215  nsHelper.SetGateways (gateways);
216  nsHelper.Install (networkServer);
217 
218  //Create a forwarder for each gateway
219  forHelper.Install (gateways);
220 
222  // Simulation //
224 
225  Simulator::Stop (appStopTime + Hours (1));
226 
227  NS_LOG_INFO ("Running simulation...");
228  Simulator::Run ();
229 
230  Simulator::Destroy ();
231 
233  // Print results to stdout //
235  NS_LOG_INFO ("Computing performance metrics...");
236 
237  LoraPacketTracker &tracker = helper.GetPacketTracker ();
238 
239  std::cout << tracker.PrintPhyPacketsPerGw (Seconds (0), appStopTime + Hours (1), nDevices)
240  << std::endl;
241 
242  return 0;
243 }

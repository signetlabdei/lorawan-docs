96 void
97 LoraNetDevice::CompleteConfig (void)
98 {
99  NS_LOG_FUNCTION_NOARGS ();
100 
101  // Verify we have all the necessary pieces
102  if (m_mac == 0 || m_phy == 0 || m_node == 0 || m_configComplete)
103  {
104  return;
105  }
106 
107  m_mac->SetPhy (m_phy);
108  m_configComplete = true;
109 }

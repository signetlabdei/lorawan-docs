207  m_cb.Disconnect (cb, path);

215  void Set (const T &v)
216  {
217  if (m_v != v)
218  {
219  m_cb (m_v, v);
220  m_v = v;
221  }
222  }

229  return m_v;
230  }

239  {

244  return *this;

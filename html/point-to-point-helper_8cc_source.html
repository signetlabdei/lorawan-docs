80 PointToPointHelper::EnablePcapInternal (std::string prefix, Ptr<NetDevice> nd, bool promiscuous, bool explicitFilename)
81 {
82  //
83  // All of the Pcap enable functions vector through here including the ones
84  // that are wandering through all of devices on perhaps all of the nodes in
85  // the system. We can only deal with devices of type PointToPointNetDevice.
86  //
87  Ptr<PointToPointNetDevice> device = nd->GetObject<PointToPointNetDevice> ();
88  if (device == 0)
89  {
90  NS_LOG_INFO ("PointToPointHelper::EnablePcapInternal(): Device " << device << " not of type ns3::PointToPointNetDevice");
91  return;
92  }
93 
94  PcapHelper pcapHelper;
95 
96  std::string filename;
97  if (explicitFilename)
98  {
99  filename = prefix;
100  }
101  else
102  {
103  filename = pcapHelper.GetFilenameFromDevice (prefix, device);
104  }
105 
106  Ptr<PcapFileWrapper> file = pcapHelper.CreateFile (filename, std::ios::out,
107  PcapHelper::DLT_PPP);
108  pcapHelper.HookDefaultSink<PointToPointNetDevice> (device, "PromiscSniffer", file);
109 }

111 void
112 PointToPointHelper::EnableAsciiInternal (
113  Ptr<OutputStreamWrapper> stream,
114  std::string prefix,
115  Ptr<NetDevice> nd,
116  bool explicitFilename)
117 {
118  //
119  // All of the ascii enable functions vector through here including the ones
120  // that are wandering through all of devices on perhaps all of the nodes in
121  // the system. We can only deal with devices of type PointToPointNetDevice.
122  //
123  Ptr<PointToPointNetDevice> device = nd->GetObject<PointToPointNetDevice> ();
124  if (device == 0)
125  {
126  NS_LOG_INFO ("PointToPointHelper::EnableAsciiInternal(): Device " << device <<
127  " not of type ns3::PointToPointNetDevice");
128  return;
129  }
130 
131  //
132  // Our default trace sinks are going to use packet printing, so we have to
133  // make sure that is turned on.
134  //
135  Packet::EnablePrinting ();
136 
137  //
138  // If we are not provided an OutputStreamWrapper, we are expected to create
139  // one using the usual trace filename conventions and do a Hook*WithoutContext
140  // since there will be one file per context and therefore the context would
141  // be redundant.
142  //
143  if (stream == 0)
144  {
145  //
146  // Set up an output stream object to deal with private ofstream copy
147  // constructor and lifetime issues. Let the helper decide the actual
148  // name of the file given the prefix.
149  //
150  AsciiTraceHelper asciiTraceHelper;
151 
152  std::string filename;
153  if (explicitFilename)
154  {
155  filename = prefix;
156  }
157  else
158  {
159  filename = asciiTraceHelper.GetFilenameFromDevice (prefix, device);
160  }
161 
162  Ptr<OutputStreamWrapper> theStream = asciiTraceHelper.CreateFileStream (filename);
163 
164  //
165  // The MacRx trace source provides our "r" event.
166  //
167  asciiTraceHelper.HookDefaultReceiveSinkWithoutContext<PointToPointNetDevice> (device, "MacRx", theStream);
168 
169  //
170  // The "+", '-', and 'd' events are driven by trace sources actually in the
171  // transmit queue.
172  //
173  Ptr<Queue<Packet> > queue = device->GetQueue ();
174  asciiTraceHelper.HookDefaultEnqueueSinkWithoutContext<Queue<Packet> > (queue, "Enqueue", theStream);
175  asciiTraceHelper.HookDefaultDropSinkWithoutContext<Queue<Packet> > (queue, "Drop", theStream);
176  asciiTraceHelper.HookDefaultDequeueSinkWithoutContext<Queue<Packet> > (queue, "Dequeue", theStream);
177 
178  // PhyRxDrop trace source for "d" event
179  asciiTraceHelper.HookDefaultDropSinkWithoutContext<PointToPointNetDevice> (device, "PhyRxDrop", theStream);
180 
181  return;
182  }
183 
184  //
185  // If we are provided an OutputStreamWrapper, we are expected to use it, and
186  // to providd a context. We are free to come up with our own context if we
187  // want, and use the AsciiTraceHelper Hook*WithContext functions, but for
188  // compatibility and simplicity, we just use Config::Connect and let it deal
189  // with the context.
190  //
191  // Note that we are going to use the default trace sinks provided by the
192  // ascii trace helper. There is actually no AsciiTraceHelper in sight here,
193  // but the default trace sinks are actually publicly available static
194  // functions that are always there waiting for just such a case.
195  //
196  uint32_t nodeid = nd->GetNode ()->GetId ();
197  uint32_t deviceid = nd->GetIfIndex ();
198  std::ostringstream oss;
199 
200  oss << "/NodeList/" << nd->GetNode ()->GetId () << "/DeviceList/" << deviceid << "/$ns3::PointToPointNetDevice/MacRx";
201  Config::Connect (oss.str (), MakeBoundCallback (&AsciiTraceHelper::DefaultReceiveSinkWithContext, stream));
202 
203  oss.str ("");
204  oss << "/NodeList/" << nodeid << "/DeviceList/" << deviceid << "/$ns3::PointToPointNetDevice/TxQueue/Enqueue";
205  Config::Connect (oss.str (), MakeBoundCallback (&AsciiTraceHelper::DefaultEnqueueSinkWithContext, stream));
206 
207  oss.str ("");
208  oss << "/NodeList/" << nodeid << "/DeviceList/" << deviceid << "/$ns3::PointToPointNetDevice/TxQueue/Dequeue";
209  Config::Connect (oss.str (), MakeBoundCallback (&AsciiTraceHelper::DefaultDequeueSinkWithContext, stream));
210 
211  oss.str ("");
212  oss << "/NodeList/" << nodeid << "/DeviceList/" << deviceid << "/$ns3::PointToPointNetDevice/TxQueue/Drop";
213  Config::Connect (oss.str (), MakeBoundCallback (&AsciiTraceHelper::DefaultDropSinkWithContext, stream));
214 
215  oss.str ("");
216  oss << "/NodeList/" << nodeid << "/DeviceList/" << deviceid << "/$ns3::PointToPointNetDevice/PhyRxDrop";
217  Config::Connect (oss.str (), MakeBoundCallback (&AsciiTraceHelper::DefaultDropSinkWithContext, stream));
218 }

227 NetDeviceContainer
228 PointToPointHelper::Install (Ptr<Node> a, Ptr<Node> b)
229 {
230  NetDeviceContainer container;
231 
232  Ptr<PointToPointNetDevice> devA = m_deviceFactory.Create<PointToPointNetDevice> ();
233  devA->SetAddress (Mac48Address::Allocate ());
234  a->AddDevice (devA);
235  Ptr<Queue<Packet> > queueA = m_queueFactory.Create<Queue<Packet> > ();
236  devA->SetQueue (queueA);
237  Ptr<PointToPointNetDevice> devB = m_deviceFactory.Create<PointToPointNetDevice> ();
238  devB->SetAddress (Mac48Address::Allocate ());
239  b->AddDevice (devB);
240  Ptr<Queue<Packet> > queueB = m_queueFactory.Create<Queue<Packet> > ();
241  devB->SetQueue (queueB);
242  // Aggregate NetDeviceQueueInterface objects
243  Ptr<NetDeviceQueueInterface> ndqiA = CreateObject<NetDeviceQueueInterface> ();
244  ndqiA->GetTxQueue (0)->ConnectQueueTraces (queueA);
245  devA->AggregateObject (ndqiA);
246  Ptr<NetDeviceQueueInterface> ndqiB = CreateObject<NetDeviceQueueInterface> ();
247  ndqiB->GetTxQueue (0)->ConnectQueueTraces (queueB);
248  devB->AggregateObject (ndqiB);
249 
250  // If MPI is enabled, we need to see if both nodes have the same system id
251  // (rank), and the rank is the same as this instance. If both are true,
252  //use a normal p2p channel, otherwise use a remote channel
253  bool useNormalChannel = true;
254  Ptr<PointToPointChannel> channel = 0;
255 
256  if (MpiInterface::IsEnabled ())
257  {
258  uint32_t n1SystemId = a->GetSystemId ();
259  uint32_t n2SystemId = b->GetSystemId ();
260  uint32_t currSystemId = MpiInterface::GetSystemId ();
261  if (n1SystemId != currSystemId || n2SystemId != currSystemId)
262  {
263  useNormalChannel = false;
264  }
265  }
266  if (useNormalChannel)
267  {
268  channel = m_channelFactory.Create<PointToPointChannel> ();
269  }
270  else
271  {
272  channel = m_remoteChannelFactory.Create<PointToPointRemoteChannel> ();
273  Ptr<MpiReceiver> mpiRecA = CreateObject<MpiReceiver> ();
274  Ptr<MpiReceiver> mpiRecB = CreateObject<MpiReceiver> ();
275  mpiRecA->SetReceiveCallback (MakeCallback (&PointToPointNetDevice::Receive, devA));
276  mpiRecB->SetReceiveCallback (MakeCallback (&PointToPointNetDevice::Receive, devB));
277  devA->AggregateObject (mpiRecA);
278  devB->AggregateObject (mpiRecB);
279  }
280 
281  devA->Attach (channel);
282  devB->Attach (channel);
283  container.Add (devA);
284  container.Add (devB);
285 
286  return container;
287 }

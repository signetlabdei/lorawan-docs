48 EthernetTrailer::CheckFcs (Ptr<const Packet> p) const
49 {
50  NS_LOG_FUNCTION (this << p);
51  int len = p->GetSize ();
52  uint8_t *buffer;
53  uint32_t crc;
54 
55  if (!m_calcFcs)
56  {
57  return true;
58  }
59 
60  buffer = new uint8_t[len];

63  delete[] buffer;

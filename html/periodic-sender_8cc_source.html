105 void
106 PeriodicSender::SendPacket (void)
107 {
108  NS_LOG_FUNCTION (this);
109 
110  // Create and send a new packet
111  Ptr<Packet> packet;
112  if (m_pktSizeRV)
113  {
114  int randomsize = m_pktSizeRV->GetInteger ();
115  packet = Create<Packet> (m_basePktSize + randomsize);
116  }
117  else
118  {
119  packet = Create<Packet> (m_basePktSize);
120  }
121  m_mac->Send (packet);
122 
123  // Schedule the next SendPacket event
124  m_sendEvent = Simulator::Schedule (m_interval, &PeriodicSender::SendPacket,
125  this);
126 
127  NS_LOG_DEBUG ("Sent a packet of size " << packet->GetSize ());
128 }

130 void
131 PeriodicSender::StartApplication (void)
132 {
133  NS_LOG_FUNCTION (this);
134 
135  // Make sure we have a MAC layer
136  if (m_mac == 0)
137  {
138  // Assumes there's only one device
139  Ptr<LoraNetDevice> loraNetDevice = m_node->GetDevice (0)->GetObject<LoraNetDevice> ();
140 
141  m_mac = loraNetDevice->GetMac ();
142  NS_ASSERT (m_mac != 0);
143  }
144 
145  // Schedule the next SendPacket event
146  Simulator::Cancel (m_sendEvent);
147  NS_LOG_DEBUG ("Starting up application with a first event with a " <<
148  m_initialDelay.GetSeconds () << " seconds delay");
149  m_sendEvent = Simulator::Schedule (m_initialDelay,
150  &PeriodicSender::SendPacket, this);
151  NS_LOG_DEBUG ("Event Id: " << m_sendEvent.GetUid ());
152 }

102  m_low = ip1;

105  else

107  m_low = ip2;

118  if (m_low < L.m_low)
119  {
120  return true;
121  }
122  else if (m_low == L.m_low)
123  {
124  return (m_high < L.m_high);
125  }
126  else
127  {
128  return false;
129  }
130  }

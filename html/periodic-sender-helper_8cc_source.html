81 Ptr<Application>
82 PeriodicSenderHelper::InstallPriv (Ptr<Node> node) const
83 {
84  NS_LOG_FUNCTION (this << node);
85 
86  Ptr<PeriodicSender> app = m_factory.Create<PeriodicSender> ();
87 
88  Time interval;
89  if (m_period == Seconds (0))
90  {
91  double intervalProb = m_intervalProb->GetValue ();
92  NS_LOG_DEBUG ("IntervalProb = " << intervalProb);
93 
94  // Based on TR 45.820
95  if (intervalProb < 0.4)
96  {
97  interval = Days (1);
98  }
99  else if (0.4 <= intervalProb && intervalProb < 0.8)
100  {
101  interval = Hours (2);
102  }
103  else if (0.8 <= intervalProb && intervalProb < 0.95)
104  {
105  interval = Hours (1);
106  }
107  else
108  {
109  interval = Minutes (30);
110  }
111  }
112  else
113  {
114  interval = m_period;
115  }
116 
117  app->SetInterval (interval);
118  NS_LOG_DEBUG ("Created an application with interval = " <<
119  interval.GetHours () << " hours");
120 
121  app->SetInitialDelay (Seconds (m_initialDelay->GetValue (0, interval.GetSeconds ())));
122  app->SetPacketSize (m_pktSize);
123  if (m_pktSizeRV)
124  {
125  app->SetPacketSizeRandomVariable (m_pktSizeRV);
126  }
127 
128  app->SetNode (node);
129  node->AddApplication (app);
130 
131  return app;
132 }

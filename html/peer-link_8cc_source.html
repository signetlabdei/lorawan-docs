374 void
375 PeerLink::StateMachine (PeerEvent event, PmpReasonCode reasoncode)
376 {
377  switch (m_state)
378  {
379  case IDLE:
380  switch (event)
381  {
382  case CNCL:
383  case CLS_ACPT:
384  m_state = IDLE;
385  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, IDLE, IDLE);
386  break;
387  case REQ_RJCT:
388  SendPeerLinkClose (reasoncode);
389  break;
390  case ACTOPN:
391  m_state = OPN_SNT;
392  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, IDLE, OPN_SNT);
393  SendPeerLinkOpen ();
394  SetRetryTimer ();
395  break;
396  case OPN_ACPT:
397  m_state = OPN_RCVD;
398  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, IDLE, OPN_RCVD);
399  SendPeerLinkConfirm ();
400  SendPeerLinkOpen ();
401  SetRetryTimer ();
402  break;
403  default:
404  //11B.5.3.4 of 802.11s Draft D3.0
405  //All other events shall be ignored in this state
406  break;
407  }
408  break;
409  case OPN_SNT:
410  switch (event)
411  {
412  case TOR1:
413  SendPeerLinkOpen ();
414  m_retryCounter++;
415  SetRetryTimer ();
416  break;
417  case CNF_ACPT:
418  m_state = CNF_RCVD;
419  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, OPN_SNT, CNF_RCVD);
420  ClearRetryTimer ();
421  SetConfirmTimer ();
422  break;
423  case OPN_ACPT:
424  m_state = OPN_RCVD;
425  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, OPN_SNT, OPN_RCVD);
426  SendPeerLinkConfirm ();
427  break;
428  case CLS_ACPT:
429  m_state = HOLDING;
430  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, OPN_SNT, HOLDING);
431  ClearRetryTimer ();
432  SendPeerLinkClose (REASON11S_MESH_CLOSE_RCVD);
433  SetHoldingTimer ();
434  break;
435  case OPN_RJCT:
436  case CNF_RJCT:
437  m_state = HOLDING;
438  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, OPN_SNT, HOLDING);
439  ClearRetryTimer ();
440  SendPeerLinkClose (reasoncode);
441  SetHoldingTimer ();
442  break;
443  case TOR2:
444  m_state = HOLDING;
445  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, OPN_SNT, HOLDING);
446  ClearRetryTimer ();
447  SendPeerLinkClose (REASON11S_MESH_MAX_RETRIES);
448  SetHoldingTimer ();
449  break;
450  case CNCL:
451  m_state = HOLDING;
452  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, OPN_SNT, HOLDING);
453  ClearRetryTimer ();
454  SendPeerLinkClose (REASON11S_PEERING_CANCELLED);
455  SetHoldingTimer ();
456  break;
457  default:
458  //11B.5.3.5 of 802.11s Draft D3.0
459  //All other events shall be ignored in this state
460  break;
461  }
462  break;
463  case CNF_RCVD:
464  switch (event)
465  {
466  case CNF_ACPT:
467  break;
468  case OPN_ACPT:
469  m_state = ESTAB;
470  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, CNF_RCVD, ESTAB);
471  ClearConfirmTimer ();
472  SendPeerLinkConfirm ();
473  NS_ASSERT (m_peerMeshPointAddress != Mac48Address::GetBroadcast ());
474  break;
475  case CLS_ACPT:
476  m_state = HOLDING;
477  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, CNF_RCVD, HOLDING);
478  ClearConfirmTimer ();
479  SendPeerLinkClose (REASON11S_MESH_CLOSE_RCVD);
480  SetHoldingTimer ();
481  break;
482  case CNF_RJCT:
483  case OPN_RJCT:
484  m_state = HOLDING;
485  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, CNF_RCVD, HOLDING);
486  ClearConfirmTimer ();
487  SendPeerLinkClose (reasoncode);
488  SetHoldingTimer ();
489  break;
490  case CNCL:
491  m_state = HOLDING;
492  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, CNF_RCVD, HOLDING);
493  ClearConfirmTimer ();
494  SendPeerLinkClose (REASON11S_PEERING_CANCELLED);
495  SetHoldingTimer ();
496  break;
497  case TOC:
498  m_state = HOLDING;
499  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, CNF_RCVD, HOLDING);
500  SendPeerLinkClose (REASON11S_MESH_CONFIRM_TIMEOUT);
501  SetHoldingTimer ();
502  break;
503  default:
504  //11B.5.3.6 of 802.11s Draft D3.0
505  //All other events shall be ignored in this state
506  break;
507  }
508  break;
509  case OPN_RCVD:
510  switch (event)
511  {
512  case TOR1:
513  SendPeerLinkOpen ();
514  m_retryCounter++;
515  SetRetryTimer ();
516  break;
517  case CNF_ACPT:
518  m_state = ESTAB;
519  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, OPN_RCVD, ESTAB);
520  ClearRetryTimer ();
521  NS_ASSERT (m_peerMeshPointAddress != Mac48Address::GetBroadcast ());
522  break;
523  case CLS_ACPT:
524  m_state = HOLDING;
525  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, OPN_RCVD, HOLDING);
526  ClearRetryTimer ();
527  SendPeerLinkClose (REASON11S_MESH_CLOSE_RCVD);
528  SetHoldingTimer ();
529  break;
530  case OPN_RJCT:
531  case CNF_RJCT:
532  m_state = HOLDING;
533  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, OPN_RCVD, HOLDING);
534  ClearRetryTimer ();
535  SendPeerLinkClose (reasoncode);
536  SetHoldingTimer ();
537  break;
538  case TOR2:
539  m_state = HOLDING;
540  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, OPN_RCVD, HOLDING);
541  ClearRetryTimer ();
542  SendPeerLinkClose (REASON11S_MESH_MAX_RETRIES);
543  SetHoldingTimer ();
544  break;
545  case CNCL:
546  m_state = HOLDING;
547  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, OPN_RCVD, HOLDING);
548  ClearRetryTimer ();
549  SendPeerLinkClose (REASON11S_PEERING_CANCELLED);
550  SetHoldingTimer ();
551  break;
552  default:
553  //11B.5.3.7 of 802.11s Draft D3.0
554  //All other events shall be ignored in this state
555  break;
556  }
557  break;
558  case ESTAB:
559  switch (event)
560  {
561  case OPN_ACPT:
562  SendPeerLinkConfirm ();
563  break;
564  case CLS_ACPT:
565  m_state = HOLDING;
566  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, ESTAB, HOLDING);
567  SendPeerLinkClose (REASON11S_MESH_CLOSE_RCVD);
568  SetHoldingTimer ();
569  break;
570  case OPN_RJCT:
571  case CNF_RJCT:
572  m_state = HOLDING;
573  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, ESTAB, HOLDING);
574  ClearRetryTimer ();
575  SendPeerLinkClose (reasoncode);
576  SetHoldingTimer ();
577  break;
578  case CNCL:
579  m_state = HOLDING;
580  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, ESTAB, HOLDING);
581  SendPeerLinkClose (REASON11S_PEERING_CANCELLED);
582  SetHoldingTimer ();
583  break;
584  default:
585  //11B.5.3.8 of 802.11s Draft D3.0
586  //All other events shall be ignored in this state
587  break;
588  }
589  break;
590  case HOLDING:
591  switch (event)
592  {
593  case CLS_ACPT:
594  ClearHoldingTimer ();
595  // fall through:
596  case TOH:
597  m_state = IDLE;
598  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, HOLDING, IDLE);
599  break;
600  case OPN_ACPT:
601  case CNF_ACPT:
602  m_state = HOLDING;
603  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, HOLDING, HOLDING);
604  // reason not spec in D2.0
605  SendPeerLinkClose (REASON11S_PEERING_CANCELLED);
606  break;
607  case OPN_RJCT:
608  case CNF_RJCT:
609  m_state = HOLDING;
610  m_linkStatusCallback (m_interface, m_peerAddress, m_peerMeshPointAddress, HOLDING, HOLDING);
611  SendPeerLinkClose (reasoncode);
612  break;
613  default:
614  //11B.5.3.9 of 802.11s Draft D3.0
615  //All other events shall be ignored in this state
616  break;
617  }
618  break;
619  }

129 void
130 SixlowpanHc1ImplTest::DoRun (void)
131 {
132  // Create topology
133  InternetStackHelper internet;
134  internet.SetIpv4StackInstall (false);
135 
136  // Receiver Node
137  Ptr<Node> rxNode = CreateObject<Node> ();
138  internet.Install (rxNode);
139  Ptr<SimpleNetDevice> rxDev;
140  { // first interface
141  rxDev = CreateObject<SimpleNetDevice> ();
142  rxDev->SetAddress (Mac48Address::ConvertFrom (Mac48Address::Allocate ()));
143  rxNode->AddDevice (rxDev);
144 
145  Ptr<SixLowPanNetDevice> rxSix = CreateObject<SixLowPanNetDevice> ();
146  rxSix->SetAttribute ("ForceEtherType", BooleanValue (true) );
147  rxSix->SetAttribute ("Rfc6282", BooleanValue (false) );
148  rxNode->AddDevice (rxSix);
149  rxSix->SetNetDevice (rxDev);
150 
151  Ptr<Ipv6> ipv6 = rxNode->GetObject<Ipv6> ();
152  ipv6->AddInterface (rxDev);
153  uint32_t netdev_idx = ipv6->AddInterface (rxSix);
154  Ipv6InterfaceAddress ipv6Addr = Ipv6InterfaceAddress (Ipv6Address ("2001:0100::1"), Ipv6Prefix (64));
155  ipv6->AddAddress (netdev_idx, ipv6Addr);
156  ipv6->SetUp (netdev_idx);
157  }
158 
159  // Sender Node
160  Ptr<Node> txNode = CreateObject<Node> ();
161  internet.Install (txNode);
162  Ptr<SimpleNetDevice> txDev;
163  {
164  txDev = CreateObject<SimpleNetDevice> ();
165  txDev->SetAddress (Mac48Address::ConvertFrom (Mac48Address::Allocate ()));
166  txNode->AddDevice (txDev);
167 
168  Ptr<SixLowPanNetDevice> txSix = CreateObject<SixLowPanNetDevice> ();
169  txSix->SetAttribute ("ForceEtherType", BooleanValue (true) );
170  txSix->SetAttribute ("Rfc6282", BooleanValue (false) );
171  txNode->AddDevice (txSix);
172  txSix->SetNetDevice (txDev);
173 
174  Ptr<Ipv6> ipv6 = txNode->GetObject<Ipv6> ();
175  ipv6->AddInterface (txDev);
176  uint32_t netdev_idx = ipv6->AddInterface (txSix);
177  Ipv6InterfaceAddress ipv6Addr = Ipv6InterfaceAddress (Ipv6Address ("2001:0100::2"), Ipv6Prefix (64));
178  ipv6->AddAddress (netdev_idx, ipv6Addr);
179  ipv6->SetUp (netdev_idx);
180  }
181 
182  // link the two nodes
183  Ptr<SimpleChannel> channel1 = CreateObject<SimpleChannel> ();
184  rxDev->SetChannel (channel1);
185  txDev->SetChannel (channel1);
186 
187  // Create the UDP sockets
188  Ptr<SocketFactory> rxSocketFactory = rxNode->GetObject<UdpSocketFactory> ();
189  Ptr<Socket> rxSocket = rxSocketFactory->CreateSocket ();
190  NS_TEST_EXPECT_MSG_EQ (rxSocket->Bind (Inet6SocketAddress (Ipv6Address ("2001:0100::1"), 1234)), 0, "trivial");
191  rxSocket->SetRecvCallback (MakeCallback (&SixlowpanHc1ImplTest::ReceivePkt, this));
192 
193  Ptr<SocketFactory> txSocketFactory = txNode->GetObject<UdpSocketFactory> ();
194  Ptr<Socket> txSocket = txSocketFactory->CreateSocket ();
195  txSocket->SetAllowBroadcast (true);
196  // ------ Now the tests ------------
197 
198  // Unicast test
199  SendData (txSocket, "2001:0100::1");
200  NS_TEST_EXPECT_MSG_EQ (m_receivedPacket->GetSize (), 180, "trivial");
201  uint8_t rxBuffer [180];
202  uint8_t txBuffer [180] = "\"Can you tell me where my country lies?\" \\ said the unifaun to his true love's eyes. \\ \"It lies with me!\" cried the Queen of Maybe \\ - for her merchandise, he traded in his prize.";
203  m_receivedPacket->CopyData (rxBuffer, 180);
204  NS_TEST_EXPECT_MSG_EQ (memcmp (rxBuffer, txBuffer, 180), 0, "trivial");
205 
206  m_receivedPacket->RemoveAllByteTags ();
207 
208  Simulator::Destroy ();
209 
210 }

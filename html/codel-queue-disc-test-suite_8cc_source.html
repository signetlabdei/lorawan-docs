34 // The following code borrowed from Linux codel.h, for unit testing
35 #define REC_INV_SQRT_BITS_ns3 (8 * sizeof(uint16_t))
36 /* or sizeof_in_bits(rec_inv_sqrt) */
37 /* needed shift to get a Q0.32 number from rec_inv_sqrt */
38 #define REC_INV_SQRT_SHIFT_ns3 (32 - REC_INV_SQRT_BITS_ns3)
39 
40 static uint16_t _codel_Newton_step (uint16_t rec_inv_sqrt, uint32_t count)
41 {
42  uint32_t invsqrt = ((uint32_t)rec_inv_sqrt) << REC_INV_SQRT_SHIFT_ns3;
43  uint32_t invsqrt2 = ((uint64_t)invsqrt * invsqrt) >> 32;
44  uint64_t val = (3LL << 32) - ((uint64_t)count * invsqrt2);
45 
46  val >>= 2; /* avoid overflow in following multiply */
47  val = (val * invsqrt) >> (32 - 2 + 1);
48  return static_cast<uint16_t>(val >> REC_INV_SQRT_SHIFT_ns3);
49 }
50 
51 static uint32_t _reciprocal_scale (uint32_t val, uint32_t ep_ro)
52 {
53  return (uint32_t)(((uint64_t)val * ep_ro) >> 32);
54 }

518 void
519 CoDelQueueDiscBasicDrop::Dequeue (Ptr<CoDelQueueDisc> queue, uint32_t modeSize)
520 {
521  uint32_t initialDropCount = queue->GetStats ().GetNDroppedPackets (CoDelQueueDisc::TARGET_EXCEEDED_DROP);
522  uint32_t initialQSize = queue->GetCurrentSize ().GetValue ();
523  uint32_t initialDropNext = queue->GetDropNext ();
524  Time currentTime = Simulator::Now ();
525  uint32_t currentDropCount = 0;
526 
527  if (initialDropCount > 0 && currentTime.GetMicroSeconds () >= initialDropNext)
528  {
529  queue->TraceConnectWithoutContext ("DropNext", MakeCallback (&CoDelQueueDiscBasicDrop::DropNextTracer, this));
530  }
531 
532  if (initialQSize != 0)
533  {
534  Ptr<QueueDiscItem> item = queue->Dequeue ();
535  if (initialDropCount == 0 && currentTime > queue->GetTarget ())
536  {
537  if (currentTime < queue->GetInterval ())
538  {
539  currentDropCount = queue->GetStats ().GetNDroppedPackets (CoDelQueueDisc::TARGET_EXCEEDED_DROP);
540  NS_TEST_EXPECT_MSG_EQ (currentDropCount, 0, "We are not in dropping state."
541  "Sojourn time has just gone above target from below."
542  "Hence, there should be no packet drops");
543  NS_TEST_EXPECT_MSG_EQ (queue->GetCurrentSize ().GetValue (), initialQSize - modeSize, "There should be 1 packet dequeued.");
544 
545  }
546  else if (currentTime >= queue->GetInterval ())
547  {
548  currentDropCount = queue->GetStats ().GetNDroppedPackets (CoDelQueueDisc::TARGET_EXCEEDED_DROP);
549  NS_TEST_EXPECT_MSG_EQ (queue->GetCurrentSize ().GetValue (), initialQSize - 2 * modeSize, "Sojourn time has been above target for at least interval."
550  "We enter the dropping state, perform initial packet drop, and dequeue the next."
551  "So there should be 2 more packets dequeued.");
552  NS_TEST_EXPECT_MSG_EQ (currentDropCount, 1, "There should be 1 packet drop");
553  }
554  }
555  else if (initialDropCount > 0)
556  { // In dropping state
557  if (currentTime.GetMicroSeconds () < initialDropNext)
558  {
559  currentDropCount = queue->GetStats ().GetNDroppedPackets (CoDelQueueDisc::TARGET_EXCEEDED_DROP);
560  NS_TEST_EXPECT_MSG_EQ (queue->GetCurrentSize ().GetValue (), initialQSize - modeSize, "We are in dropping state."
561  "Sojourn is still above target."
562  "However, it's not time for next drop."
563  "So there should be only 1 more packet dequeued");
564 
565  NS_TEST_EXPECT_MSG_EQ (currentDropCount, 1, "There should still be only 1 packet drop from the last dequeue");
566  }
567  else if (currentTime.GetMicroSeconds () >= initialDropNext)
568  {
569  currentDropCount = queue->GetStats ().GetNDroppedPackets (CoDelQueueDisc::TARGET_EXCEEDED_DROP);
570  NS_TEST_EXPECT_MSG_EQ (queue->GetCurrentSize ().GetValue (), initialQSize - (m_dropNextCount + 1) * modeSize, "We are in dropping state."
571  "It's time for next drop."
572  "The number of packets dequeued equals to the number of times m_dropNext is updated plus initial dequeue");
573  NS_TEST_EXPECT_MSG_EQ (currentDropCount, 1 + m_dropNextCount, "The number of drops equals to the number of times m_dropNext is updated plus 1 from last dequeue");
574  }
575  }
576  }

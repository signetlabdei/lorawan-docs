75 void
76 GtpuHeader::Serialize (Buffer::Iterator start) const
77 {
78  Buffer::Iterator i = start;
79  uint8_t firstByte = m_version << 5 | m_protocolType << 4 | 0x1 << 3;
80  firstByte |= m_extensionHeaderFlag << 2 | m_sequenceNumberFlag << 1 | m_nPduNumberFlag;
81  i.WriteU8 (firstByte);
82  i.WriteU8 (m_messageType);
83  i.WriteHtonU16 (m_length);
84  i.WriteHtonU32 (m_teid);
85  i.WriteHtonU16 (m_sequenceNumber);
86  i.WriteU8 (m_nPduNumber);
87  i.WriteU8 (m_nextExtensionType);
88 
89 }
90 uint32_t
91 GtpuHeader::Deserialize (Buffer::Iterator start)
92 {
93  Buffer::Iterator i = start;
94  uint8_t firstByte = i.ReadU8 ();
95  m_version = firstByte >> 5 & 0x7;
96  m_protocolType = firstByte >> 4 & 0x1;
97  m_extensionHeaderFlag = firstByte >> 2 & 0x1;
98  m_sequenceNumberFlag = firstByte >> 1 & 0x1;
99  m_nPduNumberFlag = firstByte & 0x1;
100  m_messageType = i.ReadU8 ();
101  m_length = i.ReadNtohU16 ();
102  m_teid = i.ReadNtohU32 ();
103  m_sequenceNumber = i.ReadNtohU16 ();
104  m_nPduNumber = i.ReadU8 ();
105  m_nextExtensionType = i.ReadU8 ();
106  return GetSerializedSize ();
107 }

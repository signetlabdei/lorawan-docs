113  std::vector<Vector>
114  HexGridPositionAllocator::AddRing (std::vector<Vector> positions)
115  {
116  NS_LOG_FUNCTION (this);
117 
118  // Make a copy of the vector
119  std::vector<Vector> copy = positions;
120 
121  // Iterate on the given vector
122  for (std::vector<Vector>::iterator it = positions.begin ();
123  it != positions.end (); it++)
124  {
125  // Get the current position
126  Vector currentPosition = *it;
127  NS_LOG_DEBUG ("Current position " << currentPosition);
128 
129  // Iterate to create the 6 surrounding positions
130  // The angle is with respect to a vertical line
131  Vector newPosition;
132  for (double angle = 0; angle < 2*pi; angle+=pi/3)
133  {
134  newPosition = Vector (currentPosition.x+2*m_radius*std::sin(angle),
135  currentPosition.y+2*m_radius*std::cos(angle),
136  currentPosition.z);
137  NS_LOG_DEBUG ("New position: " << newPosition);
138 
139  // If the newly created position is not already in the copy, add it
140  bool found = false;
141  for (std::vector<Vector>::iterator it = copy.begin ();
142  it != copy.end (); it++)
143  {
144  // If the vector is already in the vector
145  // 1 is an EPSILON used to determine whether two floats are equal
146  if (CalculateDistance(newPosition, *it) < 10)
147  {
148  found = true;
149  break;
150  }
151  }
152  if (found == false)
153  {
154  NS_LOG_DEBUG ("Adding position " << newPosition);
155  copy.push_back (newPosition);
156  }
157  }
158  }
159  return copy;
160  }

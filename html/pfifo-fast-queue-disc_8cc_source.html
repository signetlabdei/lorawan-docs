64 bool
65 PfifoFastQueueDisc::DoEnqueue (Ptr<QueueDiscItem> item)
66 {
67  NS_LOG_FUNCTION (this << item);
68 
69  if (GetCurrentSize () >= GetMaxSize ())
70  {
71  NS_LOG_LOGIC ("Queue disc limit exceeded -- dropping packet");
72  DropBeforeEnqueue (item, LIMIT_EXCEEDED_DROP);
73  return false;
74  }
75 
76  uint8_t priority = 0;
77  SocketPriorityTag priorityTag;
78  if (item->GetPacket ()->PeekPacketTag (priorityTag))
79  {
80  priority = priorityTag.GetPriority ();
81  }
82 
83  uint32_t band = prio2band[priority & 0x0f];
84 
85  bool retval = GetInternalQueue (band)->Enqueue (item);
86 
87  // If Queue::Enqueue fails, QueueDisc::DropBeforeEnqueue is called by the
88  // internal queue because QueueDisc::AddInternalQueue sets the trace callback
89 
90  if (!retval)
91  {
92  NS_LOG_WARN ("Packet enqueue failed. Check the size of the internal queues");
93  }
94 
95  NS_LOG_LOGIC ("Number packets band " << band << ": " << GetInternalQueue (band)->GetNPackets ());
96 
97  return retval;
98 }

102 {
103  NS_LOG_FUNCTION (this);
104 
105  Ptr<QueueDiscItem> item;
106 
107  for (uint32_t i = 0; i < GetNInternalQueues (); i++)
108  {

142 bool
143 PfifoFastQueueDisc::CheckConfig (void)
144 {
145  NS_LOG_FUNCTION (this);
146  if (GetNQueueDiscClasses () > 0)
147  {
148  NS_LOG_ERROR ("PfifoFastQueueDisc cannot have classes");
149  return false;
150  }
151 
152  if (GetNPacketFilters () != 0)
153  {
154  NS_LOG_ERROR ("PfifoFastQueueDisc needs no packet filter");
155  return false;
156  }
157 
158  if (GetNInternalQueues () == 0)
159  {
160  // create 3 DropTail queues with GetLimit() packets each
161  ObjectFactory factory;
162  factory.SetTypeId ("ns3::DropTailQueue<QueueDiscItem>");
163  factory.Set ("MaxSize", QueueSizeValue (GetMaxSize ()));
164  AddInternalQueue (factory.Create<InternalQueue> ());
165  AddInternalQueue (factory.Create<InternalQueue> ());
166  AddInternalQueue (factory.Create<InternalQueue> ());
167  }
168 
169  if (GetNInternalQueues () != 3)
170  {
171  NS_LOG_ERROR ("PfifoFastQueueDisc needs 3 internal queues");
172  return false;
173  }
174 
175  if (GetInternalQueue (0)-> GetMaxSize ().GetUnit () != QueueSizeUnit::PACKETS ||
176  GetInternalQueue (1)-> GetMaxSize ().GetUnit () != QueueSizeUnit::PACKETS ||
177  GetInternalQueue (2)-> GetMaxSize ().GetUnit () != QueueSizeUnit::PACKETS)
178  {
179  NS_LOG_ERROR ("PfifoFastQueueDisc needs 3 internal queues operating in packet mode");
180  return false;
181  }
182 
183  for (uint8_t i = 0; i < 2; i++)
184  {
185  if (GetInternalQueue (i)->GetMaxSize () < GetMaxSize ())
186  {
187  NS_LOG_ERROR ("The capacity of some internal queue(s) is less than the queue disc capacity");
188  return false;
189  }
190  }
191 
192  return true;
193 }

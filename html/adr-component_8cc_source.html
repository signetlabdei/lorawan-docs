92 void
93 AdrComponent::BeforeSendingReply (Ptr<EndDeviceStatus> status,
94  Ptr<NetworkStatus> networkStatus)
95 {
96  NS_LOG_FUNCTION (this << status << networkStatus);
97 
98  Ptr<Packet> myPacket = status->GetLastPacketReceivedFromDevice ()->Copy ();
99  LorawanMacHeader mHdr;
100  LoraFrameHeader fHdr;
101  fHdr.SetAsUplink ();
102  myPacket->RemoveHeader (mHdr);
103  myPacket->RemoveHeader (fHdr);
104 
105  //Execute the ADR algotithm only if the request bit is set
106  if (fHdr.GetAdr ())
107  {
108  if (int(status->GetReceivedPacketList ().size ()) < historyRange)
109  {
110  NS_LOG_ERROR ("Not enough packets received by this device (" << status->GetReceivedPacketList ().size () << ") for the algorithm to work (need " << historyRange << ")");
111  }
112  else
113  {
114  NS_LOG_DEBUG ("New ADR request");
115 
116  //Get the SF used by the device
117  uint8_t spreadingFactor = status->GetFirstReceiveWindowSpreadingFactor ();
118 
119  //Get the device transmission power (dBm)
120  uint8_t transmissionPower = status->GetMac ()->GetTransmissionPower ();
121 
122  //New parameters for the end-device
123  uint8_t newDataRate;
124  uint8_t newTxPower;
125 
126  //ADR Algorithm
127  AdrImplementation (&newDataRate,
128  &newTxPower,
129  status);
130 
131  // Change the power back to the default if we don't want to change it
132  if (!m_toggleTxPower)
133  {
134  newTxPower = transmissionPower;
135  }
136 
137  if (newDataRate != SfToDr (spreadingFactor) || newTxPower != transmissionPower)
138  {
139  //Create a list with mandatory channel indexes
140  int channels[] = {0, 1, 2};
141  std::list<int> enabledChannels (channels,
142  channels + sizeof(channels) /
143  sizeof(int));
144 
145  //Repetitions Setting
146  const int rep = 1;
147 
148  NS_LOG_DEBUG ("Sending LinkAdrReq with DR = " << (unsigned)newDataRate << " and TP = " << (unsigned)newTxPower << " dBm");
149 
150  status->m_reply.frameHeader.AddLinkAdrReq (newDataRate,
151  GetTxPowerIndex (newTxPower),
152  enabledChannels,
153  rep);
154  status->m_reply.frameHeader.SetAsDownlink ();
155  status->m_reply.macHeader.SetMType (LorawanMacHeader::UNCONFIRMED_DATA_DOWN);
156 
157  status->m_reply.needsReply = true;
158  }
159  else
160  {
161  NS_LOG_DEBUG ("Skipped request");
162  }
163  }
164  }
165  else
166  {
167  // Do nothing
168  }
169 }

177 void AdrComponent::AdrImplementation (uint8_t *newDataRate,
178  uint8_t *newTxPower,
179  Ptr<EndDeviceStatus> status)
180 {
181  //Compute the maximum or median SNR, based on the boolean value historyAveraging
182  double m_SNR = 0;
183  switch (historyAveraging)
184  {
185  case AdrComponent::AVERAGE:
186  m_SNR = GetAverageSNR (status->GetReceivedPacketList (),
187  historyRange);
188  break;
189  case AdrComponent::MAXIMUM:
190  m_SNR = GetMaxSNR (status->GetReceivedPacketList (),
191  historyRange);
192  break;
193  case AdrComponent::MINIMUM:
194  m_SNR = GetMinSNR (status->GetReceivedPacketList (),
195  historyRange);
196  }
197 
198  NS_LOG_DEBUG ("m_SNR = " << m_SNR);
199 
200  //Get the SF used by the device
201  uint8_t spreadingFactor = status->GetFirstReceiveWindowSpreadingFactor ();
202 
203  NS_LOG_DEBUG ("SF = " << (unsigned)spreadingFactor);
204 
205  //Get the device data rate and use it to get the SNR demodulation treshold
206  double req_SNR = treshold[SfToDr (spreadingFactor)];
207 
208  NS_LOG_DEBUG ("Required SNR = " << req_SNR);
209 
210  //Get the device transmission power (dBm)
211  double transmissionPower = status->GetMac ()->GetTransmissionPower ();
212 
213  NS_LOG_DEBUG ("Transmission Power = " << transmissionPower);
214 
215  //Compute the SNR margin taking into consideration the SNR of
216  //previously received packets
217  double margin_SNR = m_SNR - req_SNR;
218 
219  NS_LOG_DEBUG ("Margin = " << margin_SNR);
220 
221  //Number of steps to decrement the SF (thereby increasing the Data Rate)
222  //and the TP.
223  int steps = std::floor (margin_SNR / 3);
224 
225  NS_LOG_DEBUG ("steps = " << steps);
226 
227  //If the number of steps is positive (margin_SNR is positive, so its
228  //decimal value is high) increment the data rate, if there are some
229  //leftover steps after reaching the maximum possible data rate
230  //(corresponding to the minimum SF) decrement the transmission power as
231  //well for the number of steps left.
232  //If, on the other hand, the number of steps is negative (margin_SNR is
233  //negative, so its decimal value is low) increase the transmission power
234  //(note that the SF is not incremented as this particular algorithm
235  //expects the node itself to raise its SF whenever necessary).
236  while (steps > 0 && spreadingFactor > min_spreadingFactor)
237  {
238  spreadingFactor--;
239  steps--;
240  NS_LOG_DEBUG ("Decreased SF by 1");
241  }
242  while (steps > 0 && transmissionPower > min_transmissionPower)
243  {
244  transmissionPower -= 2;
245  steps--;
246  NS_LOG_DEBUG ("Decreased Ptx by 2");
247  }
248  while (steps < 0 && transmissionPower < max_transmissionPower)
249  {
250  transmissionPower += 2;
251  steps++;
252  NS_LOG_DEBUG ("Increased Ptx by 2");
253  }
254 
255  *newDataRate = SfToDr (spreadingFactor);
256  *newTxPower = transmissionPower;
257 }

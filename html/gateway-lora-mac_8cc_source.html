53 void
54 GatewayLoraMac::Send (Ptr<Packet> packet)
55 {
56  NS_LOG_FUNCTION (this << packet);
57 
58  // Get DataRate to send this packet with
59  LoraTag tag;
60  packet->RemovePacketTag (tag);
61  uint8_t dataRate = tag.GetDataRate ();
62  double frequency = tag.GetFrequency ();
63  NS_LOG_DEBUG ("DR: " << unsigned (dataRate));
64  NS_LOG_DEBUG ("SF: " << unsigned (GetSfFromDataRate (dataRate)));
65  NS_LOG_DEBUG ("BW: " << GetBandwidthFromDataRate (dataRate));
66  NS_LOG_DEBUG ("Freq: " << frequency << " MHz");
67  packet->AddPacketTag (tag);
68 
69  LoraTxParameters params;
70  params.sf = GetSfFromDataRate (dataRate);
71  params.headerDisabled = false;
72  params.codingRate = 1;
73  params.bandwidthHz = GetBandwidthFromDataRate (dataRate);
74  params.nPreamble = 8;
75  params.crcEnabled = 1;
76  params.lowDataRateOptimizationEnabled = 0;
77 
78  // Get the duration
79  Time duration = m_phy->GetOnAirTime (packet, params);
80 
81  NS_LOG_DEBUG ("Duration: " << duration.GetSeconds ());
82 
83  // Find the channel with the desired frequency
84  double sendingPower = m_channelHelper.GetTxPowerForChannel
85  (CreateObject<LogicalLoraChannel> (frequency));
86 
87  // Add the event to the channelHelper to keep track of duty cycle
88  m_channelHelper.AddEvent (duration, CreateObject<LogicalLoraChannel>
89  (frequency));
90 
91  // Send the packet to the PHY layer to send it on the channel
92  m_phy->Send (packet, params, frequency, sendingPower);
93 }

101 void
102 GatewayLoraMac::Receive (Ptr<Packet const> packet)
103 {
104  NS_LOG_FUNCTION (this << packet);
105 
106  // Make a copy of the packet to work on
107  Ptr<Packet> packetCopy = packet->Copy ();
108 
109  // Only forward the packet if it's uplink
110  LoraMacHeader macHdr;
111  packetCopy->PeekHeader (macHdr);
112 
113  if (macHdr.IsUplink ())
114  {
115  m_device->GetObject<LoraNetDevice> ()->Receive (packetCopy);
116  m_receivedPacket (packet);
117  }
118  else
119  {
120  NS_LOG_DEBUG ("Not forwarding downlink message to NetDevice");
121  }

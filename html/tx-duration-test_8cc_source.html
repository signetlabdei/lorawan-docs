89 bool
90 TxDurationTest::CheckPayloadDuration (uint32_t size, WifiMode payloadMode, uint16_t channelWidth, uint16_t guardInterval, WifiPreamble preamble, Time knownDuration)
91 {
92  WifiTxVector txVector;
93  txVector.SetMode (payloadMode);
94  txVector.SetPreambleType (preamble);
95  txVector.SetChannelWidth (channelWidth);
96  txVector.SetGuardInterval (guardInterval);
97  txVector.SetNss (1);
98  txVector.SetStbc (0);
99  txVector.SetNess (0);
100  uint16_t testedFrequency = CHANNEL_1_MHZ;
101  Ptr<YansWifiPhy> phy = CreateObject<YansWifiPhy> ();
102  if (payloadMode.GetModulationClass () == WIFI_MOD_CLASS_OFDM
103  || payloadMode.GetModulationClass () == WIFI_MOD_CLASS_HT
104  || payloadMode.GetModulationClass () == WIFI_MOD_CLASS_VHT
105  || payloadMode.GetModulationClass () == WIFI_MOD_CLASS_HE)
106  {
107  testedFrequency = CHANNEL_36_MHZ;
108  }
109  Time calculatedDuration = phy->GetPayloadDuration (size, txVector, testedFrequency);
110  if (calculatedDuration != knownDuration)
111  {
112  std::cerr << "size=" << size
113  << " mode=" << payloadMode
114  << " channelWidth=" << channelWidth
115  << " guardInterval=" << guardInterval
116  << " datarate=" << payloadMode.GetDataRate (channelWidth, guardInterval, 1)
117  << " known=" << knownDuration
118  << " calculated=" << calculatedDuration
119  << std::endl;
120  return false;
121  }
122  if (payloadMode.GetModulationClass () == WIFI_MOD_CLASS_HT || payloadMode.GetModulationClass () == WIFI_MOD_CLASS_HE)
123  {
124  //Durations vary depending on frequency; test also 2.4 GHz (bug 1971)
125  testedFrequency = CHANNEL_1_MHZ;
126  calculatedDuration = phy->GetPayloadDuration (size, txVector, testedFrequency);
127  knownDuration += MicroSeconds (6);
128  if (calculatedDuration != knownDuration)
129  {
130  std::cerr << "size=" << size
131  << " mode=" << payloadMode
132  << " channelWidth=" << channelWidth
133  << " guardInterval=" << guardInterval
134  << " datarate=" << payloadMode.GetDataRate (channelWidth, guardInterval, 1)
135  << " known=" << knownDuration
136  << " calculated=" << calculatedDuration
137  << std::endl;
138  return false;
139  }
140  }
141  return true;
142 }

61 AdhocWifiMac::SetAddress (Mac48Address address)
62 {
63  NS_LOG_FUNCTION (this << address);
64  //In an IBSS, the BSSID is supposed to be generated per Section
65  //11.1.3 of IEEE 802.11. We don't currently do this - instead we
66  //make an IBSS STA a bit like an AP, with the BSSID for frames
67  //transmitted by each STA set to that STA's address.
68  //
69  //This is why we're overriding this method.
70  RegularWifiMac::SetAddress (address);
71  RegularWifiMac::SetBssid (address);
72 }

74 void
75 AdhocWifiMac::Enqueue (Ptr<Packet> packet, Mac48Address to)
76 {
77  NS_LOG_FUNCTION (this << packet << to);
78  if (m_stationManager->IsBrandNew (to))
79  {
80  //In ad hoc mode, we assume that every destination supports all the rates we support.
81  if (GetHtSupported ())
82  {
83  m_stationManager->AddAllSupportedMcs (to);
84  m_stationManager->AddStationHtCapabilities (to, GetHtCapabilities ());
85  }
86  if (GetVhtSupported ())
87  {
88  m_stationManager->AddStationVhtCapabilities (to, GetVhtCapabilities ());
89  }
90  if (GetHeSupported ())
91  {
92  m_stationManager->AddStationHeCapabilities (to, GetHeCapabilities ());
93  }
94  m_stationManager->AddAllSupportedModes (to);
95  m_stationManager->RecordDisassociated (to);
96  }
97 
98  WifiMacHeader hdr;
99 
100  //If we are not a QoS STA then we definitely want to use AC_BE to
101  //transmit the packet. A TID of zero will map to AC_BE (through \c
102  //QosUtilsMapTidToAc()), so we use that as our default here.
103  uint8_t tid = 0;
104 
105  //For now, a STA that supports QoS does not support non-QoS
106  //associations, and vice versa. In future the STA model should fall
107  //back to non-QoS if talking to a peer that is also non-QoS. At
108  //that point there will need to be per-station QoS state maintained
109  //by the association state machine, and consulted here.
110  if (GetQosSupported ())
111  {
112  hdr.SetType (WIFI_MAC_QOSDATA);
113  hdr.SetQosAckPolicy (WifiMacHeader::NORMAL_ACK);
114  hdr.SetQosNoEosp ();
115  hdr.SetQosNoAmsdu ();
116  //Transmission of multiple frames in the same TXOP is not
117  //supported for now
118  hdr.SetQosTxopLimit (0);
119 
120  //Fill in the QoS control field in the MAC header
121  tid = QosUtilsGetTidForPacket (packet);
122  //Any value greater than 7 is invalid and likely indicates that
123  //the packet had no QoS tag, so we revert to zero, which will
124  //mean that AC_BE is used.
125  if (tid > 7)
126  {
127  tid = 0;
128  }
129  hdr.SetQosTid (tid);
130  }
131  else
132  {
133  hdr.SetType (WIFI_MAC_DATA);
134  }
135 
136  if (GetHtSupported ())
137  {
138  hdr.SetNoOrder (); // explicitly set to 0 for the time being since HT control field is not yet implemented (set it to 1 when implemented)
139  }
140  hdr.SetAddr1 (to);
141  hdr.SetAddr2 (m_low->GetAddress ());
142  hdr.SetAddr3 (GetBssid ());
143  hdr.SetDsNotFrom ();
144  hdr.SetDsNotTo ();
145 
146  if (GetQosSupported ())
147  {
148  //Sanity check that the TID is valid
149  NS_ASSERT (tid < 8);
150  m_edca[QosUtilsMapTidToAc (tid)]->Queue (packet, hdr);
151  }
152  else
153  {
154  m_txop->Queue (packet, hdr);
155  }
156 }

170 void
171 AdhocWifiMac::Receive (Ptr<WifiMacQueueItem> mpdu)
172 {
173  NS_LOG_FUNCTION (this << *mpdu);
174  const WifiMacHeader* hdr = &mpdu->GetHeader ();
175  NS_ASSERT (!hdr->IsCtl ());
176  Mac48Address from = hdr->GetAddr2 ();
177  Mac48Address to = hdr->GetAddr1 ();
178  if (m_stationManager->IsBrandNew (from))
179  {
180  //In ad hoc mode, we assume that every destination supports all the rates we support.
181  if (GetHtSupported ())
182  {
183  m_stationManager->AddAllSupportedMcs (from);
184  m_stationManager->AddStationHtCapabilities (from, GetHtCapabilities ());
185  }
186  if (GetVhtSupported ())
187  {
188  m_stationManager->AddStationVhtCapabilities (from, GetVhtCapabilities ());
189  }
190  if (GetHeSupported ())
191  {
192  m_stationManager->AddStationHeCapabilities (from, GetHeCapabilities ());
193  }
194  m_stationManager->AddAllSupportedModes (from);
195  m_stationManager->RecordDisassociated (from);
196  }
197  if (hdr->IsData ())
198  {
199  if (hdr->IsQosData () && hdr->IsQosAmsdu ())
200  {
201  NS_LOG_DEBUG ("Received A-MSDU from" << from);
202  DeaggregateAmsduAndForward (mpdu);
203  }
204  else
205  {
206  ForwardUp (mpdu->GetPacket ()->Copy (), from, to);
207  }
208  return;
209  }
210 
211  //Invoke the receive handler of our parent class to deal with any
212  //other frames. Specifically, this will handle Block Ack-related
213  //Management Action frames.
214  RegularWifiMac::Receive (mpdu);
215 }

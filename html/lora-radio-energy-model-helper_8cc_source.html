73 LoraRadioEnergyModelHelper::DoInstall (Ptr<NetDevice> device,
74  Ptr<EnergySource> source) const
75 {
76  NS_ASSERT (device != NULL);
77  NS_ASSERT (source != NULL);
78  // check if device is LoraNetDevice
79  std::string deviceName = device->GetInstanceTypeId ().GetName ();
80  if (deviceName.compare ("ns3::LoraNetDevice") != 0)
81  {
82  NS_FATAL_ERROR ("NetDevice type is not LoraNetDevice!");
83  }
84  Ptr<Node> node = device->GetNode ();
85  Ptr<LoraRadioEnergyModel> model = m_radioEnergy.Create ()->GetObject<LoraRadioEnergyModel> ();
86  NS_ASSERT (model != NULL);
87  // set energy source pointer
88  model->SetEnergySource (source);
89 
90  // set energy depletion callback
91  // if none is specified, make a callback to EndDeviceLoraPhy::SetSleepMode
92  Ptr<LoraNetDevice> loraDevice = device->GetObject<LoraNetDevice> ();
93  Ptr<EndDeviceLoraPhy> loraPhy = loraDevice->GetPhy ()->GetObject<EndDeviceLoraPhy> ();
94  // add model to device model list in energy source
95  source->AppendDeviceEnergyModel (model);
96  // create and register energy model phy listener
97  loraPhy->RegisterListener (model->GetPhyListener ());
98 
99  if (m_txCurrentModel.GetTypeId ().GetUid ())
100  {
101  Ptr<LoraTxCurrentModel> txcurrent = m_txCurrentModel.Create<LoraTxCurrentModel> ();
102  model->SetTxCurrentModel (txcurrent);
103  }
104  return model;

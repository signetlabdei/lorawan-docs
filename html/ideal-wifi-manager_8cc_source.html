103 void
104 IdealWifiManager::DoInitialize ()
105 {
106  NS_LOG_FUNCTION (this);
107  WifiMode mode;
108  WifiTxVector txVector;
109  uint8_t nss = 1;
110  uint8_t nModes = GetPhy ()->GetNModes ();
111  for (uint8_t i = 0; i < nModes; i++)
112  {
113  mode = GetPhy ()->GetMode (i);
114  txVector.SetChannelWidth (GetChannelWidthForNonHtMode (mode));
115  txVector.SetNss (nss);
116  txVector.SetMode (mode);
117  NS_LOG_DEBUG ("Initialize, adding mode = " << mode.GetUniqueName ());
118  AddSnrThreshold (txVector, GetPhy ()->CalculateSnr (txVector, m_ber));
119  }
120  // Add all MCSes
121  if (GetHtSupported ())
122  {
123  nModes = GetPhy ()->GetNMcs ();
124  for (uint8_t i = 0; i < nModes; i++)
125  {
126  for (uint16_t j = 20; j <= GetPhy ()->GetChannelWidth (); j *= 2)
127  {
128  txVector.SetChannelWidth (j);
129  mode = GetPhy ()->GetMcs (i);
130  if (mode.GetModulationClass () == WIFI_MOD_CLASS_HT)
131  {
132  uint16_t guardInterval = GetShortGuardIntervalSupported () ? 400 : 800;
133  txVector.SetGuardInterval (guardInterval);
134  //derive NSS from the MCS index
135  nss = (mode.GetMcsValue () / 8) + 1;
136  NS_LOG_DEBUG ("Initialize, adding mode = " << mode.GetUniqueName () <<
137  " channel width " << j <<
138  " nss " << +nss <<
139  " GI " << guardInterval);
140  NS_LOG_DEBUG ("In SetupPhy, adding mode = " << mode.GetUniqueName ());
141  txVector.SetNss (nss);
142  txVector.SetMode (mode);
143  AddSnrThreshold (txVector, GetPhy ()->CalculateSnr (txVector, m_ber));
144  }
145  else //VHT or HE
146  {
147  uint16_t guardInterval;
148  if (mode.GetModulationClass () == WIFI_MOD_CLASS_VHT)
149  {
150  guardInterval = GetShortGuardIntervalSupported () ? 400 : 800;
151  }
152  else
153  {
154  guardInterval = GetGuardInterval ();
155  }
156  txVector.SetGuardInterval (guardInterval);
157  for (uint8_t k = 1; k <= GetPhy ()->GetMaxSupportedTxSpatialStreams (); k++)
158  {
159  NS_LOG_DEBUG ("Initialize, adding mode = " << mode.GetUniqueName () <<
160  " channel width " << j <<
161  " nss " << +k <<
162  " GI " << guardInterval);
163  if (mode.IsAllowed (j, k))
164  {
165  NS_LOG_DEBUG ("In SetupPhy, adding mode = " << mode.GetUniqueName ());
166  txVector.SetNss (k);
167  txVector.SetMode (mode);
168  AddSnrThreshold (txVector, GetPhy ()->CalculateSnr (txVector, m_ber));
169  }
170  else
171  {
172  NS_LOG_DEBUG ("In SetupPhy, mode = " << mode.GetUniqueName () << " disallowed");
173  }
174  }
175  }
176  }
177  }
178  }

181 double
182 IdealWifiManager::GetSnrThreshold (WifiTxVector txVector) const
183 {
184  NS_LOG_FUNCTION (this << txVector.GetMode ().GetUniqueName ());
185  for (Thresholds::const_iterator i = m_thresholds.begin (); i != m_thresholds.end (); i++)
186  {
187  NS_LOG_DEBUG ("Checking " << i->second.GetMode ().GetUniqueName () <<
188  " nss " << +i->second.GetNss () <<
189  " GI " << i->second.GetGuardInterval () <<
190  " width " << i->second.GetChannelWidth ());
191  NS_LOG_DEBUG ("against TxVector " << txVector.GetMode ().GetUniqueName () <<
192  " nss " << +txVector.GetNss () <<
193  " GI " << txVector.GetGuardInterval () <<
194  " width " << txVector.GetChannelWidth ());
195  if (txVector.GetMode () == i->second.GetMode ()
196  && txVector.GetNss () == i->second.GetNss ()
197  && txVector.GetChannelWidth () == i->second.GetChannelWidth ())
198  {
199  return i->first;
200  }
201  }
202  NS_ASSERT (false);
203  return 0.0;
204 }

301 WifiTxVector
302 IdealWifiManager::DoGetDataTxVector (WifiRemoteStation *st)
303 {
304  NS_LOG_FUNCTION (this << st);
305  IdealWifiRemoteStation *station = static_cast<IdealWifiRemoteStation*> (st);
306  //We search within the Supported rate set the mode with the
307  //highest data rate for which the SNR threshold is smaller than m_lastSnr
308  //to ensure correct packet delivery.
309  WifiMode maxMode = GetDefaultMode ();
310  WifiTxVector txVector;
311  WifiMode mode;
312  uint64_t bestRate = 0;
313  uint8_t selectedNss = 1;
314  uint16_t guardInterval;
315  uint16_t channelWidth = std::min (GetChannelWidth (station), GetPhy ()->GetChannelWidth ());
316  txVector.SetChannelWidth (channelWidth);
317  if (station->m_lastSnrCached != CACHE_INITIAL_VALUE && station->m_lastSnrObserved == station->m_lastSnrCached)
318  {
319  // SNR has not changed, so skip the search and use the last
320  // mode selected
321  maxMode = station->m_lastMode;
322  selectedNss = station->m_nss;
323  NS_LOG_DEBUG ("Using cached mode = " << maxMode.GetUniqueName () <<
324  " last snr observed " << station->m_lastSnrObserved <<
325  " cached " << station->m_lastSnrCached <<
326  " nss " << +selectedNss);
327  }
328  else
329  {
330  if (GetHtSupported () && GetHtSupported (st))
331  {
332  for (uint8_t i = 0; i < GetNMcsSupported (station); i++)
333  {
334  mode = GetMcsSupported (station, i);
335  txVector.SetMode (mode);
336  if (mode.GetModulationClass () == WIFI_MOD_CLASS_HT)
337  {
338  guardInterval = static_cast<uint16_t> (std::max (GetShortGuardIntervalSupported (station) ? 400 : 800, GetShortGuardIntervalSupported () ? 400 : 800));
339  txVector.SetGuardInterval (guardInterval);
340  // If the node and peer are both VHT capable, only search VHT modes
341  if (GetVhtSupported () && GetVhtSupported (station))
342  {
343  continue;
344  }
345  // If the node and peer are both HE capable, only search HE modes
346  if (GetHeSupported () && GetHeSupported (station))
347  {
348  continue;
349  }
350  // Derive NSS from the MCS index. There is a different mode for each possible NSS value.
351  uint8_t nss = (mode.GetMcsValue () / 8) + 1;
352  txVector.SetNss (nss);
353  if (!txVector.IsValid ()
354  || nss > std::min (GetMaxNumberOfTransmitStreams (), GetNumberOfSupportedStreams (st)))
355  {
356  NS_LOG_DEBUG ("Skipping mode " << mode.GetUniqueName () <<
357  " nss " << +nss <<
358  " width " << txVector.GetChannelWidth ());
359  continue;
360  }
361  double threshold = GetSnrThreshold (txVector);
362  uint64_t dataRate = mode.GetDataRate (txVector.GetChannelWidth (), txVector.GetGuardInterval (), nss);
363  NS_LOG_DEBUG ("Testing mode " << mode.GetUniqueName () <<
364  " data rate " << dataRate <<
365  " threshold " << threshold << " last snr observed " <<
366  station->m_lastSnrObserved << " cached " <<
367  station->m_lastSnrCached);
368  if (dataRate > bestRate && threshold < station->m_lastSnrObserved)
369  {
370  NS_LOG_DEBUG ("Candidate mode = " << mode.GetUniqueName () <<
371  " data rate " << dataRate <<
372  " threshold " << threshold <<
373  " last snr observed " <<
374  station->m_lastSnrObserved);
375  bestRate = dataRate;
376  maxMode = mode;
377  selectedNss = nss;
378  }
379  }
380  else if (mode.GetModulationClass () == WIFI_MOD_CLASS_VHT)
381  {
382  guardInterval = static_cast<uint16_t> (std::max (GetShortGuardIntervalSupported (station) ? 400 : 800, GetShortGuardIntervalSupported () ? 400 : 800));
383  txVector.SetGuardInterval (guardInterval);
384  // If the node and peer are both HE capable, only search HE modes
385  if (GetHeSupported () && GetHeSupported (station))
386  {
387  continue;
388  }
389  // If the node and peer are not both VHT capable, only search HT modes
390  if (!GetVhtSupported () || !GetVhtSupported (station))
391  {
392  continue;
393  }
394  for (uint8_t nss = 1; nss <= std::min (GetMaxNumberOfTransmitStreams (), GetNumberOfSupportedStreams (station)); nss++)
395  {
396  txVector.SetNss (nss);
397  if (!txVector.IsValid ())
398  {
399  NS_LOG_DEBUG ("Skipping mode " << mode.GetUniqueName () <<
400  " nss " << +nss <<
401  " width " << txVector.GetChannelWidth ());
402  continue;
403  }
404  double threshold = GetSnrThreshold (txVector);
405  uint64_t dataRate = mode.GetDataRate (txVector.GetChannelWidth (), txVector.GetGuardInterval (), nss);
406  NS_LOG_DEBUG ("Testing mode = " << mode.GetUniqueName () <<
407  " data rate " << dataRate <<
408  " threshold " << threshold << " last snr observed " <<
409  station->m_lastSnrObserved << " cached " <<
410  station->m_lastSnrCached);
411  if (dataRate > bestRate && threshold < station->m_lastSnrObserved)
412  {
413  NS_LOG_DEBUG ("Candidate mode = " << mode.GetUniqueName () <<
414  " data rate " << dataRate <<
415  " threshold " << threshold <<
416  " last snr observed " <<
417  station->m_lastSnrObserved);
418  bestRate = dataRate;
419  maxMode = mode;
420  selectedNss = nss;
421  }
422  }
423  }
424  else //HE
425  {
426  guardInterval = std::max (GetGuardInterval (station), GetGuardInterval ());
427  txVector.SetGuardInterval (guardInterval);
428  // If the node and peer are not both HE capable, only search (V)HT modes
429  if (!GetHeSupported () || !GetHeSupported (station))
430  {
431  continue;
432  }
433  for (uint8_t nss = 1; nss <= std::min (GetMaxNumberOfTransmitStreams (), GetNumberOfSupportedStreams (station)); nss++)
434  {
435  txVector.SetNss (nss);
436  if (!txVector.IsValid ())
437  {
438  NS_LOG_DEBUG ("Skipping mode " << mode.GetUniqueName () <<
439  " nss " << +nss <<
440  " width " << +txVector.GetChannelWidth ());
441  continue;
442  }
443  double threshold = GetSnrThreshold (txVector);
444  uint64_t dataRate = mode.GetDataRate (txVector.GetChannelWidth (), txVector.GetGuardInterval (), nss);
445  NS_LOG_DEBUG ("Testing mode = " << mode.GetUniqueName () <<
446  " data rate " << dataRate <<
447  " threshold " << threshold << " last snr observed " <<
448  station->m_lastSnrObserved << " cached " <<
449  station->m_lastSnrCached);
450  if (dataRate > bestRate && threshold < station->m_lastSnrObserved)
451  {
452  NS_LOG_DEBUG ("Candidate mode = " << mode.GetUniqueName () <<
453  " data rate " << dataRate <<
454  " threshold " << threshold <<
455  " last snr observed " <<
456  station->m_lastSnrObserved);
457  bestRate = dataRate;
458  maxMode = mode;
459  selectedNss = nss;
460  }
461  }
462  }
463  }
464  }
465  else
466  {
467  // Non-HT selection
468  selectedNss = 1;
469  for (uint8_t i = 0; i < GetNSupported (station); i++)
470  {
471  mode = GetSupported (station, i);
472  txVector.SetMode (mode);
473  txVector.SetNss (selectedNss);
474  txVector.SetChannelWidth (GetChannelWidthForNonHtMode (mode));
475  double threshold = GetSnrThreshold (txVector);
476  uint64_t dataRate = mode.GetDataRate (txVector.GetChannelWidth (), txVector.GetGuardInterval (), txVector.GetNss ());
477  NS_LOG_DEBUG ("mode = " << mode.GetUniqueName () <<
478  " threshold " << threshold <<
479  " last snr observed " <<
480  station->m_lastSnrObserved);
481  if (dataRate > bestRate && threshold < station->m_lastSnrObserved)
482  {
483  NS_LOG_DEBUG ("Candidate mode = " << mode.GetUniqueName () <<
484  " data rate " << dataRate <<
485  " threshold " << threshold <<
486  " last snr observed " <<
487  station->m_lastSnrObserved);
488  bestRate = dataRate;
489  maxMode = mode;
490  }
491  }
492  }
493  NS_LOG_DEBUG ("Updating cached values for station to " << maxMode.GetUniqueName () << " snr " << station->m_lastSnrObserved);
494  station->m_lastSnrCached = station->m_lastSnrObserved;
495  station->m_lastMode = maxMode;
496  station->m_nss = selectedNss;
497  }
498  NS_LOG_DEBUG ("Found maxMode: " << maxMode << " channelWidth: " << channelWidth);
499  if (maxMode.GetModulationClass () == WIFI_MOD_CLASS_HE)
500  {
501  guardInterval = std::max (GetGuardInterval (station), GetGuardInterval ());
502  }
503  else if ((maxMode.GetModulationClass () == WIFI_MOD_CLASS_HT) || (maxMode.GetModulationClass () == WIFI_MOD_CLASS_VHT))
504  {
505  guardInterval = static_cast<uint16_t> (std::max (GetShortGuardIntervalSupported (station) ? 400 : 800, GetShortGuardIntervalSupported () ? 400 : 800));
506  }
507  else
508  {
509  guardInterval = 800;
510  }
511  if (m_currentRate != maxMode.GetDataRate (channelWidth, guardInterval, selectedNss))
512  {
513  NS_LOG_DEBUG ("New datarate: " << maxMode.GetDataRate (channelWidth, guardInterval, selectedNss));
514  m_currentRate = maxMode.GetDataRate (channelWidth, guardInterval, selectedNss);
515  }
516  return WifiTxVector (maxMode, GetDefaultTxPowerLevel (), GetPreambleForTransmission (maxMode.GetModulationClass (), GetShortPreambleEnabled (), UseGreenfieldForDestination (GetAddress (station))), guardInterval, GetNumberOfAntennas (), selectedNss, 0, GetChannelWidthForTransmission (maxMode, channelWidth), GetAggregation (station), false);
517 }

519 WifiTxVector
520 IdealWifiManager::DoGetRtsTxVector (WifiRemoteStation *st)
521 {
522  NS_LOG_FUNCTION (this << st);
523  IdealWifiRemoteStation *station = static_cast<IdealWifiRemoteStation*> (st);
524  //We search within the Basic rate set the mode with the highest
525  //SNR threshold possible which is smaller than m_lastSnr to
526  //ensure correct packet delivery.
527  double maxThreshold = 0.0;
528  WifiTxVector txVector;
529  WifiMode mode;
530  uint8_t nss = 1;
531  WifiMode maxMode = GetDefaultMode ();
532  //RTS is sent in a non-HT frame
533  for (uint8_t i = 0; i < GetNBasicModes (); i++)
534  {
535  mode = GetBasicMode (i);
536  txVector.SetMode (mode);
537  txVector.SetNss (nss);
538  txVector.SetChannelWidth (GetChannelWidthForNonHtMode (mode));
539  double threshold = GetSnrThreshold (txVector);
540  if (threshold > maxThreshold && threshold < station->m_lastSnrObserved)
541  {
542  maxThreshold = threshold;
543  maxMode = mode;
544  }
545  }
546  return WifiTxVector (maxMode, GetDefaultTxPowerLevel (), GetPreambleForTransmission (maxMode.GetModulationClass (), GetShortPreambleEnabled (), UseGreenfieldForDestination (GetAddress (station))), 800, GetNumberOfAntennas (), nss, 0, GetChannelWidthForNonHtMode (maxMode), GetAggregation (station), false);

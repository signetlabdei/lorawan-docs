78 Time
79 WifiMac::GetDefaultCtsAckTimeout (void)
80 {
81  /* Cts_Timeout and Ack_Timeout are specified in the Annex C
82  (Formal description of MAC operation, see details on the
83  Trsp timer setting at page 346)
84  */
85  Time ctsTimeout = GetDefaultSifs ();
86  ctsTimeout += GetDefaultCtsAckDelay ();
87  ctsTimeout += MicroSeconds (GetDefaultMaxPropagationDelay ().GetMicroSeconds () * 2);
88  ctsTimeout += GetDefaultSlot ();
89  return ctsTimeout;
90 }

265 WifiMac::ConfigureStandard (WifiPhyStandard standard)
266 {
267  NS_LOG_FUNCTION (this << standard);
268  switch (standard)
269  {
270  case WIFI_PHY_STANDARD_80211a:
271  Configure80211a ();
272  break;
273  case WIFI_PHY_STANDARD_80211b:
274  Configure80211b ();
275  break;
276  case WIFI_PHY_STANDARD_80211g:
277  Configure80211g ();
278  break;
279  case WIFI_PHY_STANDARD_80211_10MHZ:
280  Configure80211_10Mhz ();
281  break;
282  case WIFI_PHY_STANDARD_80211_5MHZ:
283  Configure80211_5Mhz ();
284  break;
285  case WIFI_PHY_STANDARD_holland:
286  Configure80211a ();
287  break;
288  case WIFI_PHY_STANDARD_80211n_2_4GHZ:
289  Configure80211n_2_4Ghz ();
290  break;
291  case WIFI_PHY_STANDARD_80211n_5GHZ:
292  Configure80211n_5Ghz ();
293  break;
294  case WIFI_PHY_STANDARD_80211ac:
295  Configure80211ac ();
296  break;
297  case WIFI_PHY_STANDARD_80211ax_2_4GHZ:
298  Configure80211ax_2_4Ghz ();
299  break;
300  case WIFI_PHY_STANDARD_80211ax_5GHZ:
301  Configure80211ax_5Ghz ();
302  break;
303  case WIFI_PHY_STANDARD_UNSPECIFIED:
304  default:
305  NS_FATAL_ERROR ("Wifi standard not found");
306  break;
307  }
308  FinishConfigureStandard (standard);
309 }

411 void
412 WifiMac::ConfigureDcf (Ptr<Txop> dcf, uint32_t cwmin, uint32_t cwmax, bool isDsss, AcIndex ac)
413 {
414  NS_LOG_FUNCTION (this << dcf << cwmin << cwmax << isDsss << ac);
415  /* see IEEE 802.11 section 7.3.2.29 */
416  switch (ac)
417  {
418  case AC_VO:
419  dcf->SetMinCw ((cwmin + 1) / 4 - 1);
420  dcf->SetMaxCw ((cwmin + 1) / 2 - 1);
421  dcf->SetAifsn (2);
422  if (isDsss)
423  {
424  dcf->SetTxopLimit (MicroSeconds (3264));
425  }
426  else
427  {
428  dcf->SetTxopLimit (MicroSeconds (1504));
429  }
430  break;
431  case AC_VI:
432  dcf->SetMinCw ((cwmin + 1) / 2 - 1);
433  dcf->SetMaxCw (cwmin);
434  dcf->SetAifsn (2);
435  if (isDsss)
436  {
437  dcf->SetTxopLimit (MicroSeconds (6016));
438  }
439  else
440  {
441  dcf->SetTxopLimit (MicroSeconds (3008));
442  }
443  break;
444  case AC_BE:
445  dcf->SetMinCw (cwmin);
446  dcf->SetMaxCw (cwmax);
447  dcf->SetAifsn (3);
448  dcf->SetTxopLimit (MicroSeconds (0));
449  break;
450  case AC_BK:
451  dcf->SetMinCw (cwmin);
452  dcf->SetMaxCw (cwmax);
453  dcf->SetAifsn (7);
454  dcf->SetTxopLimit (MicroSeconds (0));
455  break;
456  case AC_BE_NQOS:
457  dcf->SetMinCw (cwmin);
458  dcf->SetMaxCw (cwmax);
459  dcf->SetAifsn (2);
460  dcf->SetTxopLimit (MicroSeconds (0));
461  break;
462  case AC_UNDEF:
463  NS_FATAL_ERROR ("I don't know what to do with this");
464  break;
465  }
466 }

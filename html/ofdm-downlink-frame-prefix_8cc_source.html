247 void
248 OfdmDownlinkFramePrefix::Serialize (Buffer::Iterator start) const
249 {
250  Buffer::Iterator i = start;
251  WriteTo (i, m_baseStationId);
252  i.WriteU32 (m_frameNumber);
253  i.WriteU8 (m_configurationChangeCount);
254 
255  for (std::vector<DlFramePrefixIe>::const_iterator iter =
256  m_dlFramePrefixElements.begin (); iter != m_dlFramePrefixElements.end (); iter++)
257  {
258  DlFramePrefixIe dlFramePrefixElement = *iter;
259  i = dlFramePrefixElement.Write (i);
260  }
261 
262  i.WriteU8 (m_hcs);
263 }

265 uint32_t
266 OfdmDownlinkFramePrefix::Deserialize (Buffer::Iterator start)
267 {
268  Buffer::Iterator i = start;
269  ReadFrom (i, m_baseStationId);
270  m_frameNumber = i.ReadU32 ();
271  m_configurationChangeCount = i.ReadU8 ();
272 
273  bool end = false;
274 
275  while (!end)
276  {
277  DlFramePrefixIe dlFramePrefixElement;
278  i = dlFramePrefixElement.Read (i);
279 
280  AddDlFramePrefixElement (dlFramePrefixElement);
281 
282  if (dlFramePrefixElement.GetDiuc () == 14)
283  {
284  end = true;
285  }
286  }
287 
288  m_hcs = i.ReadU8 ();
289 
290  return GetSerializedSize ();
291 }

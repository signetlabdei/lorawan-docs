81 OneShotSender::StartApplication (void)

83  NS_LOG_FUNCTION (this);
84 
85  // Make sure we have a MAC layer
86  if (m_mac == 0)
87  {
88  // Assumes there's only one device
89  Ptr<LoraNetDevice> loraNetDevice = m_node->GetDevice (0)->GetObject<LoraNetDevice> ();
90 
91  m_mac = loraNetDevice->GetMac ();
92  NS_ASSERT (m_mac != 0);
93  }
94 
95  // Schedule the next SendPacket event

136 Ptr<SpectrumSignalParameters>
137 WifiPhyThresholdsTest::MakeWifiSignal (double txPowerWatts)
138 {
139  WifiTxVector txVector = WifiTxVector (WifiPhy::GetOfdmRate6Mbps (), 0, WIFI_PREAMBLE_LONG, 800, 1, 1, 0, 20, false, false);
140 
141  Ptr<Packet> pkt = Create<Packet> (1000);
142  WifiMacHeader hdr;
143 
144  hdr.SetType (WIFI_MAC_QOSDATA);
145  hdr.SetQosTid (0);
146 
147  Ptr<WifiPsdu> psdu = Create<WifiPsdu> (pkt, hdr);
148  Time txDuration = m_phy->CalculateTxDuration (psdu->GetSize (), txVector, m_phy->GetFrequency ());
149 
150  Ptr<WifiPpdu> ppdu = Create<WifiPpdu> (psdu, txVector, txDuration, FREQUENCY);
151 
152  Ptr<SpectrumValue> txPowerSpectrum = WifiSpectrumValueHelper::CreateHeOfdmTxPowerSpectralDensity (FREQUENCY, CHANNEL_WIDTH, txPowerWatts, CHANNEL_WIDTH);
153  Ptr<WifiSpectrumSignalParameters> txParams = Create<WifiSpectrumSignalParameters> ();
154  txParams->psd = txPowerSpectrum;
155  txParams->txPhy = 0;
156  txParams->duration = txDuration;
157  txParams->ppdu = ppdu;
158  return txParams;
159 }

206 void
207 WifiPhyThresholdsTest::PhyStateChanged (Time start, Time duration, WifiPhyState newState)
208 {
209  NS_LOG_FUNCTION (this << start << duration << newState);
210  m_stateChanged++;
211  if (newState == WifiPhyState::IDLE)
212  {
213  m_idleStateCount++;
214  }
215  else if (newState == WifiPhyState::RX)
216  {
217  m_rxStateCount++;
218  }
219  else if (newState == WifiPhyState::CCA_BUSY)
220  {
221  m_ccabusyStateCount++;
222  }
223 }

225 void
226 WifiPhyThresholdsTest::DoSetup (void)
227 {
228  m_phy = CreateObject<SpectrumWifiPhy> ();
229  m_phy->ConfigureStandard (WIFI_PHY_STANDARD_80211ax_5GHZ);
230  Ptr<ErrorRateModel> error = CreateObject<NistErrorRateModel> ();
231  m_phy->SetErrorRateModel (error);
232  m_phy->SetChannelNumber (CHANNEL_NUMBER);
233  m_phy->SetFrequency (FREQUENCY);
234  m_phy->SetReceiveOkCallback (MakeCallback (&WifiPhyThresholdsTest::RxSuccess, this));
235  m_phy->SetReceiveErrorCallback (MakeCallback (&WifiPhyThresholdsTest::RxFailure, this));
236  m_phy->TraceConnectWithoutContext ("PhyRxDrop", MakeCallback (&WifiPhyThresholdsTest::RxDropped, this));
237  m_phy->GetState ()->TraceConnectWithoutContext ("State", MakeCallback (&WifiPhyThresholdsTest::PhyStateChanged, this));
238 }

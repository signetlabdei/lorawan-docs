63 NetworkController::OnNewPacket (Ptr<Packet const> packet)
64 {
65  NS_LOG_FUNCTION (this << packet);
66 
67  // NOTE As a future optimization, we can allow components to register their
68  // callbacks and only be called in case a certain MAC command is contained.
69  // For now, we call all components.
70 
71  // Inform each component about the new packet
72  for (auto it = m_components.begin (); it != m_components.end (); ++it)
73  {
74  (*it)->OnReceivedPacket (packet,
75  m_status->GetEndDeviceStatus (packet),
76  m_status);
77  }
78 }

104 uint32_t LtePdcpHeader::Deserialize (Buffer::Iterator start)
105 {
106  Buffer::Iterator i = start;
107  uint8_t byte_1;
108  uint8_t byte_2;
109 
110  byte_1 = i.ReadU8 ();
111  byte_2 = i.ReadU8 ();
112  m_dcBit = (byte_1 & 0x80) > 7;
113  // For now, we just support DATA PDUs
114  NS_ASSERT (m_dcBit == DATA_PDU);
115  m_sequenceNumber = ((byte_1 & 0x0F) << 8) | byte_2;
116 
117  return GetSerializedSize ();

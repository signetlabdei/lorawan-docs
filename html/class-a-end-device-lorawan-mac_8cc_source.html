75 void
76 ClassAEndDeviceLorawanMac::SendToPhy (Ptr<Packet> packetToSend)
77 {
79  // Add headers, prepare TX parameters and send the packet
81 
82  NS_LOG_DEBUG ("PacketToSend: " << packetToSend);
83 
84  // Data Rate Adaptation as in LoRaWAN specification, V1.0.2 (2016)
85  if (m_enableDRAdapt && (m_dataRate > 0)
86  && (m_retxParams.retxLeft < m_maxNumbTx)
87  && (m_retxParams.retxLeft % 2 == 0) )
88  {
89  m_txPower = 14; // Reset transmission power
90  m_dataRate = m_dataRate - 1;
91  }
92 
93  // Craft LoraTxParameters object
94  LoraTxParameters params;
95  params.sf = GetSfFromDataRate (m_dataRate);
96  params.headerDisabled = m_headerDisabled;
97  params.codingRate = m_codingRate;
98  params.bandwidthHz = GetBandwidthFromDataRate (m_dataRate);
99  params.nPreamble = m_nPreambleSymbols;
100  params.crcEnabled = 1;
101  params.lowDataRateOptimizationEnabled = 0;
102 
103  // Wake up PHY layer and directly send the packet
104 
105  Ptr<LogicalLoraChannel> txChannel = GetChannelForTx ();
106 
107  NS_LOG_DEBUG ("PacketToSend: " << packetToSend);
108  m_phy->Send (packetToSend, params, txChannel->GetFrequency (), m_txPower);
109 
111  // Register packet transmission for duty cycle
113 
114  // Compute packet duration
115  Time duration = m_phy->GetOnAirTime (packetToSend, params);
116 
117  // Register the sent packet into the DutyCycleHelper
118  m_channelHelper.AddEvent (duration, txChannel);
119 
121  // Prepare for the downlink //
123 
124  // Switch the PHY to the channel so that it will listen here for downlink
125  m_phy->GetObject<EndDeviceLoraPhy> ()->SetFrequency (txChannel->GetFrequency ());
126 
127  // Instruct the PHY on the right Spreading Factor to listen for during the window
128  // create a SetReplyDataRate function?
129  uint8_t replyDataRate = GetFirstReceiveWindowDataRate ();
130  NS_LOG_DEBUG ("m_dataRate: " << unsigned (m_dataRate) <<
131  ", m_rx1DrOffset: " << unsigned (m_rx1DrOffset) <<
132  ", replyDataRate: " << unsigned (replyDataRate) << ".");
133 
134  m_phy->GetObject<EndDeviceLoraPhy> ()->SetSpreadingFactor
135  (GetSfFromDataRate (replyDataRate));
136 
137 }

142 void
143 ClassAEndDeviceLorawanMac::Receive (Ptr<Packet const> packet)
144 {
145  NS_LOG_FUNCTION (this << packet);
146 
147  // Work on a copy of the packet
148  Ptr<Packet> packetCopy = packet->Copy ();
149 
150  // Remove the Mac Header to get some information
151  LorawanMacHeader mHdr;
152  packetCopy->RemoveHeader (mHdr);
153 
154  NS_LOG_DEBUG ("Mac Header: " << mHdr);
155 
156  // Only keep analyzing the packet if it's downlink
157  if (!mHdr.IsUplink ())
158  {
159  NS_LOG_INFO ("Found a downlink packet.");
160 
161  // Remove the Frame Header
162  LoraFrameHeader fHdr;
163  fHdr.SetAsDownlink ();
164  packetCopy->RemoveHeader (fHdr);
165 
166  NS_LOG_DEBUG ("Frame Header: " << fHdr);
167 
168  // Determine whether this packet is for us
169  bool messageForUs = (m_address == fHdr.GetAddress ());
170 
171  if (messageForUs)
172  {
173  NS_LOG_INFO ("The message is for us!");
174 
175  // If it exists, cancel the second receive window event
176  // THIS WILL BE GetReceiveWindow()
177  Simulator::Cancel (m_secondReceiveWindow);
178 
179 
180  // Parse the MAC commands
181  ParseCommands (fHdr);
182 
183  // TODO Pass the packet up to the NetDevice
184 
185 
186  // Call the trace source
187  m_receivedPacket (packet);
188  }
189  else
190  {
191  NS_LOG_DEBUG ("The message is intended for another recipient.");
192 
193  // In this case, we are either receiving in the first receive window
194  // and finishing reception inside the second one, or receiving a
195  // packet in the second receive window and finding out, after the
196  // fact, that the packet is not for us. In either case, if we no
197  // longer have any retransmissions left, we declare failure.
198  if (m_retxParams.waitingAck && m_secondReceiveWindow.IsExpired ())
199  {
200  if (m_retxParams.retxLeft == 0)
201  {
202  uint8_t txs = m_maxNumbTx - (m_retxParams.retxLeft);
203  m_requiredTxCallback (txs, false, m_retxParams.firstAttempt, m_retxParams.packet);
204  NS_LOG_DEBUG ("Failure: no more retransmissions left. Used " << unsigned(txs) << " transmissions.");
205 
206  // Reset retransmission parameters
207  resetRetransmissionParameters ();
208  }
209  else // Reschedule
210  {
211  this->Send (m_retxParams.packet);
212  NS_LOG_INFO ("We have " << unsigned(m_retxParams.retxLeft) << " retransmissions left: rescheduling transmission.");
213  }
214  }
215  }
216  }
217  else if (m_retxParams.waitingAck && m_secondReceiveWindow.IsExpired ())
218  {
219  NS_LOG_INFO ("The packet we are receiving is in uplink.");
220  if (m_retxParams.retxLeft > 0)
221  {
222  this->Send (m_retxParams.packet);
223  NS_LOG_INFO ("We have " << unsigned(m_retxParams.retxLeft) << " retransmissions left: rescheduling transmission.");
224  }
225  else
226  {
227  uint8_t txs = m_maxNumbTx - (m_retxParams.retxLeft);
228  m_requiredTxCallback (txs, false, m_retxParams.firstAttempt, m_retxParams.packet);
229  NS_LOG_DEBUG ("Failure: no more retransmissions left. Used " << unsigned(txs) << " transmissions.");
230 
231  // Reset retransmission parameters
232  resetRetransmissionParameters ();
233  }
234  }
235 
236  m_phy->GetObject<EndDeviceLoraPhy> ()->SwitchToSleep ();
237 }

294 {
295  NS_LOG_FUNCTION_NOARGS ();
296 
297  // Set Phy in Standby mode
298  m_phy->GetObject<EndDeviceLoraPhy> ()->SwitchToStandby ();
299

301  double tSym = pow (2, GetSfFromDataRate (GetFirstReceiveWindowDataRate ())) / GetBandwidthFromDataRate ( GetFirstReceiveWindowDataRate ());

311 void
312 ClassAEndDeviceLorawanMac::CloseFirstReceiveWindow (void)
313 {
314  NS_LOG_FUNCTION_NOARGS ();
315 
316  Ptr<EndDeviceLoraPhy> phy = m_phy->GetObject<EndDeviceLoraPhy> ();
317 
318  // Check the Phy layer's state:
319  // - RX -> We are receiving a preamble.
320  // - STANDBY -> Nothing was received.
321  // - SLEEP -> We have received a packet.
322  // We should never be in TX or SLEEP mode at this point
323  switch (phy->GetState ())
324  {
325  case EndDeviceLoraPhy::TX:
326  NS_ABORT_MSG ("PHY was in TX mode when attempting to " <<
327  "close a receive window.");
328  break;
329  case EndDeviceLoraPhy::RX:
330  // PHY is receiving: let it finish. The Receive method will switch it back to SLEEP.
331  break;
332  case EndDeviceLoraPhy::SLEEP:
333  // PHY has received, and the MAC's Receive already put the device to sleep
334  break;
335  case EndDeviceLoraPhy::STANDBY:
336  // Turn PHY layer to SLEEP
337  phy->SwitchToSleep ();
338  break;
339  }

343 ClassAEndDeviceLorawanMac::OpenSecondReceiveWindow (void)
344 {
345  NS_LOG_FUNCTION_NOARGS ();
346 
347  // Check for receiver status: if it's locked on a packet, don't open this
348  // window at all.
349  if (m_phy->GetObject<EndDeviceLoraPhy> ()->GetState () == EndDeviceLoraPhy::RX)
350  {
351  NS_LOG_INFO ("Won't open second receive window since we are in RX mode.");
352 
353  return;
354  }
355 
356  // Set Phy in Standby mode
357  m_phy->GetObject<EndDeviceLoraPhy> ()->SwitchToStandby ();
358 
359  // Switch to appropriate channel and data rate
360  NS_LOG_INFO ("Using parameters: " << m_secondReceiveWindowFrequency << "Hz, DR"
361  << unsigned(m_secondReceiveWindowDataRate));
362 
363  m_phy->GetObject<EndDeviceLoraPhy> ()->SetFrequency
364  (m_secondReceiveWindowFrequency);
365  m_phy->GetObject<EndDeviceLoraPhy> ()->SetSpreadingFactor (GetSfFromDataRate
366  (m_secondReceiveWindowDataRate));
367 
368  //Calculate the duration of a single symbol for the second receive window DR
369  double tSym = pow (2, GetSfFromDataRate (GetSecondReceiveWindowDataRate ())) / GetBandwidthFromDataRate ( GetSecondReceiveWindowDataRate ());
370 
371  // Schedule return to sleep after "at least the time required by the end
372  // device's radio transceiver to effectively detect a downlink preamble"
373  // (LoraWAN specification)

379 void
380 ClassAEndDeviceLorawanMac::CloseSecondReceiveWindow (void)
381 {
382  NS_LOG_FUNCTION_NOARGS ();
383 
384  Ptr<EndDeviceLoraPhy> phy = m_phy->GetObject<EndDeviceLoraPhy> ();
385 
386  // NS_ASSERT (phy->m_state != EndDeviceLoraPhy::TX &&
387  // phy->m_state != EndDeviceLoraPhy::SLEEP);
388 
389  // Check the Phy layer's state:
390  // - RX -> We have received a preamble.
391  // - STANDBY -> Nothing was detected.
392  switch (phy->GetState ())
393  {
394  case EndDeviceLoraPhy::TX:
395  break;
396  case EndDeviceLoraPhy::SLEEP:
397  break;
398  case EndDeviceLoraPhy::RX:
399  // PHY is receiving: let it finish
400  NS_LOG_DEBUG ("PHY is receiving: Receive will handle the result.");
401  return;
402  case EndDeviceLoraPhy::STANDBY:
403  // Turn PHY layer to sleep
404  phy->SwitchToSleep ();
405  break;
406  }
407 
408  if (m_retxParams.waitingAck)
409  {
410  NS_LOG_DEBUG ("No reception initiated by PHY: rescheduling transmission.");
411  if (m_retxParams.retxLeft > 0 )
412  {
413  NS_LOG_INFO ("We have " << unsigned(m_retxParams.retxLeft) << " retransmissions left: rescheduling transmission.");
414  this->Send (m_retxParams.packet);
415  }
416 
417  else if (m_retxParams.retxLeft == 0 && m_phy->GetObject<EndDeviceLoraPhy> ()->GetState () != EndDeviceLoraPhy::RX)
418  {
419  uint8_t txs = m_maxNumbTx - (m_retxParams.retxLeft);
420  m_requiredTxCallback (txs, false, m_retxParams.firstAttempt, m_retxParams.packet);
421  NS_LOG_DEBUG ("Failure: no more retransmissions left. Used " << unsigned(txs) << " transmissions.");
422 
423  // Reset retransmission parameters
424  resetRetransmissionParameters ();
425  }
426 
427  else
428  {
429  NS_ABORT_MSG ("The number of retransmissions left is negative ! ");
430  }
431  }
432  else
433  {
434  uint8_t txs = m_maxNumbTx - (m_retxParams.retxLeft );
435  m_requiredTxCallback (txs, true, m_retxParams.firstAttempt, m_retxParams.packet);
436  NS_LOG_INFO ("We have " << unsigned(m_retxParams.retxLeft) <<
437  " transmissions left. We were not transmitting confirmed messages.");
438 
439  // Reset retransmission parameters
440  resetRetransmissionParameters ();
441  }

448 Time
449 ClassAEndDeviceLorawanMac::GetNextClassTransmissionDelay (Time waitingTime)
450 {
451  NS_LOG_FUNCTION_NOARGS ();
452 
453  // This is a new packet from APP; it can not be sent until the end of the
454  // second receive window (if the second recieve window has not closed yet)
455  if (!m_retxParams.waitingAck)
456  {
457  if (!m_closeFirstWindow.IsExpired () ||
458  !m_closeSecondWindow.IsExpired () ||
459  !m_secondReceiveWindow.IsExpired () )
460  {
461  NS_LOG_WARN ("Attempting to send when there are receive windows:" <<
462  " Transmission postponed.");
463  // Compute the duration of a single symbol for the second receive window DR
464  double tSym = pow (2, GetSfFromDataRate (GetSecondReceiveWindowDataRate ())) / GetBandwidthFromDataRate (GetSecondReceiveWindowDataRate ());
465  // Compute the closing time of the second receive window
466  Time endSecondRxWindow = Time(m_secondReceiveWindow.GetTs()) + Seconds (m_receiveWindowDurationInSymbols*tSym);
467 
468  NS_LOG_DEBUG("Duration until endSecondRxWindow for new transmission:" << (endSecondRxWindow - Simulator::Now()).GetSeconds());
469  waitingTime = std::max (waitingTime, endSecondRxWindow - Simulator::Now());
470  }
471  }
472  // This is a retransmitted packet, it can not be sent until the end of
473  // ACK_TIMEOUT (this timer starts when the second receive window was open)
474  else
475  {
476  double ack_timeout = m_uniformRV->GetValue (1,3);
477  // Compute the duration until ACK_TIMEOUT (It may be a negative number, but it doesn't matter.)
478  Time retransmitWaitingTime = Time(m_secondReceiveWindow.GetTs()) - Simulator::Now() + Seconds (ack_timeout);
479 
480  NS_LOG_DEBUG("ack_timeout:" << ack_timeout <<
481  " retransmitWaitingTime:" << retransmitWaitingTime.GetSeconds());
482  waitingTime = std::max (waitingTime, retransmitWaitingTime);
483  }
484 
485  return waitingTime;
486 }

541  }

546  {
547  dataRateOk = false;
548  }
549

551  m_secondReceiveWindowDataRate = rx2DataRate;
552  m_rx1DrOffset = rx1DrOffset;
553  m_secondReceiveWindowFrequency = frequency;

107  // Create the OnOff application to send UDP datagrams of size
108  // 210 bytes at a rate of 448 Kb/s
109  uint16_t port = 9; // Discard port (RFC 863)

111  Address (InetSocketAddress (ifInAddrC.GetLocal (), port)));

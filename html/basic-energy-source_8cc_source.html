151 void
152 BasicEnergySource::UpdateEnergySource (void)
153 {
154  NS_LOG_FUNCTION (this);
155  NS_LOG_DEBUG ("BasicEnergySource:Updating remaining energy.");
156 
157  double remainingEnergy = m_remainingEnergyJ;
158  CalculateRemainingEnergy ();
159 
160  m_lastUpdateTime = Simulator::Now ();
161 
162  if (!m_depleted && m_remainingEnergyJ <= m_lowBatteryTh * m_initialEnergyJ)
163  {
164  m_depleted = true;
165  HandleEnergyDrainedEvent ();
166  }
167  else if (m_depleted && m_remainingEnergyJ > m_highBatteryTh * m_initialEnergyJ)
168  {
169  m_depleted = false;
170  HandleEnergyRechargedEvent ();
171  }
172  else if (m_remainingEnergyJ != remainingEnergy)
173  {
174  NotifyEnergyChanged ();
175  }
176 
177  if (m_energyUpdateEvent.IsExpired ())
178  {
179  m_energyUpdateEvent = Simulator::Schedule (m_energyUpdateInterval,
180  &BasicEnergySource::UpdateEnergySource,
181  this);
182  }
183 }

219 void
220 BasicEnergySource::CalculateRemainingEnergy (void)
221 {
222  NS_LOG_FUNCTION (this);
223  double totalCurrentA = CalculateTotalCurrent ();
224  Time duration = Simulator::Now () - m_lastUpdateTime;
225  NS_ASSERT (duration.IsPositive ());
226  // energy = current * voltage * time
227  double energyToDecreaseJ = (totalCurrentA * m_supplyVoltageV * duration.GetNanoSeconds ()) / 1e9;
228  NS_ASSERT (m_remainingEnergyJ >= energyToDecreaseJ);
229  m_remainingEnergyJ -= energyToDecreaseJ;
230  NS_LOG_DEBUG ("BasicEnergySource:Remaining energy = " << m_remainingEnergyJ);
231 }

223 void
224 LogicalLoraChannelHelper::AddEvent (Time duration,
225  Ptr<LogicalLoraChannel> channel)
226 {
227  NS_LOG_FUNCTION (this << duration << channel);
228 
229  Ptr<SubBand> subBand = GetSubBandFromChannel (channel);
230 
231  double dutyCycle = subBand->GetDutyCycle ();
232  double timeOnAir = duration.GetSeconds ();
233 
234  // Computation of necessary waiting time on this sub-band
235  subBand->SetNextTransmissionTime (Simulator::Now () + Seconds
236  (timeOnAir / dutyCycle - timeOnAir));
237 
238  // Computation of necessary aggregate waiting time
239  m_nextAggregatedTransmissionTime = Simulator::Now () + Seconds
240  (timeOnAir / m_aggregatedDutyCycle - timeOnAir);
241 
242  NS_LOG_DEBUG ("Time on air: " << timeOnAir);
243  NS_LOG_DEBUG ("m_aggregatedDutyCycle: " << m_aggregatedDutyCycle);
244  NS_LOG_DEBUG ("Current time: " << Simulator::Now ().GetSeconds ());
245  NS_LOG_DEBUG ("Next transmission on this sub-band allowed at time: " <<
246  (subBand->GetNextTransmissionTime ()).GetSeconds ());
247  NS_LOG_DEBUG ("Next aggregated transmission allowed at time " <<
248  m_nextAggregatedTransmissionTime.GetSeconds ());
249 }

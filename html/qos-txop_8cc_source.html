252 Ptr<const WifiMacQueueItem>
253 QosTxop::PeekNextFrame (uint8_t tid, Mac48Address recipient)
254 {
255  NS_LOG_FUNCTION (this);
256  WifiMacQueue::ConstIterator it = WifiMacQueue::EMPTY;
257 
258  // lambda to peek the next frame
259  auto peek = [this, &tid, &recipient, &it] (Ptr<WifiMacQueue> queue)
260  {
261  if (tid == 8 && recipient.IsBroadcast ()) // undefined TID and recipient
262  {
263  return queue->PeekFirstAvailable (m_qosBlockedDestinations, it);
264  }
265  return queue->PeekByTidAndAddress (tid, recipient, it);
266  };
267 
268  // check if there is a packet in the BlockAckManager retransmit queue
269  it = peek (m_baManager->GetRetransmitQueue ());
270  // remove old packets
271  while (it != m_baManager->GetRetransmitQueue ()->end () && IsQosOldPacket (*it))
272  {
273  NS_LOG_DEBUG ("removing an old packet from BlockAckManager retransmit queue: " << **it);
274  it = m_baManager->GetRetransmitQueue ()->Remove (it);
275  it = peek (m_baManager->GetRetransmitQueue ());
276  }
277  if (it != m_baManager->GetRetransmitQueue ()->end ())
278  {
279  NS_LOG_DEBUG ("packet peeked from BlockAckManager retransmit queue: " << **it);
280  return *it;
281  }
282 
283  // otherwise, check if there is a packet in the EDCA queue
284  it = WifiMacQueue::EMPTY;
285  it = peek (m_queue);
286  if (it != m_queue->end ())
287  {
288  // peek the next sequence number and check if it is within the transmit window
289  // in case of QoS data frame
290  uint16_t sequence = m_txMiddle->PeekNextSequenceNumberFor (&(*it)->GetHeader ());
291  if ((*it)->GetHeader ().IsQosData ())
292  {
293  Mac48Address recipient = (*it)->GetHeader ().GetAddr1 ();
294  uint8_t tid = (*it)->GetHeader ().GetQosTid ();
295 
296  if (GetBaAgreementEstablished (recipient, tid)
297  && !IsInWindow (sequence, GetBaStartingSequence (recipient, tid), GetBaBufferSize (recipient, tid)))
298  {
299  NS_LOG_DEBUG ("packet beyond the end of the current transmit window");
300  return 0;
301  }
302  }
303 
304  WifiMacHeader hdr = (*it)->GetHeader ();
305  hdr.SetSequenceNumber (sequence);
306  hdr.SetFragmentNumber (0);
307  hdr.SetNoMoreFragments ();
308  hdr.SetNoRetry ();
309  Ptr<const WifiMacQueueItem> item = Create<const WifiMacQueueItem> ((*it)->GetPacket (), hdr, (*it)->GetTimeStamp ());
310  NS_LOG_DEBUG ("packet peeked from EDCA queue: " << *item);
311  return item;
312  }
313 
314  return 0;
315 }

317 Ptr<WifiMacQueueItem>
318 QosTxop::DequeuePeekedFrame (Ptr<const WifiMacQueueItem> peekedItem, WifiTxVector txVector,
319  bool aggregate, uint32_t ampduSize, Time ppduDurationLimit)
320 {
321  NS_LOG_FUNCTION (this << peekedItem << txVector << ampduSize << ppduDurationLimit);
322  NS_ASSERT (peekedItem != 0);
323 
324  // do not dequeue the frame if it is a QoS data frame that does not meet the
325  // max A-MPDU size limit (if applicable) or the duration limit (if applicable)
326  if (peekedItem->GetHeader ().IsQosData () &&
327  !m_low->IsWithinSizeAndTimeLimits (peekedItem, txVector, ampduSize, ppduDurationLimit))
328  {
329  return 0;
330  }
331 
332  Mac48Address recipient = peekedItem->GetHeader ().GetAddr1 ();
333  Ptr<WifiMacQueueItem> item;
334  Ptr<const WifiMacQueueItem> testItem;
335  WifiMacQueue::ConstIterator testIt;
336 
337  // the packet can only have been peeked from the block ack manager retransmit
338  // queue if:
339  // - the peeked packet is a QoS Data frame AND
340  // - the peeked packet is not a broadcast frame AND
341  // - an agreement has been established
342  if (peekedItem->GetHeader ().IsQosData () && !recipient.IsBroadcast ()
343  && GetBaAgreementEstablished (recipient, peekedItem->GetHeader ().GetQosTid ()))
344  {
345  uint8_t tid = peekedItem->GetHeader ().GetQosTid ();
346  testIt = m_baManager->GetRetransmitQueue ()->PeekByTidAndAddress (tid, recipient);
347 
348  if (testIt != m_baManager->GetRetransmitQueue ()->end ())
349  {
350  testItem = *testIt;
351  // if not null, the test packet must equal the peeked packet
352  NS_ASSERT (testItem->GetPacket () == peekedItem->GetPacket ());
353  // we should not be asked to dequeue an old packet
354  NS_ASSERT (!QosUtilsIsOldPacket (GetBaStartingSequence (recipient, tid),
355  testItem->GetHeader ().GetSequenceNumber ()));
356  item = m_baManager->GetRetransmitQueue ()->Dequeue (testIt);
357  NS_LOG_DEBUG ("dequeued from BA manager queue: " << *item);
358  return item;
359  }
360  }
361 
362  // the packet has been peeked from the EDCA queue.
363  uint16_t sequence = m_txMiddle->GetNextSequenceNumberFor (&peekedItem->GetHeader ());
364 
365  // If it is a QoS Data frame and it is not a broadcast frame, attempt A-MSDU
366  // aggregation if aggregate is true
367  if (peekedItem->GetHeader ().IsQosData ())
368  {
369  uint8_t tid = peekedItem->GetHeader ().GetQosTid ();
370  testIt = m_queue->PeekByTidAndAddress (tid, recipient);
371 
372  NS_ASSERT (testIt != m_queue->end () && (*testIt)->GetPacket () == peekedItem->GetPacket ());
373 
374  // check if the peeked packet is within the transmit window
375  if (GetBaAgreementEstablished (recipient, tid)
376  && !IsInWindow (sequence, GetBaStartingSequence (recipient, tid), GetBaBufferSize (recipient, tid)))
377  {
378  NS_LOG_DEBUG ("packet beyond the end of the current transmit window");
379  return 0;
380  }
381 
382  // try A-MSDU aggregation
383  if (m_low->GetMsduAggregator () != 0 && !recipient.IsBroadcast () && aggregate)
384  {
385  item = m_low->GetMsduAggregator ()->GetNextAmsdu (recipient, tid, txVector, ampduSize, ppduDurationLimit);
386  }
387 
388  if (item != 0)
389  {
390  NS_LOG_DEBUG ("tx unicast A-MSDU");
391  }
392  else // aggregation was not attempted or failed
393  {
394  item = m_queue->Dequeue (testIt);
395  }
396  }
397  else
398  {
399  // the peeked packet is a non-QoS Data frame (e.g., a DELBA Request), hence
400  // it was not peeked by TID, hence it must be the head of the queue
401  item = m_queue->DequeueFirstAvailable (m_qosBlockedDestinations);
402  NS_ASSERT (item != 0 && item->GetPacket () == peekedItem->GetPacket ());
403  }
404 
405  // Assign a sequence number to the MSDU or A-MSDU dequeued from the EDCA queue
406  NS_ASSERT (item != 0);
407  item->GetHeader ().SetSequenceNumber (sequence);
408  item->GetHeader ().SetFragmentNumber (0);
409  item->GetHeader ().SetNoMoreFragments ();
410  item->GetHeader ().SetNoRetry ();
411  NS_LOG_DEBUG ("dequeued from EDCA queue: " << *item);
412 
413  return item;
414 }

416 MacLowTransmissionParameters
417 QosTxop::GetTransmissionParameters (Ptr<const WifiMacQueueItem> frame) const
418 {
419  NS_LOG_FUNCTION (this << *frame);
420 
421  MacLowTransmissionParameters params;
422  Mac48Address recipient = frame->GetHeader ().GetAddr1 ();
423 
424  params.DisableNextData ();
425 
426  // broadcast frames
427  if (recipient.IsBroadcast ())
428  {
429  params.DisableRts ();
430  params.DisableAck ();
431  return params;
432  }
433  if (frame->GetHeader ().IsMgt ())
434  {
435  params.DisableRts ();
436  params.EnableAck ();
437  return params;
438  }
439 
440  // Enable/disable RTS
441  if (!frame->GetHeader ().IsBlockAckReq ()
442  && m_stationManager->NeedRts (&frame->GetHeader (), frame->GetPacket ())
443  && !m_low->IsCfPeriod ())
444  {
445  params.EnableRts ();
446  }
447  else
448  {
449  params.DisableRts ();
450  }
451 
452  // Select ack technique.
453  if (frame->GetHeader ().IsQosData ())
454  {
455  // Assume normal Ack by default
456  params.EnableAck ();
457  }
458  else if (frame->GetHeader ().IsBlockAckReq ())
459  {
460  // assume a BlockAck variant. Later, if this frame is not aggregated,
461  // the acknowledgment type will be switched to normal Ack
462  if (m_blockAckType == BASIC_BLOCK_ACK)
463  {
464  params.EnableBlockAck (BlockAckType::BASIC_BLOCK_ACK);
465  }
466  else if (m_blockAckType == COMPRESSED_BLOCK_ACK)
467  {
468  CtrlBAckRequestHeader baReqHdr;
469  frame->GetPacket ()->PeekHeader (baReqHdr);
470  uint8_t tid = baReqHdr.GetTidInfo ();
471 
472  if (GetBaBufferSize (recipient, tid) > 64)
473  {
474  params.EnableBlockAck (BlockAckType::EXTENDED_COMPRESSED_BLOCK_ACK);
475  }
476  else
477  {
478  params.EnableBlockAck (BlockAckType::COMPRESSED_BLOCK_ACK);
479  }
480  }
481  else if (m_blockAckType == MULTI_TID_BLOCK_ACK)
482  {
483  NS_FATAL_ERROR ("Multi-tid block ack is not supported");
484  }
485  }
486 
487  return params;
488 }

499 void
500 QosTxop::NotifyAccessGranted (void)
501 {
502  NS_LOG_FUNCTION (this);
503  NS_ASSERT (m_accessRequested);
504  m_accessRequested = false;
505  m_isAccessRequestedForRts = false;
506  m_startTxop = Simulator::Now ();
507  // discard the current packet if it is a QoS Data frame with expired lifetime
508  if (m_currentPacket != 0 && m_currentHdr.IsQosData ()
509  && (m_currentPacketTimestamp + m_queue->GetMaxDelay () < Simulator::Now ()))
510  {
511  NS_LOG_DEBUG ("the lifetime of current packet expired");
512  m_currentPacket = 0;
513  }
514  // If the current packet is a QoS Data frame, then there must be no block ack agreement
515  // established with the receiver for the TID of the packet. Indeed, retransmission
516  // of MPDUs sent under a block ack agreement is handled through the retransmit queue.
517  NS_ASSERT (m_currentPacket == 0 || !m_currentHdr.IsQosData ()
518  || !GetBaAgreementEstablished (m_currentHdr.GetAddr1 (), m_currentHdr.GetQosTid ()));
519 
520  if (m_currentPacket == 0)
521  {
522  Ptr<const WifiMacQueueItem> peekedItem = m_baManager->GetBar ();
523  if (peekedItem != 0)
524  {
525  m_currentHdr = peekedItem->GetHeader ();
526  m_currentPacket = peekedItem->GetPacket ();
527  m_currentPacketTimestamp = Simulator::Now ();
528  }
529  else
530  {
531  peekedItem = PeekNextFrame ();
532  if (peekedItem == 0)
533  {
534  NS_LOG_DEBUG ("no packets available for transmission");
535  return;
536  }
537  // check if a block ack agreement needs to be established
538  m_currentHdr = peekedItem->GetHeader ();
539  m_currentPacket = peekedItem->GetPacket ();
540  if (m_currentHdr.IsQosData () && !m_currentHdr.GetAddr1 ().IsBroadcast ()
541  && m_stationManager->GetQosSupported (m_currentHdr.GetAddr1 ())
542  && (!m_baManager->ExistsAgreement (m_currentHdr.GetAddr1 (), m_currentHdr.GetQosTid ())
543  || m_baManager->ExistsAgreementInState (m_currentHdr.GetAddr1 (), m_currentHdr.GetQosTid (), OriginatorBlockAckAgreement::RESET))
544  && SetupBlockAckIfNeeded ())
545  {
546  return;
547  }
548 
549  m_stationManager->UpdateFragmentationThreshold ();
550  Ptr<WifiMacQueueItem> item;
551  // non-broadcast QoS Data frames may be sent in MU PPDUs. Given that at this stage
552  // we do not know the bandwidth it would be given nor the selected acknowledgment
553  // sequence, we cannot determine the constraints on size and duration limit. Hence,
554  // we only peek the non-broadcast QoS Data frame. MacLow will be in charge of
555  // computing the correct limits and dequeue the frame.
556  if (peekedItem->GetHeader ().IsQosData () && !peekedItem->GetHeader ().GetAddr1 ().IsBroadcast ()
557  && !NeedFragmentation ())
558  {
559  item = Copy (peekedItem);
560  }
561  else
562  {
563  // compute the limit on the PPDU duration due to the TXOP duration, if any
564  Time ppduDurationLimit = Time::Min ();
565  if (peekedItem->GetHeader ().IsQosData () && GetTxopLimit ().IsStrictlyPositive ())
566  {
567  MacLowTransmissionParameters params = GetTransmissionParameters (peekedItem);
568  ppduDurationLimit = GetTxopRemaining () - m_low->CalculateOverheadTxTime (peekedItem, params);
569  }
570 
571  // dequeue the peeked item if it fits within the TXOP duration, if any
572  item = DequeuePeekedFrame (peekedItem, m_low->GetDataTxVector (peekedItem),
573  !NeedFragmentation (), 0, ppduDurationLimit);
574  }
575 
576  if (item == 0)
577  {
578  NS_LOG_DEBUG ("Not enough time in the current TXOP");
579  return;
580  }
581  m_currentPacket = item->GetPacket ();
582  m_currentHdr = item->GetHeader ();
583  m_currentPacketTimestamp = item->GetTimeStamp ();
584 
585  m_fragmentNumber = 0;
586  }
587  NS_ASSERT (m_currentPacket != 0);
588  }
589  Ptr<WifiMacQueueItem> mpdu = Create <WifiMacQueueItem> (m_currentPacket, m_currentHdr,
590  m_currentPacketTimestamp);
591  m_currentParams = GetTransmissionParameters (mpdu);
592 
593  if (m_currentHdr.GetAddr1 ().IsGroup ())
594  {
595  NS_LOG_DEBUG ("tx broadcast");
596  m_low->StartTransmission (mpdu, m_currentParams, this);
597  }
598  //With COMPRESSED_BLOCK_ACK fragmentation must be avoided.
599  else if (((m_currentHdr.IsQosData () && !m_currentHdr.IsQosAmsdu ())
600  || (m_currentHdr.IsData () && !m_currentHdr.IsQosData ()))
601  && (GetBlockAckThreshold () == 0 || m_blockAckType == BASIC_BLOCK_ACK)
602  && NeedFragmentation ())
603  {
604  m_currentIsFragmented = true;
605  m_currentParams.DisableRts ();
606  WifiMacHeader hdr;
607  Ptr<Packet> fragment = GetFragmentPacket (&hdr);
608  if (IsLastFragment ())
609  {
610  NS_LOG_DEBUG ("fragmenting last fragment size=" << fragment->GetSize ());
611  m_currentParams.DisableNextData ();
612  }
613  else
614  {
615  NS_LOG_DEBUG ("fragmenting size=" << fragment->GetSize ());

757 {

759  if (!m_currentIsFragmented
760  || !m_currentParams.HasNextPacket ()
761  || m_currentHdr.IsQosAmsdu ())
762  {
763  NS_LOG_DEBUG ("got ack. tx done.");

765  {
766  m_txOkCallback (m_currentHdr);
767  }
768

784  else
785  {
786  m_low->DestroyBlockAckAgreement (m_currentHdr.GetAddr1 (), delBa.GetTid ());
787  }
788  }
789  else if (actionHdr.GetAction ().blockAck == WifiActionHeader::BLOCK_ACK_ADDBA_REQUEST)

792  MgtAddBaRequestHeader addBa;
793  p->PeekHeader (addBa);
794  Simulator::Schedule (m_addBaResponseTimeout,
795  &QosTxop::AddBaResponseTimeout, this,
796  m_currentHdr.GetAddr1 (), addBa.GetTid ());
797  }
798  }
799  }
800  if (m_currentHdr.IsQosData () && GetBaAgreementEstablished (m_currentHdr.GetAddr1 (), m_currentHdr.GetQosTid ()))
801  {
802  // notify the BA manager that the current packet was acknowledged
803  m_baManager->NotifyGotAck (Create<const WifiMacQueueItem> (m_currentPacket, m_currentHdr,
804  m_currentPacketTimestamp));
805  }
806  m_currentPacket = 0;
807  ResetCw ();
808  }
809  else
810  {
811  NS_LOG_DEBUG ("got ack. tx not done, size=" << m_currentPacket->GetSize ());
812  }

873 void
874 QosTxop::MissedBlockAck (uint8_t nMpdus)
875 {
876  NS_LOG_FUNCTION (this << +nMpdus);
877  /*
878  * If the BlockAck frame is lost, the originator may transmit a BlockAckReq
879  * frame to solicit an immediate BlockAck frame or it may retransmit the Data
880  * frames. (IEEE std 802.11-2016 sec. 10.24.7.7
881  */
882  uint8_t tid = GetTid (m_currentPacket, m_currentHdr);
883  if (m_useExplicitBarAfterMissedBlockAck || m_currentHdr.IsBlockAckReq ())
884  {
885  if (NeedBarRetransmission ())
886  {
887  NS_LOG_DEBUG ("Retransmit block ack request");
888  if (m_currentHdr.IsBlockAckReq ())
889  {
890  m_currentHdr.SetRetry ();
891  UpdateFailedCw ();
892  m_cwTrace = GetCw ();
893  }
894  else // missed BlockAck after data frame with Implicit BAR Ack Policy
895  {
896  Ptr<const WifiMacQueueItem> bar = PrepareBlockAckRequest (m_currentHdr.GetAddr1 (), tid);
897  ScheduleBar (bar);
898  m_currentPacket = 0;
899  }
900  }
901  else
902  {
903  NS_LOG_DEBUG ("Block Ack Request Fail");
904  // if a BA agreement exists, we can get here if there is no outstanding
905  // MPDU whose lifetime has not expired yet.
906  m_stationManager->ReportFinalDataFailed (m_currentHdr.GetAddr1 (), &m_currentHdr,
907  m_currentPacket->GetSize ());
908  if (m_baManager->ExistsAgreementInState (m_currentHdr.GetAddr1 (), tid,
909  OriginatorBlockAckAgreement::ESTABLISHED))
910  {
911  // If there is any (expired) outstanding MPDU, request the BA manager to discard
912  // it, which involves the scheduling of a BAR to advance the recipient's window
913  if (m_baManager->GetNBufferedPackets (m_currentHdr.GetAddr1 (), tid) > 0)
914  {
915  m_baManager->DiscardOutstandingMpdus (m_currentHdr.GetAddr1 (), tid);
916  }
917  // otherwise, it means that we have not received a BlockAck in response to a
918  // BlockAckRequest sent while no frame was outstanding, whose purpose was therefore
919  // to advance the recipient's window. Schedule a BlockAckRequest with
920  // skipIfNoDataQueued set to true, so that the BlockAckRequest is only sent
921  // if there are data frames queued for this recipient.
922  else
923  {
924  ScheduleBar (PrepareBlockAckRequest (m_currentHdr.GetAddr1 (), tid), true);
925  }
926  }
927  //to reset the Txop.
928  m_currentPacket = 0;
929  ResetCw ();
930  m_cwTrace = GetCw ();
931  }
932  }
933  else
934  {
935  if (GetAmpduExist (m_currentHdr.GetAddr1 ()))
936  {
937  m_stationManager->ReportAmpduTxStatus (m_currentHdr.GetAddr1 (), 0, nMpdus, 0, 0);
938  }
939  // implicit BAR and do not use BAR after missed BlockAck, hence try to retransmit data frames
940  if (!NeedDataRetransmission (m_currentPacket, m_currentHdr))
941  {
942  NS_LOG_DEBUG ("Block Ack Fail");
943  m_stationManager->ReportFinalDataFailed (m_currentHdr.GetAddr1 (), &m_currentHdr,
944  m_currentPacket->GetSize ());
945  if (!m_txFailedCallback.IsNull ())
946  {
947  m_txFailedCallback (m_currentHdr);
948  }
949  if (m_currentHdr.IsAction ())
950  {
951  WifiActionHeader actionHdr;
952  m_currentPacket->PeekHeader (actionHdr);
953  if (actionHdr.GetCategory () == WifiActionHeader::BLOCK_ACK)
954  {
955  uint8_t tid = GetTid (m_currentPacket, m_currentHdr);
956  if (m_baManager->ExistsAgreementInState (m_currentHdr.GetAddr1 (), tid, OriginatorBlockAckAgreement::PENDING))
957  {
958  NS_LOG_DEBUG ("No ACK after ADDBA request");
959  m_baManager->NotifyAgreementNoReply (m_currentHdr.GetAddr1 (), tid);
960  Simulator::Schedule (m_failedAddBaTimeout, &QosTxop::ResetBa, this, m_currentHdr.GetAddr1 (), tid);
961  }
962  }
963  }
964  //to reset the Txop.
965  m_baManager->DiscardOutstandingMpdus (m_currentHdr.GetAddr1 (), GetTid (m_currentPacket, m_currentHdr));
966  m_currentPacket = 0;
967  ResetCw ();
968  m_cwTrace = GetCw ();
969  }
970  else
971  {
972  NS_LOG_DEBUG ("Retransmit");
973  m_stationManager->ReportDataFailed (m_currentHdr.GetAddr1 (), &m_currentHdr, m_currentPacket->GetSize ());
974  m_baManager->NotifyMissedBlockAck (m_currentHdr.GetAddr1 (), tid);
975  m_currentPacket = 0;
976  UpdateFailedCw ();
977  m_cwTrace = GetCw ();
978  }
979  }
980  GenerateBackoff ();
981  RestartAccessIfNeeded ();
982 }

1079 {

1081  NS_ASSERT (GetTxopLimit ().IsStrictlyPositive () && GetTxopRemaining ().IsStrictlyPositive ());

1086

1088  {
1089  nextFrame = PeekNextFrame ();
1090  }
1091 
1092  if (nextFrame != 0)
1093  {

1095

1097  {

1099  m_currentHdr = nextFrame->GetHeader ();
1100  m_currentPacket = nextFrame->GetPacket ();

1105  && SetupBlockAckIfNeeded ())

1118  // we only peek the non-broadcast QoS Data frame. MacLow will be in charge of
1119  // computing the correct limits and dequeue the frame.

1122  item = Copy (nextFrame);
1123  }

1133  m_currentHdr = item->GetHeader ();
1134  m_currentPacketTimestamp = item->GetTimeStamp ();
1135  NS_ASSERT (m_currentPacket != 0);
1136

1138  m_stationManager->UpdateFragmentationThreshold ();
1139  m_fragmentNumber = 0;
1140  GetLow ()->StartTransmission (item, m_currentParams, this);
1141  return;
1142  }
1143  }

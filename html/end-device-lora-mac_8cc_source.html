107 void
108 EndDeviceLoraMac::Send (Ptr<Packet> packet)
109 {
110  NS_LOG_FUNCTION (this << packet);
111 
112  // Check that there are no scheduled receive windows.
113  // We cannot send a packet if we are in the process of transmitting or waiting
114  // for reception.
115  if (!m_closeWindow.IsExpired () || !m_secondReceiveWindow.IsExpired ())
116  {
117  NS_LOG_WARN ("Attempting to send when there are receive windows" <<
118  " Transmission canceled");
119  return;
120  }
121 
122  // Check that payload length is below the allowed maximum
123  if (packet->GetSize () > m_maxAppPayloadForDataRate.at (m_dataRate))
124  {
125  NS_LOG_WARN ("Attempting to send a packet larger than the maximum allowed"
126  << " size at this DataRate (DR" << unsigned(m_dataRate) <<
127  "). Transmission canceled.");
128  return;
129  }
130 
131  // Check that we can transmit according to the aggregate duty cycle timer
132  if (m_channelHelper.GetAggregatedWaitingTime () != Seconds (0))
133  {
134  NS_LOG_WARN ("Attempting to send, but the aggregate duty cycle won't allow it");
135  return;
136  }
137 
138  // Pick a channel on which to transmit the packet
139  Ptr<LogicalLoraChannel> txChannel = GetChannelForTx ();
140 
141  if (txChannel) // Proceed with transmission
142  {
144  // Add headers, prepare TX parameters and send the packet
146 
147  // Add the Lora Frame Header to the packet
148  LoraFrameHeader frameHdr;
149  ApplyNecessaryOptions (frameHdr);
150  packet->AddHeader (frameHdr);
151  NS_LOG_INFO ("Added frame header of size " << frameHdr.GetSerializedSize () <<
152  " bytes");
153 
154  // Add the Lora Mac header to the packet
155  LoraMacHeader macHdr;
156  ApplyNecessaryOptions (macHdr);
157  packet->AddHeader (macHdr);
158  NS_LOG_INFO ("Added MAC header of size " << macHdr.GetSerializedSize () <<
159  " bytes");
160 
161  // Craft LoraTxParameters object
162  LoraTxParameters params;
163  params.sf = GetSfFromDataRate (m_dataRate);
164  params.headerDisabled = m_headerDisabled;
165  params.codingRate = m_codingRate;
166  params.bandwidthHz = GetBandwidthFromDataRate (m_dataRate);
167  params.nPreamble = m_nPreambleSymbols;
168  params.crcEnabled = 1;
169  params.lowDataRateOptimizationEnabled = 0;
170 
171  // Wake up PHY layer and directly send the packet
172 
173  // Make sure we can transmit at the current power on this channel
174  NS_ASSERT (m_txPower <= m_channelHelper.GetTxPowerForChannel (txChannel));
175  m_phy->GetObject<EndDeviceLoraPhy> ()->SwitchToStandby ();
176  m_phy->Send (packet, params, txChannel->GetFrequency (), m_txPower);
177 
179  // Register packet transmission for duty cycle
181 
182  // Compute packet duration
183  Time duration = m_phy->GetOnAirTime (packet, params);
184 
185  // Register the sent packet into the DutyCycleHelper
186  m_channelHelper.AddEvent (duration, txChannel);
187 
189  // Prepare for the downlink //
191 
192  // Switch the PHY to the channel so that it will listen here for downlink
193  m_phy->GetObject<EndDeviceLoraPhy> ()->SetFrequency (txChannel->GetFrequency ());
194 
195  // Instruct the PHY on the right Spreading Factor to listen for during the window
196  uint8_t replyDataRate = GetFirstReceiveWindowDataRate ();
197  NS_LOG_DEBUG ("m_dataRate: " << unsigned (m_dataRate) <<
198  ", m_rx1DrOffset: " << unsigned (m_rx1DrOffset) <<
199  ", replyDataRate: " << unsigned (replyDataRate));
200 
201  m_phy->GetObject<EndDeviceLoraPhy> ()->SetSpreadingFactor
202  (GetSfFromDataRate (replyDataRate));
203 
204  }
205  else // Transmission cannot be performed
206  {
207  m_cannotSendBecauseDutyCycle (packet);
208  }
209 }

211 void
212 EndDeviceLoraMac::Receive (Ptr<Packet const> packet)
213 {
214  NS_LOG_FUNCTION (this << packet);
215 
216  // Work on a copy of the packet
217  Ptr<Packet> packetCopy = packet->Copy ();
218 
219  // Remove the Mac Header to get some information
220  LoraMacHeader mHdr;
221  packetCopy->RemoveHeader (mHdr);
222 
223  // Only keep analyzing the packet if it's downlink
224  if (!mHdr.IsUplink ())
225  {
226  NS_LOG_INFO ("Found a downlink packet");
227 
228  // Remove the Frame Header
229  LoraFrameHeader fHdr;
230  fHdr.SetAsDownlink ();
231  packetCopy->RemoveHeader (fHdr);
232 
233  // Determine whether this packet is for us
234  bool messageForUs = (m_address == fHdr.GetAddress ());
235 
236  if (messageForUs)
237  {
238  NS_LOG_INFO ("The message is for us!");
239 
240  // If it exists, cancel the second receive window event
241  Simulator::Cancel (m_secondReceiveWindow);
242 
243  // Parse the MAC commands
244  ParseCommands (fHdr);
245 
246  // TODO Pass the packet up to the NetDevice
247 
248 
249  // Call the trace source
250  m_receivedPacket (packet);
251  }
252  else
253  {
254  NS_LOG_DEBUG ("The message is intended for another recipient.");
255  }
256  }

259 void
260 EndDeviceLoraMac::ParseCommands (LoraFrameHeader frameHeader)
261 {
262  NS_LOG_FUNCTION (this << frameHeader);
263 
264  std::list<Ptr<MacCommand> > commands = frameHeader.GetCommands ();
265  std::list<Ptr<MacCommand> >::iterator it;
266  for (it = commands.begin (); it != commands.end (); it++)
267  {
268  NS_LOG_DEBUG ("Iterating over the MAC commands");
269  enum MacCommandType type = (*it)->GetCommandType ();
270  switch (type)
271  {
272  case (LINK_CHECK_ANS):
273  {
274  NS_LOG_DEBUG ("Detected a LinkCheckAns command");
275 
276  // Cast the command
277  Ptr<LinkCheckAns> linkCheckAns = (*it)->GetObject<LinkCheckAns> ();
278 
279  // Call the appropriate function to take action
280  OnLinkCheckAns (linkCheckAns->GetMargin (), linkCheckAns->GetGwCnt ());
281 
282  break;
283  }
284  case (LINK_ADR_REQ):
285  {
286  NS_LOG_DEBUG ("Detected a LinkAdrReq command");
287 
288  // Cast the command
289  Ptr<LinkAdrReq> linkAdrReq = (*it)->GetObject<LinkAdrReq> ();
290 
291  // Call the appropriate function to take action
292  OnLinkAdrReq (linkAdrReq->GetDataRate (), linkAdrReq->GetTxPower (),
293  linkAdrReq->GetEnabledChannelsList (),
294  linkAdrReq->GetRepetitions ());
295 
296  break;
297  }
298  case (DUTY_CYCLE_REQ):
299  {
300  NS_LOG_DEBUG ("Detected a DutyCycleReq command");
301 
302  // Cast the command
303  Ptr<DutyCycleReq> dutyCycleReq = (*it)->GetObject<DutyCycleReq> ();
304 
305  // Call the appropriate function to take action
306  OnDutyCycleReq (dutyCycleReq->GetMaximumAllowedDutyCycle ());
307 
308  break;
309  }
310  case (RX_PARAM_SETUP_REQ):
311  {
312  NS_LOG_DEBUG ("Detected a RxParamSetupReq command");
313 
314  // Cast the command
315  Ptr<RxParamSetupReq> rxParamSetupReq = (*it)->GetObject<RxParamSetupReq> ();
316 
317  // Call the appropriate function to take action
318  OnRxParamSetupReq (rxParamSetupReq->GetRx1DrOffset (),
319  rxParamSetupReq->GetRx2DataRate (),
320  rxParamSetupReq->GetFrequency ());
321 
322  break;
323  }
324  case (DEV_STATUS_REQ):
325  {
326  NS_LOG_DEBUG ("Detected a DevStatusReq command");
327 
328  // Cast the command
329  Ptr<DevStatusReq> devStatusReq = (*it)->GetObject<DevStatusReq> ();
330 
331  // Call the appropriate function to take action
332  OnDevStatusReq ();
333 
334  break;
335  }
336  case (NEW_CHANNEL_REQ):
337  {
338  NS_LOG_DEBUG ("Detected a NewChannelReq command");
339 
340  // Cast the command
341  Ptr<NewChannelReq> newChannelReq = (*it)->GetObject<NewChannelReq> ();
342 
343  // Call the appropriate function to take action
344  OnNewChannelReq (newChannelReq->GetChannelIndex (), newChannelReq->GetFrequency (), newChannelReq->GetMinDataRate (), newChannelReq->GetMaxDataRate ());
345 
346  break;
347  }
348  case (RX_TIMING_SETUP_REQ):
349  {
350  break;
351  }
352  case (TX_PARAM_SETUP_REQ):
353  {
354  break;
355  }
356  case (DL_CHANNEL_REQ):
357  {
358  break;
359  }
360  default:
361  {
362  NS_LOG_ERROR ("CID not recognized");
363  break;
364  }
365  }
366  }

445 void
446 EndDeviceLoraMac::CloseFirstReceiveWindow (void)
447 {
448  NS_LOG_FUNCTION_NOARGS ();
449 
450  Ptr<EndDeviceLoraPhy> phy = m_phy->GetObject<EndDeviceLoraPhy> ();
451 
452  // Check the Phy layer's state:
453  // - RX -> We have received a preamble.
454  // - STANDBY -> Nothing was detected.
455  // We should never be in TX or SLEEP mode at this point
456  switch (phy->GetState ())
457  {
458  case EndDeviceLoraPhy::TX:
459  case EndDeviceLoraPhy::SLEEP:
460  NS_ABORT_MSG ("PHY was in TX or SLEEP mode when attempting to " <<
461  "close a receive window");
462  break;
463  case EndDeviceLoraPhy::RX:
464  // PHY is receiving: let it finish. The Receive method will switch it back to SLEEP.
465  break;
466  case EndDeviceLoraPhy::STANDBY:
467  // Turn PHY layer to SLEEP
468  phy->SwitchToSleep ();
469  break;
470  }
471 }

474 EndDeviceLoraMac::OpenSecondReceiveWindow (void)
475 {
476  NS_LOG_FUNCTION_NOARGS ();
477 
478  // Check for receiver status: if it's locked on a packet, don't open this
479  // window at all.
480  if (m_phy->GetObject<EndDeviceLoraPhy> ()->GetState () == EndDeviceLoraPhy::RX)
481  {
482  NS_LOG_INFO ("Won't open second receive window since we are in RX mode");
483

485  }
486 
487  // Set Phy in Standby mode
488  m_phy->GetObject<EndDeviceLoraPhy> ()->SwitchToStandby ();
489 
490  // Switch to appropriate channel and data rate
491  NS_LOG_INFO ("Using parameters: " << m_secondReceiveWindowFrequency << "Hz, DR"
492  << unsigned(m_secondReceiveWindowDataRate));
493 
494  m_phy->GetObject<EndDeviceLoraPhy> ()->SetFrequency
495  (m_secondReceiveWindowFrequency);
496  m_phy->GetObject<EndDeviceLoraPhy> ()->SetSpreadingFactor (GetSfFromDataRate
497  (m_secondReceiveWindowDataRate));
498 
499  // Schedule return to sleep after "at least the time required by the end
500  // device's radio transceiver to effectively detect a downlink preamble"
501  // (LoraWAN specification)
502  m_closeWindow = Simulator::Schedule (m_receiveWindowDuration,
503  &EndDeviceLoraMac::CloseSecondReceiveWindow, this);

508 {
509  NS_LOG_FUNCTION_NOARGS ();
510 
511  Ptr<EndDeviceLoraPhy> phy = m_phy->GetObject<EndDeviceLoraPhy> ();

535 Ptr<LogicalLoraChannel>
536 EndDeviceLoraMac::GetChannelForTx (void)
537 {
538  NS_LOG_FUNCTION_NOARGS ();
539 
540  // Pick a random channel to transmit on
541  std::vector<Ptr<LogicalLoraChannel> > logicalChannels;
542  logicalChannels = m_channelHelper.GetChannelList (); // Use a separate list to do the shuffle
543  logicalChannels = Shuffle (logicalChannels);
544 
545  // Try every channel
546  std::vector<Ptr<LogicalLoraChannel> >::iterator it;
547  for (it = logicalChannels.begin (); it != logicalChannels.end (); ++it)
548  {
549  // Pointer to the current channel
550  Ptr<LogicalLoraChannel> logicalChannel = *it;
551  double frequency = logicalChannel->GetFrequency ();
552 
553  NS_LOG_DEBUG ("Frequency of the current channel: " << frequency);
554 
555  // Verify that we can send the packet
556  Time waitingTime = m_channelHelper.GetWaitingTime (logicalChannel);
557 
558  NS_LOG_DEBUG ("Waiting time for current channel = " <<
559  waitingTime.GetSeconds ());
560 
561  // Send immediately if we can
562  if (waitingTime == Seconds (0))
563  {
564  return *it;
565  }
566  else
567  {
568  NS_LOG_DEBUG ("Packet cannot be immediately transmitted on " <<
569  "the current channel because of duty cycle limitations");
570  }
571  }
572  return 0; // In this case, no suitable channel was found
573 }

576 EndDeviceLoraMac::Shuffle (std::vector<Ptr<LogicalLoraChannel> > vector)
577 {
578  NS_LOG_FUNCTION_NOARGS ();
579 
580  int size = vector.size ();
581 
582  for (int i = 0; i < size; ++i)
583  {
584  uint16_t random = std::floor (m_uniformRV->GetValue (0, size));
585  Ptr<LogicalLoraChannel> temp = vector.at (random);

587  vector.at (i) = temp;
588  }
589 
590  return vector;

638 void
639 EndDeviceLoraMac::OnLinkAdrReq (uint8_t dataRate, uint8_t txPower,
640  std::list<int> enabledChannels, int repetitions)
641 {
642  NS_LOG_FUNCTION (this << unsigned (dataRate) << unsigned (txPower) <<
643  repetitions);
644 
645  // Three bools for three requirements before setting things up
646  bool channelMaskOk = true;
647  bool dataRateOk = true;
648  bool txPowerOk = true;
649 
650  // Check the channel mask
652  // Check whether all specified channels exist on this device
653  auto channelList = m_channelHelper.GetChannelList ();
654  int channelListSize = channelList.size ();
655 
656  for (auto it = enabledChannels.begin (); it != enabledChannels.end (); it++)
657  {
658  if ((*it) > channelListSize)
659  {
660  channelMaskOk = false;
661  break;
662  }
663  }
664 
665  // Check the dataRate
667  // We need to know we can use it at all
668  // To assess this, we try and convert it to a SF/BW combination and check if
669  // those values are valid. Since GetSfFromDataRate and
670  // GetBandwidthFromDataRate return 0 if the dataRate is not recognized, we
671  // can check against this.
672  uint8_t sf = GetSfFromDataRate (dataRate);
673  double bw = GetBandwidthFromDataRate (dataRate);
674  NS_LOG_DEBUG ("SF: " << unsigned (sf) << ", BW: " << bw);
675  if (sf == 0 || bw == 0)
676  {
677  dataRateOk = false;
678  }
679 
680  // We need to know we can use it in at least one of the enabled channels
681  // Cycle through available channels, stop when at least one is enabled for the
682  // specified dataRate.
683  if (dataRateOk && channelMaskOk) // If false, skip the check
684  {
685  bool foundAvailableChannel = false;
686  for (auto it = enabledChannels.begin (); it != enabledChannels.end (); it++)
687  {
688  NS_LOG_DEBUG ("MinDR: " << unsigned (channelList.at (*it)->GetMinimumDataRate ()));
689  NS_LOG_DEBUG ("MaxDR: " << unsigned (channelList.at (*it)->GetMaximumDataRate ()));
690  if (channelList.at (*it)->GetMinimumDataRate () <= dataRate &&
691  channelList.at (*it)->GetMaximumDataRate () >= dataRate)
692  {
693  foundAvailableChannel = true;
694  break;
695  }
696  }
697 
698  if (!foundAvailableChannel)
699  {
700  dataRateOk = false;
701  }
702  }
703 
704  // Check the txPower
706  // Check whether we can use this transmission power
707  if (GetDbmForTxPower (txPower) == 0)
708  {
709  txPowerOk = false;
710  }
711 
712  NS_LOG_DEBUG ("Finished checking. " <<
713  "ChannelMaskOk: " << channelMaskOk << ", " <<
714  "DataRateOk: " << dataRateOk << ", " <<
715  "txPowerOk: " << txPowerOk);
716 
717  // If all checks are successful, set parameters up

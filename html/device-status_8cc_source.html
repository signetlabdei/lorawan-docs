98 std::list<Address>
99 DeviceStatus::GetSortedGatewayAddresses (void)
100 {
101  NS_LOG_FUNCTION (this);
102 
103  // Copy the map pairs into a vector
104  std::vector<std::pair<Address, double> > pairs;
105  for (auto it = m_gateways.begin (); it != m_gateways.end (); ++it)
106  {
107  pairs.push_back (*it);
108  }
109 
110  // Sort the vector
111  std::sort (pairs.begin (), pairs.end (),
112  [] (const std::pair<Address, double>&p1,
113  const std::pair<Address, double>&p2)
114  { return p1.second > p2.second; }
115  );
116 
117  // Create a new array with only the addresses
118  std::list<Address> addresses;
119  for (auto it = pairs.begin (); it != pairs.end (); ++it)
120  {
121  addresses.push_back ((*it).first);
122  }
123 
124  // Return the list
125  return addresses;
126 }

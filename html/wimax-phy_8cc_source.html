40 {

48  "Wimax channel",

130 void
131 WimaxPhy::StartScanning (uint64_t frequency, Time timeout, Callback<void, bool, uint64_t> callback)
132 {
133  NS_ASSERT_MSG (m_state == PHY_STATE_IDLE || m_state == PHY_STATE_SCANNING,
134  "Error while scanning: The PHY state should be PHY_STATE_SCANNING or PHY_STATE_IDLE");
135 
136  m_state = PHY_STATE_SCANNING;
137  m_scanningFrequency = frequency;
138  m_dlChnlSrchTimeoutEvent = Simulator::Schedule (timeout, &WimaxPhy::EndScanning, this);
139  m_scanningCallback = callback;
140 }

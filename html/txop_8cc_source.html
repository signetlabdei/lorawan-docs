475 void
476 Txop::NotifyAccessGranted (void)
477 {
478  NS_LOG_FUNCTION (this);
479  NS_ASSERT (m_accessRequested);
480  m_accessRequested = false;
481  if (m_currentPacket == 0)
482  {
483  if (m_queue->IsEmpty ())
484  {
485  NS_LOG_DEBUG ("queue empty");
486  return;
487  }
488  Ptr<WifiMacQueueItem> item = m_queue->Dequeue ();
489  NS_ASSERT (item != 0);
490  m_currentPacket = item->GetPacket ();
491  m_currentHdr = item->GetHeader ();
492  NS_ASSERT (m_currentPacket != 0);
493  uint16_t sequence = m_txMiddle->GetNextSequenceNumberFor (&m_currentHdr);
494  m_currentHdr.SetSequenceNumber (sequence);
495  m_stationManager->UpdateFragmentationThreshold ();
496  m_currentHdr.SetFragmentNumber (0);
497  m_currentHdr.SetNoMoreFragments ();
498  m_currentHdr.SetNoRetry ();
499  m_fragmentNumber = 0;
500  NS_LOG_DEBUG ("dequeued size=" << m_currentPacket->GetSize () <<
501  ", to=" << m_currentHdr.GetAddr1 () <<
502  ", seq=" << m_currentHdr.GetSequenceControl ());
503  }
504  if (m_currentHdr.GetAddr1 ().IsGroup ())
505  {
506  m_currentParams.DisableRts ();
507  m_currentParams.DisableAck ();
508  m_currentParams.DisableNextData ();
509  NS_LOG_DEBUG ("tx broadcast");
510  GetLow ()->StartTransmission (Create<WifiMacQueueItem> (m_currentPacket, m_currentHdr),
511  m_currentParams, this);
512  }
513  else
514  {
515  m_currentParams.EnableAck ();
516  if (NeedFragmentation ())
517  {
518  m_currentParams.DisableRts ();
519  WifiMacHeader hdr;
520  Ptr<Packet> fragment = GetFragmentPacket (&hdr);
521  if (IsLastFragment ())
522  {
523  NS_LOG_DEBUG ("fragmenting last fragment size=" << fragment->GetSize ());
524  m_currentParams.DisableNextData ();
525  }
526  else
527  {
528  NS_LOG_DEBUG ("fragmenting size=" << fragment->GetSize ());
529  m_currentParams.EnableNextData (GetNextFragmentSize ());
530  }
531  GetLow ()->StartTransmission (Create<WifiMacQueueItem> (fragment, hdr),
532  m_currentParams, this);
533  }
534  else
535  {
536  if (m_stationManager->NeedRts (&m_currentHdr, m_currentPacket) && !m_low->IsCfPeriod ())
537  {
538  m_currentParams.EnableRts ();
539  }
540  else
541  {
542  m_currentParams.DisableRts ();
543  }
544  m_currentParams.DisableNextData ();
545  GetLow ()->StartTransmission (Create<WifiMacQueueItem> (m_currentPacket, m_currentHdr),
546  m_currentParams, this);
547  }
548  }

610 Txop::MissedCts (void)

615  {
616  NS_LOG_DEBUG ("Cts Fail");

618  if (!m_txFailedCallback.IsNull ())

620  m_txFailedCallback (m_currentHdr);

622  //to reset the Txop.

624  ResetCw ();
625  m_cwTrace = GetCw ();
626  }

629  UpdateFailedCw ();
630  m_cwTrace = GetCw ();
631  }
632  GenerateBackoff ();
633  RestartAccessIfNeeded ();

636 void
637 Txop::GotAck (void)
638 {
639  NS_LOG_FUNCTION (this);
640  if (!NeedFragmentation ()
641  || IsLastFragment ())
642  {
643  NS_LOG_DEBUG ("got ack. tx done.");
644  if (!m_txOkCallback.IsNull ())
645  {
646  m_txOkCallback (m_currentHdr);
647  }
648 
649  /* we are not fragmenting or we are done fragmenting
650  * so we can get rid of that packet now.
651  */
652  m_currentPacket = 0;
653  ResetCw ();
654  m_cwTrace = GetCw ();
655  GenerateBackoff ();
656  RestartAccessIfNeeded ();
657  }
658  else
659  {
660  NS_LOG_DEBUG ("got ack. tx not done, size=" << m_currentPacket->GetSize ());
661  }
662 }

688  m_currentHdr.SetRetry ();

769  NS_LOG_DEBUG ("a transmission that did not require an ACK just finished");
770  m_currentPacket = 0;
771  ResetCw ();
772  m_cwTrace = GetCw ();
773  GenerateBackoff ();

781 void
782 Txop::SendCfFrame (WifiMacType frameType, Mac48Address addr)
783 {
784  NS_LOG_FUNCTION (this << frameType << addr);
785  NS_ASSERT (m_low->IsCfPeriod ());
786  if (m_currentPacket != 0 && frameType != WIFI_MAC_CTL_END)
787  {
788  if (!NeedDataRetransmission (m_currentPacket, m_currentHdr))
789  {
790  m_stationManager->ReportFinalDataFailed (m_currentHdr.GetAddr1 (), &m_currentHdr,
791  m_currentPacket->GetSize ());
792  m_currentPacket = 0;
793  }
794  else
795  {
796  m_currentHdr.SetRetry ();
797  }
798  }
799  else if ((m_queue->GetNPacketsByAddress (addr) > 0) && (frameType != WIFI_MAC_CTL_END)) //if no packet for that dest, send to another dest?
800  {
801  Ptr<WifiMacQueueItem> item = m_queue->DequeueByAddress (addr);
802  NS_ASSERT (item != 0);
803  m_currentPacket = item->GetPacket ();
804  m_currentHdr = item->GetHeader ();
805  uint16_t sequence = m_txMiddle->GetNextSequenceNumberFor (&m_currentHdr);
806  m_currentHdr.SetSequenceNumber (sequence);
807  m_currentHdr.SetFragmentNumber (0);
808  m_currentHdr.SetNoMoreFragments ();
809  m_currentHdr.SetNoRetry ();
810  }
811  else
812  {
813  m_currentPacket = Create<Packet> ();
814  m_currentHdr.SetNoRetry ();
815  }
816 
817  if (m_currentPacket->GetSize () > 0)
818  {
819  switch (frameType)
820  {
821  case WIFI_MAC_DATA_NULL_CFPOLL:
822  m_currentHdr.SetType (WIFI_MAC_DATA_CFPOLL);
823  break;
824  case WIFI_MAC_DATA_NULL:
825  m_currentHdr.SetType (WIFI_MAC_DATA);
826  break;
827  default:
828  NS_ASSERT (false);
829  break;
830  }
831  }
832  else
833  {
834  m_currentHdr.SetType (frameType);
835  }
836  m_currentHdr.SetAddr1 (addr);
837  m_currentHdr.SetAddr2 (m_low->GetAddress ());
838  if (frameType == WIFI_MAC_DATA_NULL)
839  {
840  m_currentHdr.SetAddr3 (m_low->GetBssid ());
841  m_currentHdr.SetDsTo ();
842  m_currentHdr.SetDsNotFrom ();
843  }
844  else
845  {
846  m_currentHdr.SetAddr3 (m_low->GetAddress ());
847  m_currentHdr.SetDsNotTo ();
848  m_currentHdr.SetDsFrom ();
849  }
850  m_channelAccessManager->RequestAccess (this, true);
851 }

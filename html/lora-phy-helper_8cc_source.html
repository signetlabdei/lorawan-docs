72 Ptr<LoraPhy>
73 LoraPhyHelper::Create (Ptr<Node> node, Ptr<NetDevice> device) const
74 {
75  NS_LOG_FUNCTION (this << node->GetId () << device);
76 
77  // Create the PHY and set its channel
78  Ptr<LoraPhy> phy = m_phy.Create<LoraPhy> ();
79  phy->SetChannel (m_channel);
80 
81  // Configuration is different based on the kind of device we have to create
82  std::string typeId = m_phy.GetTypeId ().GetName ();
83  if (typeId == "ns3::SimpleGatewayLoraPhy")
84  {
85  // Inform the channel of the presence of this PHY
86  m_channel->Add (phy);
87 
88  // For now, assume that the PHY will listen to the default EU channels
89  // with this ReceivePath configuration:
90  // 3 ReceivePaths on 868.1
91  // 3 ReceivePaths on 868.3
92  // 2 ReceivePaths on 868.5
93 
94  // We expect that MacHelper instances will overwrite this setting if the
95  // device will operate in a different region
96  std::vector<double> frequencies;
97  frequencies.push_back (868.1);
98  frequencies.push_back (868.3);
99  frequencies.push_back (868.5);
100 
101  std::vector<double>::iterator it = frequencies.begin ();
102 
103  int receptionPaths = 0;
104  // Set maxReceptionPaths as a parameter
105  // int maxReceptionPaths = 8;
106  while (receptionPaths < m_maxReceptionPaths)
107  {
108  if (it == frequencies.end ())
109  {
110  it = frequencies.begin ();
111  }
112  phy->GetObject<SimpleGatewayLoraPhy> ()->AddReceptionPath (*it);
113  ++it;
114  receptionPaths++;
115  }
116 
117  }
118  else if (typeId == "ns3::SimpleEndDeviceLoraPhy")
119  {
120  // The line below can be commented to speed up uplink-only simulations.
121  // This implies that the LoraChannel instance will only know about
122  // Gateways, and it will not lose time delivering packets and interference
123  // information to devices which will never listen.
124 
125  m_channel->Add (phy);
126  }
127 
128  // Link the PHY to its net device
129  phy->SetDevice (device);
130 
131  return phy;
132 }

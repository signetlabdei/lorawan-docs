41  NetDeviceContainer
42  LoraHelper::Install ( const LoraPhyHelper &phyHelper,
43  const LorawanMacHelper &macHelper,
44  NodeContainer c) const
45  {
46  NS_LOG_FUNCTION_NOARGS ();
47 
48  NetDeviceContainer devices;
49 
50  // Go over the various nodes in which to install the NetDevice
51  for (NodeContainer::Iterator i = c.Begin (); i != c.End (); ++i)
52  {
53  Ptr<Node> node = *i;
54 
55  // Create the LoraNetDevice
56  Ptr<LoraNetDevice> device = CreateObject<LoraNetDevice> ();
57 
58  // Create the PHY
59  Ptr<LoraPhy> phy = phyHelper.Create (node, device);
60  NS_ASSERT (phy != 0);
61  device->SetPhy (phy);
62  NS_LOG_DEBUG ("Done creating the PHY");
63 
64  // Connect Trace Sources if necessary
65  if (m_packetTracker)
66  {
67  if (phyHelper.GetDeviceType () ==
68  TypeId::LookupByName ("ns3::SimpleEndDeviceLoraPhy"))
69  {
70  phy->TraceConnectWithoutContext ("StartSending",
71  MakeCallback
72  (&LoraPacketTracker::TransmissionCallback,
73  m_packetTracker));
74  }
75  else if (phyHelper.GetDeviceType () ==
76  TypeId::LookupByName ("ns3::SimpleGatewayLoraPhy"))
77  {
78  phy->TraceConnectWithoutContext ("StartSending",
79  MakeCallback
80  (&LoraPacketTracker::TransmissionCallback,
81  m_packetTracker));
82  phy->TraceConnectWithoutContext ("ReceivedPacket",
83  MakeCallback
84  (&LoraPacketTracker::PacketReceptionCallback,
85  m_packetTracker));
86  phy->TraceConnectWithoutContext ("LostPacketBecauseInterference",
87  MakeCallback
88  (&LoraPacketTracker::InterferenceCallback,
89  m_packetTracker));
90  phy->TraceConnectWithoutContext ("LostPacketBecauseNoMoreReceivers",
91  MakeCallback
92  (&LoraPacketTracker::NoMoreReceiversCallback,
93  m_packetTracker));
94  phy->TraceConnectWithoutContext ("LostPacketBecauseUnderSensitivity",
95  MakeCallback
96  (&LoraPacketTracker::UnderSensitivityCallback,
97  m_packetTracker));
98  phy->TraceConnectWithoutContext ("NoReceptionBecauseTransmitting",
99  MakeCallback
100  (&LoraPacketTracker::LostBecauseTxCallback,
101  m_packetTracker));
102  }
103  }
104 
105  // Create the MAC
106  Ptr<LorawanMac> mac = macHelper.Create (node, device);
107  NS_ASSERT (mac != 0);
108  mac->SetPhy (phy);
109  NS_LOG_DEBUG ("Done creating the MAC");
110  device->SetMac (mac);
111 
112  if (m_packetTracker)
113  {
114  if (phyHelper.GetDeviceType () ==
115  TypeId::LookupByName ("ns3::SimpleEndDeviceLoraPhy"))
116  {
117  mac->TraceConnectWithoutContext ("SentNewPacket",
118  MakeCallback
119  (&LoraPacketTracker::MacTransmissionCallback,
120  m_packetTracker));
121 
122  mac->TraceConnectWithoutContext ("RequiredTransmissions",
123  MakeCallback
124  (&LoraPacketTracker::RequiredTransmissionsCallback,
125  m_packetTracker));
126  }
127  else if (phyHelper.GetDeviceType () ==
128  TypeId::LookupByName ("ns3::SimpleGatewayLoraPhy"))
129  {
130  mac->TraceConnectWithoutContext ("SentNewPacket",
131  MakeCallback
132  (&LoraPacketTracker::MacTransmissionCallback,
133  m_packetTracker));
134 
135  mac->TraceConnectWithoutContext ("ReceivedPacket",
136  MakeCallback
137  (&LoraPacketTracker::MacGwReceptionCallback,
138  m_packetTracker));
139  }
140  }
141 
142  node->AddDevice (device);
143  devices.Add (device);
144  NS_LOG_DEBUG ("node=" << node << ", mob=" << node->GetObject<MobilityModel> ()->GetPosition ());
145  }
146  return devices;
147 }

198 void
199 LoraHelper::DoPrintDeviceStatus (NodeContainer endDevices, NodeContainer gateways,
200  std::string filename)
201 {
202  const char * c = filename.c_str ();
203  std::ofstream outputFile;
204  if (Simulator::Now () == Seconds (0))
205  {
206  // Delete contents of the file as it is opened
207  outputFile.open (c, std::ofstream::out | std::ofstream::trunc);
208  }
209  else
210  {
211  // Only append to the file
212  outputFile.open (c, std::ofstream::out | std::ofstream::app);
213  }
214 
215  Time currentTime = Simulator::Now();
216  for (NodeContainer::Iterator j = endDevices.Begin (); j != endDevices.End (); ++j)
217  {
218  Ptr<Node> object = *j;
219  Ptr<MobilityModel> position = object->GetObject<MobilityModel> ();
220  NS_ASSERT (position != 0);
221  Ptr<NetDevice> netDevice = object->GetDevice (0);
222  Ptr<LoraNetDevice> loraNetDevice = netDevice->GetObject<LoraNetDevice> ();
223  NS_ASSERT (loraNetDevice != 0);
224  Ptr<ClassAEndDeviceLorawanMac> mac = loraNetDevice->GetMac ()->GetObject<ClassAEndDeviceLorawanMac> ();
225  int dr = int(mac->GetDataRate ());
226  double txPower = mac->GetTransmissionPower ();
227  Vector pos = position->GetPosition ();
228  outputFile << currentTime.GetSeconds () << " "
229  << object->GetId () << " "
230  << pos.x << " " << pos.y << " " << dr << " "
231  << unsigned(txPower) << std::endl;
232  }
233  // for (NodeContainer::Iterator j = gateways.Begin (); j != gateways.End (); ++j)
234  // {
235  // Ptr<Node> object = *j;
236  // Ptr<MobilityModel> position = object->GetObject<MobilityModel> ();
237  // Vector pos = position->GetPosition ();
238  // outputFile << currentTime.GetSeconds () << " "
239  // << object->GetId () << " "
240  // << pos.x << " " << pos.y << " " << "-1 -1" << std::endl;
241  // }
242  outputFile.close ();
243 }

109 void
110 BasicEnergyHarvester::UpdateHarvestedPower (void)
111 {
112  NS_LOG_FUNCTION (this);
113  NS_LOG_DEBUG (Simulator::Now ().GetSeconds ()
114  << "s BasicEnergyHarvester(" << GetNode ()->GetId () << "): Updating harvesting power.");
115 
116  Time duration = Simulator::Now () - m_lastHarvestingUpdateTime;
117 
118  NS_ASSERT (duration.GetNanoSeconds () >= 0); // check if duration is valid
119 
120  double energyHarvested = 0.0;
121 
122  // do not update if simulation has finished
123  if (Simulator::IsFinished ())
124  {
125  NS_LOG_DEBUG ("BasicEnergyHarvester: Simulation Finished.");
126  return;
127  }
128 
129  m_energyHarvestingUpdateEvent.Cancel ();
130 
131  CalculateHarvestedPower ();
132 
133  energyHarvested = duration.GetSeconds () * m_harvestedPower;
134 
135  // update total energy harvested
136  m_totalEnergyHarvestedJ += energyHarvested;
137 
138  // notify energy source
139  GetEnergySource ()->UpdateEnergySource ();
140 
141  // update last harvesting time stamp
142  m_lastHarvestingUpdateTime = Simulator::Now ();
143 
144  m_energyHarvestingUpdateEvent = Simulator::Schedule (m_harvestedPowerUpdateInterval,
145  &BasicEnergyHarvester::UpdateHarvestedPower,
146  this);
147 }

69 int64_t SixLowPanHelper::AssignStreams (NetDeviceContainer c, int64_t stream)
70 {
71  int64_t currentStream = stream;
72  Ptr<NetDevice> netDevice;
73  for (NetDeviceContainer::Iterator i = c.Begin (); i != c.End (); ++i)
74  {
75  netDevice = (*i);
76  Ptr<SixLowPanNetDevice> dev = DynamicCast<SixLowPanNetDevice> (netDevice);
77  if (dev)
78  {
79  currentStream += dev->AssignStreams (currentStream);
80  }
81  }
82  return (currentStream - stream);

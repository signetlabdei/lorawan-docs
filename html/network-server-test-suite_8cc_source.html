74 {
75  NS_LOG_DEBUG ("UplinkPacketTest");
76 
77  // Create a bunch of actual devices
78  NetworkComponents components = InitializeNetwork (1, 1);
79 
80  Ptr<LoraChannel> channel = components.channel;
81  NodeContainer endDevices = components.endDevices;
82  NodeContainer gateways = components.gateways;
83  Ptr<Node> nsNode = components.nsNode;
84 
85  // Connect the trace source for received packets
86  nsNode->GetApplication (0)->TraceConnectWithoutContext
87  ("ReceivedPacket",
88  MakeCallback
89  (&UplinkPacketTest::ReceivedPacket,
90  this));
91 
92  // Send a packet
93  Simulator::Schedule (Seconds (1), &UplinkPacketTest::SendPacket, this,
94  endDevices.Get (0));
95 
96  Simulator::Stop (Seconds (5));
97  Simulator::Run ();
98  Simulator::Destroy ();
99 
100  // Check that we received the packet
101  NS_ASSERT (m_receivedPacket == true);
102 }

150 void
151 EndDeviceLorawanMac::Send (Ptr<Packet> packet)
152 {
153  NS_LOG_FUNCTION (this << packet);
154 
155  // Check that payload length is below the allowed maximum
156  if (packet->GetSize () > m_maxAppPayloadForDataRate.at (m_dataRate))
157  {
158  NS_LOG_WARN ("Attempting to send a packet larger than the maximum allowed"
159  << " size at this DataRate (DR" << unsigned(m_dataRate) <<
160  "). Transmission canceled.");
161  return;
162  }
163 
164  // If it is not possible to transmit now because of the duty cycle,
165  // or because we are receiving, schedule a tx/retx later
166 
167  Time netxTxDelay = GetNextTransmissionDelay ();
168  if (netxTxDelay != Seconds (0))
169  {
170  postponeTransmission (netxTxDelay, packet);
171  return;
172  }
173 
174  // Pick a channel on which to transmit the packet
175  Ptr<LogicalLoraChannel> txChannel = GetChannelForTx ();
176 
177  if (!(txChannel && m_retxParams.retxLeft > 0))
178  {
179  if (!txChannel)
180  {
181  m_cannotSendBecauseDutyCycle (packet);
182  }
183  else
184  {
185  NS_LOG_INFO ("Max number of transmission achieved: packet not transmitted.");
186  }
187  }
188  else
189  // the transmitting channel is available and we have not run out the maximum number of retransmissions
190  {
191  // Make sure we can transmit at the current power on this channel
192  NS_ASSERT_MSG (m_txPower <= m_channelHelper.GetTxPowerForChannel (txChannel),
193  " The selected power is too hight to be supported by this channel.");
194  DoSend (packet);
195  }

199 EndDeviceLorawanMac::postponeTransmission (Time netxTxDelay, Ptr<Packet> packet)
200 {
201  NS_LOG_FUNCTION (this);
202  // Delete previously scheduled transmissions if any.
203  Simulator::Cancel (m_nextTx);
204  m_nextTx = Simulator::Schedule (netxTxDelay, &EndDeviceLorawanMac::DoSend, this, packet);
205  NS_LOG_WARN ("Attempting to send, but the aggregate duty cycle won't allow it. Scheduling a tx at a delay "
206  << netxTxDelay.GetSeconds () << ".");
207 }

210 void
211 EndDeviceLorawanMac::DoSend (Ptr<Packet> packet)
212 {
213  NS_LOG_FUNCTION (this);
214  // Checking if this is the transmission of a new packet
215  if (packet != m_retxParams.packet)
216  {
217  NS_LOG_DEBUG ("Received a new packet from application. Resetting retransmission parameters.");
218  m_currentFCnt++;
219  NS_LOG_DEBUG ("APP packet: " << packet << ".");
220 
221  // Add the Lora Frame Header to the packet
222  LoraFrameHeader frameHdr;
223  ApplyNecessaryOptions (frameHdr);
224  packet->AddHeader (frameHdr);
225 
226  NS_LOG_INFO ("Added frame header of size " << frameHdr.GetSerializedSize () <<
227  " bytes.");
228 
229  // Add the Lora Mac header to the packet
230  LorawanMacHeader macHdr;
231  ApplyNecessaryOptions (macHdr);
232  packet->AddHeader (macHdr);
233 
234  // Reset MAC command list
235  m_macCommandList.clear ();
236 
237  if (m_retxParams.waitingAck)
238  {
239  // Call the callback to notify about the failure
240  uint8_t txs = m_maxNumbTx - (m_retxParams.retxLeft);
241  m_requiredTxCallback (txs, false, m_retxParams.firstAttempt, m_retxParams.packet);
242  NS_LOG_DEBUG (" Received new packet from the application layer: stopping retransmission procedure. Used " <<
243  unsigned(txs) << " transmissions out of a maximum of " << unsigned(m_maxNumbTx) << ".");
244  }
245 
246  // Reset retransmission parameters
247  resetRetransmissionParameters ();
248 
249  // If this is the first transmission of a confirmed packet, save parameters for the (possible) next retransmissions.
250  if (m_mType == LorawanMacHeader::CONFIRMED_DATA_UP)
251  {
252  m_retxParams.packet = packet->Copy ();
253  m_retxParams.retxLeft = m_maxNumbTx;
254  m_retxParams.waitingAck = true;
255  m_retxParams.firstAttempt = Simulator::Now ();
256  m_retxParams.retxLeft = m_retxParams.retxLeft - 1; // decreasing the number of retransmissions
257 
258  NS_LOG_DEBUG ("Message type is " << m_mType);
259  NS_LOG_DEBUG ("It is a confirmed packet. Setting retransmission parameters and decreasing the number of transmissions left.");
260 
261  NS_LOG_INFO ("Added MAC header of size " << macHdr.GetSerializedSize () <<
262  " bytes.");
263 
264  // Sent a new packet
265  NS_LOG_DEBUG ("Copied packet: " << m_retxParams.packet);
266  m_sentNewPacket (m_retxParams.packet);
267 
268  // static_cast<ClassAEndDeviceLorawanMac*>(this)->SendToPhy (m_retxParams.packet);
269  SendToPhy (m_retxParams.packet);
270  }
271  else
272  {
273  m_sentNewPacket (packet);
274  // static_cast<ClassAEndDeviceLorawanMac*>(this)->SendToPhy (packet);
275  SendToPhy (packet);
276  }
277 
278  }
279  // this is a retransmission
280  else
281  {
282  if (m_retxParams.waitingAck)
283  {
284 
285  m_currentFCnt++;
286 
287  // Remove the headers
288  LorawanMacHeader macHdr;
289  LoraFrameHeader frameHdr;
290  packet->RemoveHeader(macHdr);
291  packet->RemoveHeader(frameHdr);
292 
293  // Add the Lora Frame Header to the packet
294  frameHdr = LoraFrameHeader ();
295  ApplyNecessaryOptions (frameHdr);
296  packet->AddHeader (frameHdr);
297 
298  NS_LOG_INFO ("Added frame header of size " << frameHdr.GetSerializedSize () <<
299  " bytes.");
300 
301  // Add the Lorawan Mac header to the packet
302  macHdr = LorawanMacHeader ();
303  ApplyNecessaryOptions (macHdr);
304  packet->AddHeader (macHdr);
305  m_retxParams.retxLeft = m_retxParams.retxLeft - 1; // decreasing the number of retransmissions
306  NS_LOG_DEBUG ("Retransmitting an old packet.");
307 
308  // static_cast<ClassAEndDeviceLorawanMac*>(this)->SendToPhy (m_retxParams.packet);
309  SendToPhy (m_retxParams.packet);
310  }
311  }

331 void
332 EndDeviceLorawanMac::ParseCommands (LoraFrameHeader frameHeader)
333 {
334  NS_LOG_FUNCTION (this << frameHeader);
335 
336  if (m_retxParams.waitingAck)
337  {
338  if (frameHeader.GetAck ())
339  {
340  NS_LOG_INFO ("The message is an ACK, not waiting for it anymore.");
341 
342  NS_LOG_DEBUG ("Reset retransmission variables to default values and cancel retransmission if already scheduled.");
343 
344  uint8_t txs = m_maxNumbTx - (m_retxParams.retxLeft);
345  m_requiredTxCallback (txs, true, m_retxParams.firstAttempt, m_retxParams.packet);
346  NS_LOG_DEBUG ("Received ACK packet after " << unsigned(txs) << " transmissions: stopping retransmission procedure. ");
347 
348  // Reset retransmission parameters
349  resetRetransmissionParameters ();
350 
351  }
352  else
353  {
354  NS_LOG_ERROR ("Received downlink message not containing an ACK while we were waiting for it!");
355  }
356  }
357 
358  std::list<Ptr<MacCommand> > commands = frameHeader.GetCommands ();
359  std::list<Ptr<MacCommand> >::iterator it;
360  for (it = commands.begin (); it != commands.end (); it++)
361  {
362  NS_LOG_DEBUG ("Iterating over the MAC commands...");
363  enum MacCommandType type = (*it)->GetCommandType ();
364  switch (type)
365  {
366  case (LINK_CHECK_ANS):
367  {
368  NS_LOG_DEBUG ("Detected a LinkCheckAns command.");
369 
370  // Cast the command
371  Ptr<LinkCheckAns> linkCheckAns = (*it)->GetObject<LinkCheckAns> ();
372 
373  // Call the appropriate function to take action
374  OnLinkCheckAns (linkCheckAns->GetMargin (), linkCheckAns->GetGwCnt ());
375 
376  break;
377  }
378  case (LINK_ADR_REQ):
379  {
380  NS_LOG_DEBUG ("Detected a LinkAdrReq command.");
381 
382  // Cast the command
383  Ptr<LinkAdrReq> linkAdrReq = (*it)->GetObject<LinkAdrReq> ();
384 
385  // Call the appropriate function to take action
386  OnLinkAdrReq (linkAdrReq->GetDataRate (), linkAdrReq->GetTxPower (),
387  linkAdrReq->GetEnabledChannelsList (),
388  linkAdrReq->GetRepetitions ());
389 
390  break;
391  }
392  case (DUTY_CYCLE_REQ):
393  {
394  NS_LOG_DEBUG ("Detected a DutyCycleReq command.");
395 
396  // Cast the command
397  Ptr<DutyCycleReq> dutyCycleReq = (*it)->GetObject<DutyCycleReq> ();
398 
399  // Call the appropriate function to take action
400  OnDutyCycleReq (dutyCycleReq->GetMaximumAllowedDutyCycle ());
401 
402  break;
403  }
404  case (RX_PARAM_SETUP_REQ):
405  {
406  NS_LOG_DEBUG ("Detected a RxParamSetupReq command.");
407 
408  // Cast the command
409  Ptr<RxParamSetupReq> rxParamSetupReq = (*it)->GetObject<RxParamSetupReq> ();
410 
411  // Call the appropriate function to take action
412  OnRxParamSetupReq (rxParamSetupReq);
413 
414  break;
415  }
416  case (DEV_STATUS_REQ):
417  {
418  NS_LOG_DEBUG ("Detected a DevStatusReq command.");
419 
420  // Cast the command
421  Ptr<DevStatusReq> devStatusReq = (*it)->GetObject<DevStatusReq> ();
422 
423  // Call the appropriate function to take action
424  OnDevStatusReq ();
425 
426  break;
427  }
428  case (NEW_CHANNEL_REQ):
429  {
430  NS_LOG_DEBUG ("Detected a NewChannelReq command.");
431 
432  // Cast the command
433  Ptr<NewChannelReq> newChannelReq = (*it)->GetObject<NewChannelReq> ();
434 
435  // Call the appropriate function to take action
436  OnNewChannelReq (newChannelReq->GetChannelIndex (), newChannelReq->GetFrequency (), newChannelReq->GetMinDataRate (), newChannelReq->GetMaxDataRate ());
437 
438  break;
439  }
440  case (RX_TIMING_SETUP_REQ):
441  {
442  break;
443  }
444  case (TX_PARAM_SETUP_REQ):
445  {
446  break;
447  }
448  case (DL_CHANNEL_REQ):
449  {
450  break;
451  }
452  default:
453  {
454  NS_LOG_ERROR ("CID not recognized");
455  break;
456  }
457  }
458  }

521 Time
522 EndDeviceLorawanMac::GetNextTransmissionDelay (void)
523 {
524  NS_LOG_FUNCTION_NOARGS ();
525 
526  // Check duty cycle //
527 
528  // Pick a random channel to transmit on
529  std::vector<Ptr<LogicalLoraChannel> > logicalChannels;
530  logicalChannels = m_channelHelper.GetEnabledChannelList (); // Use a separate list to do the shuffle
531  //logicalChannels = Shuffle (logicalChannels);
532 
533 
534  Time waitingTime = Time::Max ();
535 
536  // Try every channel
537  std::vector<Ptr<LogicalLoraChannel> >::iterator it;
538  for (it = logicalChannels.begin (); it != logicalChannels.end (); ++it)
539  {
540  // Pointer to the current channel
541  Ptr<LogicalLoraChannel> logicalChannel = *it;
542  double frequency = logicalChannel->GetFrequency ();
543 
544  waitingTime = std::min (waitingTime, m_channelHelper.GetWaitingTime (logicalChannel));
545 
546  NS_LOG_DEBUG ("Waiting time before the next transmission in channel with frequecy " <<
547  frequency << " is = " << waitingTime.GetSeconds () << ".");
548  }
549 
550  waitingTime = GetNextClassTransmissionDelay (waitingTime);
551 
552  return waitingTime;
553 }

555 Ptr<LogicalLoraChannel>
556 EndDeviceLorawanMac::GetChannelForTx (void)
557 {
558  NS_LOG_FUNCTION_NOARGS ();
559 
560  // Pick a random channel to transmit on
561  std::vector<Ptr<LogicalLoraChannel> > logicalChannels;
562  logicalChannels = m_channelHelper.GetEnabledChannelList (); // Use a separate list to do the shuffle
563  logicalChannels = Shuffle (logicalChannels);
564 
565  // Try every channel
566  std::vector<Ptr<LogicalLoraChannel> >::iterator it;
567  for (it = logicalChannels.begin (); it != logicalChannels.end (); ++it)
568  {
569  // Pointer to the current channel
570  Ptr<LogicalLoraChannel> logicalChannel = *it;
571  double frequency = logicalChannel->GetFrequency ();
572 
573  NS_LOG_DEBUG ("Frequency of the current channel: " << frequency);
574 
575  // Verify that we can send the packet
576  Time waitingTime = m_channelHelper.GetWaitingTime (logicalChannel);
577 
578  NS_LOG_DEBUG ("Waiting time for current channel = " <<
579  waitingTime.GetSeconds ());
580 
581  // Send immediately if we can
582  if (waitingTime == Seconds (0))
583  {
584  return *it;
585  }
586  else
587  {
588  NS_LOG_DEBUG ("Packet cannot be immediately transmitted on " <<
589  "the current channel because of duty cycle limitations.");
590  }
591  }
592  return 0; // In this case, no suitable channel was found
593 }

596 std::vector<Ptr<LogicalLoraChannel> >
597 EndDeviceLorawanMac::Shuffle (std::vector<Ptr<LogicalLoraChannel> > vector)
598 {
599  NS_LOG_FUNCTION_NOARGS ();
600 
601  int size = vector.size ();
602 
603  for (int i = 0; i < size; ++i)
604  {
605  uint16_t random = std::floor (m_uniformRV->GetValue (0, size));
606  Ptr<LogicalLoraChannel> temp = vector.at (random);
607  vector.at (random) = vector.at (i);
608  vector.at (i) = temp;
609  }
610 
611  return vector;
612 }

699 void
700 EndDeviceLorawanMac::OnLinkAdrReq (uint8_t dataRate, uint8_t txPower,
701  std::list<int> enabledChannels, int repetitions)
702 {
703  NS_LOG_FUNCTION (this << unsigned (dataRate) << unsigned (txPower) <<
704  repetitions);
705 
706  // Three bools for three requirements before setting things up
707  bool channelMaskOk = true;
708  bool dataRateOk = true;
709  bool txPowerOk = true;
710 
711  // Check the channel mask
713  // Check whether all specified channels exist on this device
714  auto channelList = m_channelHelper.GetChannelList ();
715  int channelListSize = channelList.size ();
716 
717  for (auto it = enabledChannels.begin (); it != enabledChannels.end (); it++)
718  {
719  if ((*it) > channelListSize)
720  {
721  channelMaskOk = false;
722  break;
723  }
724  }
725 
726  // Check the dataRate
728  // We need to know we can use it at all
729  // To assess this, we try and convert it to a SF/BW combination and check if
730  // those values are valid. Since GetSfFromDataRate and
731  // GetBandwidthFromDataRate return 0 if the dataRate is not recognized, we
732  // can check against this.
733  uint8_t sf = GetSfFromDataRate (dataRate);
734  double bw = GetBandwidthFromDataRate (dataRate);
735  NS_LOG_DEBUG ("SF: " << unsigned (sf) << ", BW: " << bw);
736  if (sf == 0 || bw == 0)
737  {
738  dataRateOk = false;
739  NS_LOG_DEBUG ("Data rate non valid");
740  }
741 
742  // We need to know we can use it in at least one of the enabled channels
743  // Cycle through available channels, stop when at least one is enabled for the
744  // specified dataRate.
745  if (dataRateOk && channelMaskOk) // If false, skip the check
746  {
747  bool foundAvailableChannel = false;
748  for (auto it = enabledChannels.begin (); it != enabledChannels.end (); it++)
749  {
750  NS_LOG_DEBUG ("MinDR: " << unsigned (channelList.at (*it)->GetMinimumDataRate ()));
751  NS_LOG_DEBUG ("MaxDR: " << unsigned (channelList.at (*it)->GetMaximumDataRate ()));
752  if (channelList.at (*it)->GetMinimumDataRate () <= dataRate
753  && channelList.at (*it)->GetMaximumDataRate () >= dataRate)
754  {
755  foundAvailableChannel = true;
756  break;
757  }
758  }
759 
760  if (!foundAvailableChannel)
761  {
762  dataRateOk = false;
763  NS_LOG_DEBUG ("Available channel not found");
764  }
765  }
766 
767  // Check the txPower
769  // Check whether we can use this transmission power
770  if (GetDbmForTxPower (txPower) == 0)
771  {
772  txPowerOk = false;
773  }
774 
775  NS_LOG_DEBUG ("Finished checking. " <<
776  "ChannelMaskOk: " << channelMaskOk << ", " <<
777  "DataRateOk: " << dataRateOk << ", " <<
778  "txPowerOk: " << txPowerOk);
779 
780  // If all checks are successful, set parameters up
782  if (channelMaskOk && dataRateOk && txPowerOk)
783  {
784  // Cycle over all channels in the list
785  for (uint32_t i = 0; i < m_channelHelper.GetChannelList ().size (); i++)
786  {
787  if (std::find (enabledChannels.begin (), enabledChannels.end (), i) != enabledChannels.end ())
788  {
789  m_channelHelper.GetChannelList ().at (i)->SetEnabledForUplink ();
790  NS_LOG_DEBUG ("Channel " << i << " enabled");
791  }
792  else
793  {
794  m_channelHelper.GetChannelList ().at (i)->DisableForUplink ();
795  NS_LOG_DEBUG ("Channel " << i << " disabled");
796  }
797  }
798 
799  // Set the data rate
800  m_dataRate = dataRate;
801 
802  // Set the transmission power
803  m_txPower = GetDbmForTxPower (txPower);
804  }
805 
806  // Craft a LinkAdrAns MAC command as a response
808  m_macCommandList.push_back (CreateObject<LinkAdrAns> (txPowerOk, dataRateOk,
809  channelMaskOk));
810 }

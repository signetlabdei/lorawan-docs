115 // This collision matrix can be used for comparisons with the performance of Aloha
116 // systems, where collisions imply the loss of both packets.
117 double inf = std::numeric_limits<double>::max ();
118 double minf = std::numeric_limits<double>::min ();
119  std::vector<std::vector<double>> LoraInterferenceHelper::collisionSnirAloha= {
120  // 7 8 9 10 11 12
121  {inf, minf, minf, minf, minf, minf}, // SF7
122  {minf, inf, minf, minf, minf, minf}, // SF8
123  {minf, minf, inf, minf, minf, minf}, // SF9
124  {minf, minf, minf, inf, minf, minf}, // SF10
125  {minf, minf, minf, minf, inf, minf}, // SF11
126  {minf, minf, minf, minf, minf, inf} // SF12
127 };

129 // LoRa Collision Matrix (Goursaud)
130 // Values are inverted w.r.t. the paper since here we interpret this as an
131 // _isolation_ matrix instead of a cochannel _rejection_ matrix like in
132 // Goursaud's paper.
133  std::vector<std::vector<double>> LoraInterferenceHelper::collisionSnirGoursaud= {
134  // SF7 SF8 SF9 SF10 SF11 SF12
135  {6, -16, -18, -19, -19, -20}, // SF7
136  {-24, 6, -20, -22, -22, -22}, // SF8
137  {-27, -27, 6, -23, -25, -25}, // SF9
138  {-30, -30, -30, 6, -26, -28}, // SF10

140  {-36, -36, -36, -36, -36, 6} // SF12

248 uint8_t
249 LoraInterferenceHelper::IsDestroyedByInterference (Ptr<LoraInterferenceHelper::Event> event)
250 {
251  NS_LOG_FUNCTION (this << event);
252 
253  NS_LOG_INFO ("Current number of events in LoraInterferenceHelper: " << m_events.size ());
254 
255  // We want to see the interference affecting this event: cycle through events
256  // that overlap with this one and see whether it survives the interference or
257  // not.
258 
259  // Gather information about the event
260  double rxPowerDbm = event->GetRxPowerdBm ();
261  uint8_t sf = event->GetSpreadingFactor ();
262  double frequency = event->GetFrequency ();
263 
264  // Handy information about the time frame when the packet was received
265  Time now = Simulator::Now ();
266  Time duration = event->GetDuration ();
267  Time packetStartTime = now - duration;
268  Time packetEndTime = now;
269 
270  // Get the list of interfering events
271  std::list<Ptr<LoraInterferenceHelper::Event>>::iterator it;
272 
273  // Energy for interferers of various SFs
274  std::vector<double> cumulativeInterferenceEnergy (6, 0);
275 
276  // Cycle over the events
277  for (it = m_events.begin (); it != m_events.end ();)
278  {
279  // Pointer to the current interferer
280  Ptr<LoraInterferenceHelper::Event> interferer = *it;
281 
282  // Only consider the current event if the channel is the same: we
283  // assume there's no interchannel interference. Also skip the current
284  // event if it's the same that we want to analyze.
285  if (!(interferer->GetFrequency () == frequency) || interferer == event)
286  {
287  NS_LOG_DEBUG ("Different channel or same event");
288  it++;
289  continue; // Continues from the first line inside the for cycle
290  }
291 
292  NS_LOG_DEBUG ("Interferer on same channel");
293 
294  // Gather information about this interferer
295  uint8_t interfererSf = interferer->GetSpreadingFactor ();
296  double interfererPower = interferer->GetRxPowerdBm ();
297  Time interfererStartTime = interferer->GetStartTime ();
298  Time interfererEndTime = interferer->GetEndTime ();
299 
300  NS_LOG_INFO ("Found an interferer: sf = " << unsigned(interfererSf)
301  << ", power = " << interfererPower
302  << ", start time = " << interfererStartTime
303  << ", end time = " << interfererEndTime);
304 
305  // Compute the fraction of time the two events are overlapping
306  Time overlap = GetOverlapTime (event, interferer);
307 
308  NS_LOG_DEBUG ("The two events overlap for " << overlap.GetSeconds () << " s.");
309 
310  // Compute the equivalent energy of the interference
311  // Power [mW] = 10^(Power[dBm]/10)
312  // Power [W] = Power [mW] / 1000
313  double interfererPowerW = pow (10, interfererPower / 10) / 1000;
314  // Energy [J] = Time [s] * Power [W]
315  double interferenceEnergy = overlap.GetSeconds () * interfererPowerW;
316  cumulativeInterferenceEnergy.at (unsigned(interfererSf) - 7) += interferenceEnergy;
317  NS_LOG_DEBUG ("Interferer power in W: " << interfererPowerW);
318  NS_LOG_DEBUG ("Interference energy: " << interferenceEnergy);
319  it++;
320  }
321 
322  // For each SF, check if there was destructive interference
323  for (uint8_t currentSf = uint8_t (7); currentSf <= uint8_t (12); currentSf++)
324  {
325  NS_LOG_DEBUG ("Cumulative Interference Energy: "
326  << cumulativeInterferenceEnergy.at (unsigned(currentSf) - 7));
327 
328  // Use the computed cumulativeInterferenceEnergy to determine whether the
329  // interference with this SF destroys the packet
330  double signalPowerW = pow (10, rxPowerDbm / 10) / 1000;
331  double signalEnergy = duration.GetSeconds () * signalPowerW;
332  NS_LOG_DEBUG ("Signal power in W: " << signalPowerW);
333  NS_LOG_DEBUG ("Signal energy: " << signalEnergy);
334 
335  // Check whether the packet survives the interference of this SF
336  double snirIsolation = m_collisionSnir[unsigned(sf) - 7][unsigned(currentSf) - 7];
337  NS_LOG_DEBUG ("The needed isolation to survive is " << snirIsolation << " dB");
338  double snir =
339  10 * log10 (signalEnergy / cumulativeInterferenceEnergy.at (unsigned(currentSf) - 7));
340  NS_LOG_DEBUG ("The current SNIR is " << snir << " dB");
341 
342  if (snir >= snirIsolation)
343  {
344  // Move on and check the rest of the interferers
345  NS_LOG_DEBUG ("Packet survived interference with SF " << currentSf);
346  }
347  else
348  {
349  NS_LOG_DEBUG ("Packet destroyed by interference with SF" << unsigned(currentSf));
350 
351  return currentSf;
352  }
353  }
354  // If we get to here, it means that the packet survived all interference
355  NS_LOG_DEBUG ("Packet survived all interference");
356 
357  // Since the packet was not destroyed, we return 0.
358  return uint8_t (0);
359 }

369 Time
370 LoraInterferenceHelper::GetOverlapTime (Ptr<LoraInterferenceHelper::Event> event1,
371  Ptr<LoraInterferenceHelper::Event> event2)
372 {
373  NS_LOG_FUNCTION_NOARGS ();
374 
375  // Create the value we will return later
376  Time overlap;
377 
378  // Get handy values
379  Time s1 = event1->GetStartTime (); // Start times
380  Time s2 = event2->GetStartTime ();
381  Time e1 = event1->GetEndTime (); // End times
382  Time e2 = event2->GetEndTime ();
383 
384  // Non-overlapping events
385  if (e1 <= s2 || e2 <= s1)
386  {
387  overlap = Seconds (0);
388  }
389  // event1 before event2
390  else if (s1 < s2)
391  {
392  if (e2 < e1)
393  {
394  overlap = e2 - s2;
395  }
396  else
397  {
398  overlap = e1 - s2;
399  }
400  }
401  // event2 before event1 or they start at the same time (s1 = s2)
402  else
403  {
404  if (e1 < e2)
405  {
406  overlap = e1 - s1;
407  }
408  else
409  {
410  overlap = e2 - s1;
411  }
412  }
413 
414  return overlap;
415 }

417 } // namespace ns3
